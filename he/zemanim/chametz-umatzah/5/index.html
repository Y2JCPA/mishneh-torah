הלכות א׳–ה׳
הלכה א׳
אֵין אָסוּר מִשּׁוּם חָמֵץ בְּפֶסַח אֶלָּא חֲמֵשֶׁת מִינֵי דָּגָן בִּלְבַד. וְהֵם שְׁנֵי מִינֵי חִטִּים שֶׁהֵן הַחִטָּה וְהַכֻּסֶּמֶת. וּשְׁלֹשָׁה מִינֵי הַשְּׂעוֹרִים שֶׁהֵן הַשְּׂעוֹרָה וְשִׁבּלֶת שׁוּעָל וְהַשִּׁפּוֹן. אֲבָל קִטְנִיּוֹת כְּגוֹן אֹרֶז וְדֹחַן וּפוֹלִים וַעֲדָשִׁים וְכַיּוֹצֵא בָּהֶן אֵין בָּהֶן מִשּׁוּם חָמֵץ אֶלָּא אֲפִלּוּ לָשׁ קֶמַח אֹרֶז וְכַיּוֹצֵא בּוֹ בְּרוֹתְחִין וְכִסָּהוּ בִּבְגָדִים עַד שֶׁנִּתְפַּח כְּמוֹ בָּצֵק שֶׁהֶחֱמִיץ הֲרֵי זֶה מֻתָּר בַּאֲכִילָה שֶׁאֵין זֶה חִמּוּץ אֶלָּא סֵרָחוֹן:
הלכה ב׳
חֲמֵשֶׁת מִינֵי דָּגָן אֵלּוּ אִם לָשָׁן בְּמֵי פֵּרוֹת בִּלְבַד בְּלֹא שׁוּם מַיִם לְעוֹלָם אֵין בָּאִין לִידֵי חִמּוּץ אֶלָּא אֲפִלּוּ הִנִּיחָן כָּל הַיּוֹם עַד שֶׁנִּתְפַּח הַבָּצֵק מֻתָּר בַּאֲכִילָה [ג.] שֶׁאֵין מֵי פֵּרוֹת מַחְמִיצִין אֶלָּא מַסְרִיחִין. וּמֵי פֵּרוֹת הֵן כְּגוֹן יַיִן וְחָלָב וּדְבַשׁ וְשֶׁמֶן זַיִת וּמֵי תַּפּוּחִים וּמֵי רִמּוֹנִים וְכָל כַּיּוֹצֵא בָּהֶן מִשְּׁאָר יֵינוֹת וּשְׁמָנִים וּמַשְׁקִין. וְהוּא שֶׁלֹּא יִתְעָרֵב בָּהֶן שׁוּם מַיִם בָּעוֹלָם. וְאִם נִתְעָרֵב בָּהֶן מַיִם כָּל שֶׁהוּא הֲרֵי אֵלּוּ מַחְמִיצִין:
הלכה ג׳
אֵין מְבַשְּׁלִין חִטִּים בְּמַיִם כְּגוֹן רִיפוֹת וְלֹא קֶמַח כְּגוֹן לְבִיבוֹת. וְאִם בִּשֵּׁל הֲרֵי זֶה חָמֵץ גָּמוּר וְהוּא שֶׁיִּתְבַּקְּעוּ בַּתַּבְשִׁיל. אֵין קוֹלִין אֶת הַבָּצֵק בְּשֶׁמֶן עַל הַמַּחֲבַת. אֲבָל מְבַשְּׁלִין אֶת הַפַּת וְאֶת הַקֶּמַח הַקָּלוּי. וְאִם הִרְתִּיחַ הַמַּיִם הַרְבֵּה וְאַחַר כָּךְ הִשְׁלִיךְ לְתוֹכָן הַקֶּמַח הֲרֵי זֶה מֻתָּר מִפְּנֵי שֶׁהוּא מִתְבַּשֵּׁל מִיָּד קֹדֶם שֶׁיַּחְמִיץ. וּכְבָר נָהֲגוּ בְּשִׁנְעָר וּבִסְפָרַד וּבְכָל הַמַּעֲרָב לֶאֱסֹר דָּבָר זֶה גְּזֵרָה שֶׁמָּא לֹא יַרְתִּיחַ הַמַּיִם יָפֶה יָפֶה:
הלכה ד׳
מֻתָּר לְבַשֵּׁל הַדָּגָן אוֹ הַקֶּמַח בְּמֵי פֵּרוֹת. וְכֵן בָּצֵק שֶׁלָּשׁוֹ בְּמֵי פֵּרוֹת אִם בִּשְּׁלוֹ בְּמֵי פֵּרוֹת אוֹ קָלָהוּ עַל הַמַּחֲבַת בְּשֶׁמֶן הֲרֵי זֶה מֻתָּר שֶׁמֵּי פֵּרוֹת אֵינָן מַחְמִיצִין:
הלכה ה׳
כַּרְמֶל שֶׁמְּהַבְהֲבִין אוֹתוֹ בָּאוּר וְטוֹחֲנִין אוֹתוֹ אֵין מְבַשְּׁלִין אֶת הַקֶּמַח שֶׁלּוֹ בְּמַיִם שֶׁמָּא לֹא נִקְלָה בָּאוּר יָפֶה וְנִמְצָא מַחֲמִיץ כְּשֶׁמְּבַשְּׁלִין אוֹתוֹ. וְכֵן כְּשֶׁמּוֹלְלִין הַקְּדֵרוֹת הַחֲדָשׁוֹת אֵין מְבַשְּׁלִין בָּהֶן אֶלָּא מַצָּה אֲפוּיָה שֶׁחָזְרוּ וְטָחֲנוּ אוֹתָהּ אֲבָל קֶמַח קָלִי אָסוּר שֶׁמָּא לֹא קָלָהוּ יָפֶה וְיָבוֹא לִידֵי חִמּוּץ:
🌾 חמשת מיני הדגן וגורמי החימוץ
חמץ יכול להיווצר רק מחמשת מיני הדגן: חיטה, כוסמין, שעורה, שיבולת שועל ושיפון. קטניות (אורז, דוחן, קטניות) אינן יכולות להפוך לחמץ — הן רק מתקלקלות. מיץ פרי בלבד (יין, חלב, דבש, שמן, מיצי פירות) לעולם אינו גורם לתסיסה אמיתית; רק מים גורמים לכך. בישול קמח ישירות במים רותחים מותר מן הדין אך אסור על פי מנהג נפוץ. קמח מדגן קלוי לא יחמיץ, אך קמח שנקלה קלייה קלה עדיין חשוד.

טיפול בדגן, שמירה, ודגן פגום

הלכות ו׳–י׳
הלכה ו׳
אֵין בּוֹלְלִין אֶת הַשְּׂעוֹרִין בְּמַיִם בְּפֶסַח מִפְּנֵי שֶׁהֵן רָפִין וּמַחְמִיצִין בִּמְהֵרָה. וְאִם בָּלַל אִם רָפוּ כְּדֵי שֶׁאִם הִנִּיחָן עַל פִּי הַבִּיב שֶׁאוֹפִין עָלָיו הַחַלּוֹת יִתְבַּקְּעוּ הֲרֵי אֵלּוּ אֲסוּרִין. וְאִם לֹא הִגִּיעוּ לְרִפְיוֹן זֶה הֲרֵי אֵלּוּ מֻתָּרִין:
הלכה ז׳
הַחִטִּים מֻתָּר לִבְלל אוֹתָן בְּמַיִם כְּדֵי לְהָסִיר סֻבָּן וְטוֹחֲנִין אוֹתָן מִיָּד כְּדֶרֶךְ שֶׁטּוֹחֲנִין הַסּלֶת. וּכְבָר נָהֲגוּ כָּל יִשְׂרָאֵל בְּשִׁנְעָר וּבְאֶרֶץ הַצְּבִי וּבִסְפָרַד וּבְעָרֵי הַמַּעֲרָב שֶׁלֹּא יִבְלְלוּ הַחִטִּים בְּמַיִם גְּזֵרָה שֶׁמָּא יִשְׁהוּ וְיַחֲמִיצוּ:
הלכה ח׳
תַּבְשִׁיל שֶׁנִתְבַּשֵּׁל וְנִמְצְאוּ בּוֹ שְׂעוֹרִים אוֹ חִטִּים אִם נִתְבַּקְּעוּ הֲרֵי כָּל הַתַּבְשִׁיל אָסוּר שֶׁהֲרֵי נִתְעָרֵב בּוֹ הֶחָמֵץ. וְאִם לֹא נִתְבַּקְּעוּ מוֹצִיאִין אוֹתָן וְשׂוֹרְפָן וְאוֹכְלִין שְׁאָר הַתַּבְשִׁיל. שֶׁאֵין הַדָּגָן שֶׁנִּבְלַל וְלֹא נִתְבַּקֵּעַ חָמֵץ גָּמוּר שֶׁל תּוֹרָה. וְאֵינוֹ אֶלָּא מִדִּבְרֵי סוֹפְרִים:
הלכה ט׳
מִשּׁוּם שֶׁנֶּאֱמַר (שמות יב יז) "וּשְׁמַרְתֶּם אֶת הַמַּצּוֹת" כְּלוֹמַר הִזָּהֲרוּ בַּמַּצָּה וְשִׁמְרוּ אוֹתָהּ מִכָּל צַד חִמּוּץ. לְפִיכָךְ אָמְרוּ חֲכָמִים צָרִיךְ אָדָם לִזָּהֵר בְּדָגָן שֶׁאוֹכֵל מִמֶּנּוּ בַּפֶּסַח שֶׁלֹּא יָבוֹא עָלָיו מַיִם אַחַר שֶׁנִּקְצַר עַד שֶׁלֹּא יִהְיֶה בּוֹ שׁוּם חִמּוּץ. דָּגָן שֶׁנִּטְבַּע בַּנָּהָר אוֹ שֶׁנָּפַל עָלָיו מַיִם כְּשֵׁם שֶׁאָסוּר לְאָכְלוֹ כָּךְ אָסוּר לְקַיְּמוֹ אֶלָּא מוֹכְרוֹ לְיִשְׂרָאֵל וּמוֹדִיעוֹ כְּדֵי שֶׁיֹּאכְלֶנּוּ קֹדֶם הַפֶּסַח. וְאִם מְכָרוֹ לְנָכְרִי קֹדֶם הַפֶּסַח מוֹכֵר מְעַט לְכָל אֶחָד וְאֶחָד כְּדֵי שֶׁיִּכְלֶה קֹדֶם הַפֶּסַח שֶׁמָּא יַחֲזֹר הַנָּכְרִי וְיִמְכְּרֶנּוּ לְיִשְׂרָאֵל:
הלכה י׳
דָּגָן שֶׁנָּפַל עָלָיו דֶּלֶף כָּל זְמַן שֶׁהוּא טוֹרֵד טִפָּה אַחַר טִפָּה אֲפִלּוּ כָּל הַיּוֹם כֻּלּוֹ אֵינוֹ בָּא לִידֵי חִמּוּץ. אֲבָל כְּשֶׁיִּפָּסֵק אִם נִשְׁתַּהָה כַּשִּׁעוּר הֲרֵי זֶה אָסוּר:
👁️ הגנת הדגן ממגע עם מים
שעורה רכה ומחמיצה במהירות — אסור לערבבה במים; אם גרגרים מתרככים עד כדי פיצוץ בחום התנור, הם הופכים לחמץ. חיטה מותרת לשטיפה מן הדין, אך המנהג המקובל אוסר זאת לחלוטין. בתבשיל המכיל דגן: אם הגרגרים נסדקו, כל התבשיל אסור; אם לא, יש להסיר את הדגן ולשרפו, אך התבשיל מותר באכילה. הפסוק 'ושמרתם את המצות' מחייב שמירה מהקציר. דגן שהוצף בנהר יש למכור לפני פסח בגילוי מלא. טפטוף מתמיד של מים על דגן אינו גורם לחימוץ.

לישה, אפיית מצה, ומניעת חימוץ

הלכות י״א–כ׳
הלכה י״א׳
אֵין לָשִׁין בְּפֶסַח עִסָּה גְּדוֹלָה שֶׁמָּא תַּחֲמִיץ אֶלָּא כְּשִׁעוּר חַלָּה בִּלְבַד. וְאֵין לָשִׁין לֹא בְּחַמִּין וְלֹא בְּחַמֵּי חַמָּה. וְלֹא בְּמַיִם שֶׁנִּשְׁאֲבוּ בּוֹ בַּיּוֹם אֶלָּא בְּמַיִם שֶׁלָּנוּ. וְאִם עָבַר וְלָשׁ בְּאֶחָד מִכָּל אֵלּוּ הֲרֵי הַפַּת אֲסוּרָה:
הלכה י״ב׳
לֹא תֵּשֵׁב אִשָּׁה תַּחַת הַשֶּׁמֶשׁ וְתָלוּשׁ. וְלֹא תַּחַת הָרָקִיעַ בְּיוֹם הֶעָבִים אֲפִלּוּ בְּמָקוֹם שֶׁאֵין הַשֶּׁמֶשׁ זוֹרַחַת בּוֹ. וְלֹא תָּנִיחַ אֶת הָעִסָּה וְתִתְעַסֵּק בְּדָבָר אַחֵר. וְאִם הָיְתָה לָשָׁה וְאוֹפָה צְרִיכָה שְׁנֵי כֵּלִים שֶׁל מַיִם. אֶחָד שֶׁמְּקַטֶּפֶת בּוֹ וְאֶחָד שֶׁמְּצַנֶּנֶת בּוֹ יָדֶיהָ. וְאִם עָבְרָה וְלָשָׁה תַּחַת הַשֶּׁמֶשׁ אוֹ שֶׁלֹּא צִנְּנָה יָדֶיהָ אוֹ שֶׁעָשְׂתָה עִסָּה יֶתֶר מִשִּׁעוּר חַלָּה הַפַּת מֻתֶּרֶת. וְכַמָּה שִׁעוּר חַלָּה כְּמוֹ שָׁלֹשׁ וְאַרְבָּעִים בֵּיצִים וְחֹמֶשׁ בֵּיצָה כְּגוּף הַבֵּיצָה הַבֵּינוֹנִית לֹא כְּמִשְׁקָלָהּ:
הלכה י״ג׳

⏱️ מהירות, טמפרטורה, וכלל שמונה עשר הדקות
הבצק לא יעלה על שיעור חלה (~43 ⅕ ביצים בנפח). יש להשתמש רק במים שנחו לילה — לא חמים, שהו בשמש, או שנשאבו זה עתה. אישה לא תלוש בשמש או בצל עננים; היא זקוקה לשתי קערות מים. לישה מתמדת מונעת חימוץ ללא הגבלת זמן; בצק שהונח לנוח בזמן שנדרש ללכת מיל (~18 דקות), או שהשמיע קול חלול בטפיחה, הוא חמץ וייצרף. אם שתי עיסות נחו בו זמנית ואחת מוציאה קול חלול, שתיהן נשרפות. עיסות קישוט עבות אסורות לאנשים פרטיים. ניתן להוסיף תבלינים וזרעים לבצק; קמח בחרוסת ייצרף מיד. מצת הסדר חייבת להיות ממים בלבד — ללא מיץ פרי — כדי לזכור את 'לחם עוני'.

כלי חרס שנשתמש בו לחמץ בצונן מותר לשימוש רק לאוכל צונן של מצה — מלבד כלים לשמרים, חרוסת, או בצק מחומץ, שלעולם לא ייעשה בהם שימוש בפסח. תבנית צלייה מחרס דורשת ציפוי טיט עבה וצריפה מחדש. כלי מתכת ואבן שנשתמש בהם ככלי ראשון דורשים הגעלה — טבילה במים רותחים — לפי עיקרון 'כבולעו כך פולטו'. כלים ששימשו רק ככלי שני זקוקים רק למילוי במים רותחים. כלי חרס ששימש למשקאות חמץ חמים אינו ניתן להכשרה כלל. כלי גדול מדי לטבילה מוכשר על ידי שפיכת מים רותחים על שפתו.
7/8

🎓 עקרונות מרכזיים

🌾
רק חמשת הדגנים מייצרים חמץ
חמץ נוצר אך ורק מחיטה, כוסמין, שעורה, שיבולת שועל ושיפון. אורז, קטניות ושאר קטניות אינם יכולים להפוך לחמץ — הם רק מתקלקלים.
💧
המים הם הגורם המכריע
מיץ פרי לבדו לא גורם לתסיסה אמיתית — רק מים גורמים לכך. ברגע שמים מתערבבים עם מיץ פרי בבצק, התסיסה מתחילה מהר אף יותר מאשר עם מים לבדם.
⏱️
תנועה מתמדת מונעת חימוץ
בצק שבתנועה מתמדת לא יהפוך לחמץ ללא קשר לזמן. הסכנה מתחילה רק כשהבצק נח: הסימן הוא הזמן שנדרש ללכת מיל (~18 דקות) או הופעת קול חלול בטפיחה.
🔒
שמירה — עירנות על המצה
הפסוק 'ושמרתם את המצות' מחייב פיקוח פעיל על דגן המיועד למצה, ורצוי מהקציר. דגן שהורטב ללא פיקוח עלול להפוך את המצה לחמץ.
8/8
