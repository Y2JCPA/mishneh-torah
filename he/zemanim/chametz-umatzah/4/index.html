היקף האיסור: בעלות משפטית, לא מיקום פיזי

הלכות א׳–ב׳
הלכה א׳
כָּתוּב בַּתּוֹרָה (שמות יג ז) "לֹא יֵרָאֶה לְךָ חָמֵץ". יָכוֹל אִם טָמַן אוֹתוֹ אוֹ הִפְקִיד אוֹתוֹ בְּיַד עַכּוּ"ם לֹא יִהְיֶה עוֹבֵר תַּלְמוּד לוֹמַר (שמות יב יט) "שְׂאֹר לֹא יִמָּצֵא בְּבָתֵּיכֶם" אֲפִלּוּ הִפְקִידוֹ אוֹ הִטְמִינוֹ. יָכוֹל לֹא יִהְיֶה עוֹבֵר אֶלָּא אִם כֵּן הָיָה הֶחָמֵץ בְּבֵיתוֹ אֲבָל אִם הָיָה רָחוֹק מִבֵּיתוֹ בַּשָּׂדֶה אוֹ בְּעִיר אַחֶרֶת לֹא יִהְיֶה עוֹבֵר תַּלְמוּד לוֹמַר (שמות יג ז) (דברים טז ד) "בְּכָל גְּבֻלֶךָ" בְּכָל רְשׁוּתְךָ. יָכוֹל יְהֵא חַיָּב לְבַעֵר מֵרְשׁוּתוֹ חָמֵץ שֶׁל עַכּוּ"ם אוֹ שֶׁל הֶקְדֵּשׁ תַּלְמוּד לוֹמַר (שמות יג ז) (דברים טז ד) "לֹא יֵרָאֶה לְךָ" שֶׁלְּךָ אִי אַתָּה רוֹאֶה אֲבָל אַתָּה רוֹאֶה שֶׁל אֲחֵרִים וְשֶׁל גָּבוֹהַּ:
הלכה ב׳
הָא לָמַדְתָּ שֶׁהֶחָמֵץ שֶׁל יִשְׂרָאֵל אִם הֱנִיחוֹ בִּרְשׁוּתוֹ אֲפִלּוּ טָמוּן וַאֲפִלּוּ בְּעִיר אַחֶרֶת וַאֲפִלּוּ מֻפְקָד בְּיַד עַכּוּ"ם הֲרֵי זֶה עוֹבֵר מִשּׁוּם (שמות יג ז) (דברים טז ד) "לֹא יֵרָאֶה" וְ(שמות יב יט) "לֹא יִמָּצֵא". חָמֵץ שֶׁל הֶקְדֵּשׁ אוֹ שֶׁל עַכּוּ"ם שֶׁהָיָה אֵצֶל יִשְׂרָאֵל אֲפִלּוּ הָיָה עִמּוֹ בַּבַּיִת הֲרֵי זֶה מֻתָּר מִפְּנֵי שֶׁאֵינוֹ שֶׁלּוֹ. וַאֲפִלּוּ הָיָה שֶׁל גֵּר תּוֹשָׁב שֶׁיַּד יִשְׂרָאֵל שׁוֹלֶטֶת עָלָיו אֵין כּוֹפִין אוֹתוֹ לְהוֹצִיא הֶחָמֵץ מֵרְשׁוּתוֹ בְּפֶסַח. אֲבָל צָרִיךְ לַעֲשׂוֹת מְחִצָּה גְּבוֹהָה עֲשָׂרָה טְפָחִים בִּפְנֵי חֲמֵצוֹ שֶׁל עַכּוּ"ם שֶׁמָּא יָבוֹא לְהִסְתַּפֵּק מִמֶּנּוּ. אֲבָל שֶׁל הֶקְדֵּשׁ אֵינוֹ צָרִיךְ מִפְּנֵי שֶׁהַכּל פּוֹרְשִׁין מִן הַהֶקְדֵּשׁ כְּדֵי שֶׁלֹּא יָבוֹאוּ לִידֵי מְעִילָה:
📜 מקורות והיקף
הרמב״ם לומד משלושה פסוקים את היקף האיסור: ׳לא יראה׳ (שמות יג:ז), ׳לא ימצא׳ (שמות יב:יט), ו׳בכל גבולך׳. חמצו של יהודי אסור בכל מקום שהוא — מוסתר, בעיר אחרת, או מופקד אצל גוי. לעומת זאת, חמץ השייך לגוי או להקדש שנמצא ברשות יהודי מותר, כיוון שאינו שלו. חמץ של גוי צריך מחיצה גבוהה עשרה טפחים כדי למנוע שימוש בטעות, אבל חמץ של הקדש אינו צריך מחיצה — הכל נזהרים ממנו מפחד מעילה.

חמץ של גוי בפיקדון — אחריות שווה בעלות

הלכות ג׳–ה׳
הלכה ג׳
עַכּוּ"ם שֶׁהִפְקִיד חֲמֵצוֹ אֵצֶל יִשְׂרָאֵל אִם קִבֵּל עָלָיו יִשְׂרָאֵל אַחֲרָיוּת שֶׁאִם אָבַד אוֹ נִגְנַב יְשַׁלֵּם לוֹ דָּמָיו הֲרֵי זֶה חַיָּב לְבָעֲרוֹ הוֹאִיל וְקִבֵּל עָלָיו אַחֲרָיוּת נַעֲשָׂה כְּשֶׁלּוֹ. וְאִם לֹא קִבֵּל עָלָיו אַחֲרָיוּת מֻתָּר לְקַיְּמוֹ אֶצְלוֹ וּמֻתָּר לֶאֱכל מִמֶּנּוּ אַחַר הַפֶּסַח שֶׁבִּרְשׁוּת הָעַכּוּ"ם הוּא:
הלכה ד׳
עַכּוּ"ם אַנָּס שֶׁהִפְקִיד חֲמֵצוֹ אֵצֶל יִשְׂרָאֵל אִם יוֹדֵעַ הַיִּשְׂרָאֵל שֶׁאִם אָבַד אוֹ נִגְנַב מְחַיְּבוֹ לְשַׁלְּמוֹ וְכוֹפֵהוּ וְאוֹנְסוֹ לְשַׁלֵּם אַף עַל פִּי שֶׁלֹּא קִבֵּל אַחֲרָיוּת הֲרֵי זֶה חַיָּב לְבַעֲרוֹ. שֶׁהֲרֵי נֶחְשָׁב כְּאִלּוּ הוּא שֶׁלּוֹ מִפְּנֵי שֶׁמְּחַיְּבוֹ הָאַנָּס בְּאַחְרָיוּתוֹ:
הלכה ה׳
יִשְׂרָאֵל שֶׁהִרְהִין חֲמֵצוֹ אֵצֶל הָעַכּוּ"ם אִם אָמַר לוֹ אִם לֹא הֵבֵאתִי לְךָ מָעוֹת מִכָּאן וְעַד יוֹם פְּלוֹנִי קְנֵה חָמֵץ זֶה מֵעַכְשָׁו הֲרֵי זֶה בִּרְשׁוּת הָעַכּוּ"ם וְאוֹתוֹ הֶחָמֵץ מֻתָּר לְאַחַר הַפֶּסַח. וְהוּא שֶׁיִּהְיֶה אוֹתוֹ זְמַן שֶׁקָּבַע לוֹ קֹדֶם הַפֶּסַח. וְאִם לֹא אָמַר לוֹ קְנֵה מֵעַכְשָׁו נִמְצָא אוֹתוֹ הֶחָמֵץ כְּאִלּוּ הוּא פִּקָּדוֹן אֵצֶל הָעַכּוּ"ם וְאָסוּר בַּהֲנָיָה לְאַחַר הַפֶּסַח:
⚖️ אחריות = בעלות
גוי שהפקיד חמץ אצל יהודי: אם קיבל היהודי אחריות לשלם אם יאבד או יגנב — חייב לבערו, שכן הוא נחשב שלו. ללא אחריות — רשאי להשאירו ולאכלו אחרי פסח. אפילו גוי קשה שכופה את היהודי לשלם — אף שלא קיבל אחריות מפורשת — חייב היהודי לבערו. יהודי שמישכן חמצו לגוי: אם אמר ׳קנה מעכשיו׳ עם תאריך לפני פסח — החמץ שייך לגוי ומותר אחרי פסח; ללא ׳קנה מעכשיו׳ — נחשב כפיקדון ואסור בהנאה אחרי פסח.

מכירה ומתנה לגוי — העברה מוחלטת בלבד

הלכות ו׳–ז׳
הלכה ו׳
יִשְׂרָאֵל וְעַכּוּ"ם שֶׁהָיוּ בָּאִין בִּסְפִינָה וְהָיָה חָמֵץ בְּיַד יִשְׂרָאֵל וְהִגִּיעָה שָׁעָה חֲמִישִׁית הֲרֵי זֶה מוֹכְרוֹ לְעַכּוּ"ם אוֹ נוֹתְנוֹ לוֹ בְּמַתָּנָה וְחוֹזֵר וְלוֹקְחוֹ מִמֶּנּוּ אַחַר הַפֶּסַח. וּבִלְבַד שֶׁיִּתְּנֶנּוּ לוֹ מַתָּנָה גְּמוּרָה:
הלכה ז׳
אוֹמֵר יִשְׂרָאֵל לְעַכּוּ"ם עַד שֶׁאַתָּה לוֹקֵחַ בְּמָנֶה בּוֹא וְקַח בְּמָאתַיִם. עַד שֶׁאַתָּה לוֹקֵחַ מֵעַכּוּ"ם בּוֹא וְקַח מִיִּשְׂרָאֵל שֶׁמָּא אֶצְטָרֵךְ וְאֶקַּח מִמְּךָ אַחַר הַפֶּסַח. אֲבָל לֹא יִמְכֹּר וְלֹא יִתֵּן לוֹ עַל תְּנַאי. וְאִם עָשָׂה כֵּן הֲרֵי זֶה עוֹבֵר עַל בַּל יֵרָאֶה וּבַל יִמָּצֵא:
🤝 העברה מלאה בלבד
יהודי וגוי הנוסעים בספינה: כשמגיעה השעה החמישית — מכור לגוי או תן לו במתנה גמורה, ומותר לקנות חזרה אחרי פסח. מותר לרמוז לגוי שאולי ירצה להחזיר לאחר פסח, ואפילו להבטיח לו רווח. אבל מכירה או מתנה על תנאי אסורה — אם עשה כן, עובר היהודי על בל יראה ובל ימצא, שכן עד שהתנאי יתקיים הוא נשאר הבעלים החוקיים.

תערובות חמץ וחמץ מקולקל

הלכות ח׳–י״ב
הלכה ח׳
תַּעֲרֹבֶת חָמֵץ עוֹבְרִין עָלֶיהָ מִשּׁוּם בַּל יֵרָאֶה וּבַל יִמָּצֵא. כְּגוֹן הַמּוּרְיָס וְכוּתָח הַבַּבְלִי וְשֵׁכָר הַמָּדִי שֶׁעוֹשִׂין אוֹתוֹ מִן הַקֶּמַח. וְכָל כַּיּוֹצֵא בָּאֵלּוּ מִדְּבָרִים הַנֶּאֱכָלִים. אֲבָל דָּבָר שֶׁיֵּשׁ בּוֹ תַּעֲרֹבֶת חָמֵץ וְאֵינוֹ רָאוּי לַאֲכִילָה הֲרֵי זֶה מֻתָּר לְקַיְּמוֹ בְּפֶסַח:
הלכה ט׳
כֵּיצַד. עֲרֵבַת הָעַבְּדָנִין שֶׁנָּתַן לְתוֹכָהּ קֶמַח וְעוֹרוֹת אֲפִלּוּ נְתָנוֹ שָׁעָה אַחַת קֹדֶם הַבִּעוּר הֲרֵי זֶה מֻתָּר לְקַיְּמוֹ. וְאִם לֹא נָתַן הָעוֹרוֹת וְנָתַן הַקֶּמַח קֹדֶם שְׁלֹשָׁה יָמִים לִשְׁעַת הַבִּעוּר מֻתָּר לְקַיְּמוֹ שֶׁהֲרֵי נִפְסַד וְהִבְאִישׁ. תּוֹךְ שְׁלֹשָׁה יָמִים חַיָּב לְבָעֵר:
הלכה י׳
וְכֵן הַקִּילוֹר וְהָרְטִיָּה וְהָאִסְפְּלָנִית וְהַתְּרִיאַ"ק שֶׁנָּתַן לְתוֹכָן חָמֵץ מֻתָּר לְקַיְּמָן בְּפֶסַח שֶׁהֲרֵי נִפְסְדָה צוּרַת הֶחָמֵץ:
הלכה י״א׳
הַפַּת עַצְמָהּ שֶׁעִפְּשָׁה וְנִפְסְלָה מִלֶּאֱכל הַכֶּלֶב ומְלוּגְמָא שֶׁנִּסְרְחָה אֵינוֹ צָרִיךְ לְבָעֵר. בְּגָדִים שֶׁכִּבְּסוּ אוֹתָן בְּחֵלֶב חִטָּה וְכֵן נְיָרוֹת שֶׁדִּבְּקוּ אוֹתָן בְּחָמֵץ וְכָל כַּיּוֹצֵא בָּזֶה מֻתָּר לְקַיְּמָן בְּפֶסַח וְאֵין בָּהֶן מִשּׁוּם (שמות יג ז) (דברים טז ד) "לֹא יֵרָאֶה" וְ(שמות יב יט) "לֹא יִמָּצֵא" שֶׁאֵין צוּרַת הֶחָמֵץ עוֹמֶדֶת:
הלכה י״ב׳
דָּבָר שֶׁנִּתְעָרֵב בּוֹ חָמֵץ וְאֵינוֹ מַאֲכָל לְאָדָם כְּלָל. אוֹ שֶׁאֵינוֹ מַאֲכַל כָּל אָדָם כְּגוֹן הַתְּרִיאַ"ק וְכַיּוֹצֵא בּוֹ אַף עַל פִּי שֶׁמֻּתָּר לְקַיְּמוֹ אָסוּר לְאָכְלוֹ עַד אַחַר הַפֶּסַח. וְאַף עַל פִּי שֶׁאֵין בּוֹ מִן הֶחָמֵץ אֶלָּא כָּל שֶׁהוּא הֲרֵי זֶה אָסוּר לְאָכְלוֹ:
🧪 הצורה קובעת את המעמד
תערובת חמץ הראויה לאכילה — עובר על בל יראה ובל ימצא (ציר דגים, כותח הבבלי, שכר המדי). תערובת שאינה ראויה לאכילה — מותר להשאיר. אבוס של בורסקי עם קמח ועורות: מותר אפילו שעה אחת לפני זמן הביעור; קמח ללא עורות — בתוך שלושה ימים חייב לבערו, לפני שלושה ימים מותר כי כבר הסריח. משחות, תחבושות, רטיות ותרופות — מותר להשאיר. לחם שעלה בו עובש מעבר לרמה הראויה לכלב, בגדים שכובסו בחלב חיטה, ניירות שהודבקו בחמץ — מותר להשאיר. כל דבר שאינו מאכל המכיל חמץ — מותר להשאיר אבל אסור לאכול עד אחרי פסח, אפילו בכמות הקטנה ביותר.

בעלות משפטית, לא מיקום פיזי
איסורי בל יראה ובל ימצא תלויים בבעלות המשפטית על החמץ, ולא במיקומו הפיזי — אפילו חמץ מוסתר, בעיר אחרת, או מופקד אצל גוי אסור.
⚖️
אחריות כספית יוצרת בעלות
קבלת אחריות לשלם על חמץ (בין מרצון ובין מכפייה של גוי) הופכת את החמץ לרכושו של היהודי לצורכי הלכה, ומחייבת ביעור.
🤝
העברת הבעלות חייבת להיות מוחלטת
מכירה או מתנה של חמץ לגוי תקפה רק אם היא מלאה וללא תנאים. מכירה על תנאי אינה מועילה — היהודי נשאר הבעלים ועובר על האיסור.
🧪
צורת החמץ קובעת את מעמדו
חמץ שאיבד את צורתו כמאכל (הסריח, שימש לעיבוד עורות, נתערב בתרופות) אינו נחשב עוד כחמץ לעניין האיסור. אבל אם התערובת עדיין ראויה לאכילה, אפילו כמות קטנה ביותר מפעילה את האיסור.
