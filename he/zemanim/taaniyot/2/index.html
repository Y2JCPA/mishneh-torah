הרשימה והגאוגרפיה של החובה

הלכות א׳–ב׳
הלכה א׳
אֵלוּ הֵן הַצָּרוֹת שֶׁל צִבּוּר שֶׁמִּתְעַנִּין וּמַתְרִיעִין עֲלֵיהֶם. עַל הֲצָרַת שׂוֹנְאֵי יִשְׂרָאֵל לְיִשְׂרָאֵל. וְעַל הַחֶרֶב. וְעַל הַדֶּבֶר. וְעַל חַיָּה רָעָה. וְעַל הָאַרְבֶּה. וְעַל הֶחָסִיל. וְעַל הַשִּׁדָּפוֹן. וְעַל הַיֵּרָקוֹן. וְעַל הַמַּפּלֶת. וְעַל הֶחֳלָאִים. וְעַל הַמְּזוֹנוֹת. וְעַל הַמָּטָר:
הלכה ב׳
כָּל עִיר שֶׁיֵּשׁ בָּהּ צָרָה מִכָּל אֵלּוּ. אוֹתָהּ הָעִיר מִתְעַנָּה וּמַתְרַעַת עַד שֶׁתַּעֲבֹר הַצָּרָה. וְכָל סְבִיבוֹתֶיהָ מִתְעַנּוֹת וְלֹא מַתְרִיעוֹת אֲבָל מְבַקְּשִׁים עֲלֵיהֶם רַחֲמִים. וּבְכָל מָקוֹם אֵין צוֹעֲקִין וְלֹא מַתְרִיעִין בְּשַׁבָּת כְּמוֹ שֶׁאָמַרְנוּ. חוּץ מִצָּרַת הַמְּזוֹנוֹת שֶׁצּוֹעֲקִין עָלֶיהָ אֲפִלּוּ בְּשַׁבָּת. אֲבָל אֵין מַתְרִיעִין עָלֶיהָ בַּחֲצוֹצְרוֹת בְּשַׁבָּת:
📋 עשרה גורמים, שתי רמות
הלכה א׳ פותחת ברשימה מקיפה של עשר צרות — מדיכוי אויב וצבא עובר ועד דֶּבֶר, חיות רעות, ארבה, שדפון, בתים קורסים, מגפה, קריסה כלכלית וגשמים — שמחייבות צום ציבורי ותקיעת חצוצרות. הלכה ב׳ קובעת מדרג גאוגרפי: העיר הנפגעת צמה וגם תוקעת, בעוד הערים הסמוכות צמות אך שותקות — פרט למקרה הייחודי של אובדן פרנסה, הצרה היחידה שמתירה לזעוק לאלוהים אפילו בשבת.
4/9

חרבות, גזרות וסולידריות בפני המגפה

הלכות ג׳–ו׳
הלכה ג׳
עַל הֲצָרַת שׂוֹנְאֵי יִשְׂרָאֵל לְיִשְׂרָאֵל כֵּיצַד. עַכּוּ"ם שֶׁבָּאוּ לַעֲרֹךְ מִלְחָמָה עִם יִשְׂרָאֵל אוֹ לִטּל מֵהֶם מַס אוֹ לִקַּח מִיָּדָם אֶרֶץ אוֹ לִגְזֹר עֲלֵיהֶם גְּזֵרָה אֲפִלּוּ בְּמִצְוָה קַלָּה הֲרֵי אֵלּוּ מִתְעַנִּין וּמַתְרִיעִין עַד שֶׁיְּרֻחֲמוּ. וְכָל הֶעָרִים שֶׁסְּבִיבוֹתֵיהֶם מִתְעַנִּין אֲבָל אֵין מַתְרִיעִין אֶלָּא אִם כֵּן תָּקְעוּ לְהִתְקַבֵּץ לְעֶזְרָתָן:
הלכה ד׳
עַל הַחֶרֶב כֵּיצַד. אֲפִלּוּ חֶרֶב שֶׁל שָׁלוֹם כְּגוֹן שֶׁעָרְכוּ מִלְחָמָה עַכּוּ"ם עִם עַכּוּ"ם וְעָבְרוּ בִּמְקוֹם יִשְׂרָאֵל אַף עַל פִּי שֶׁאֵין בֵּינֵיהֶם וּבֵין יִשְׂרָאֵל מִלְחָמָה הֲרֵי זוֹ צָרָה וּמִתְעַנִּין עָלֶיהָ שֶׁנֶּאֱמַר (ויקרא כו ו) "וְחֶרֶב לֹא תַעֲבֹר בְּאַרְצְכֶם" מִכְּלַל שֶׁרְאִיַּת הַמִּלְחָמָה צָרָה:
הלכה ה׳
וְעַל הַדֶּבֶר. אֵי זוֹ הִיא דֶּבֶר עִיר שֶׁיֵּשׁ בָּהּ חֲמֵשׁ מֵאוֹת רַגְלִי וְיָצְאוּ מִמֶּנָּה שְׁלֹשָׁה מֵתִים בִּשְׁלֹשָׁה יָמִים זֶה אַחַר זֶה הֲרֵי זֶה דֶּבֶר. יָצְאוּ בְּיוֹם אֶחָד אוֹ בְּאַרְבָּעָה יָמִים אֵין זֶה דֶּבֶר. הָיוּ בָּהּ אֶלֶף וְיָצְאוּ מִמֶּנָּה שִׁשָּׁה מֵתִים בִּשְׁלֹשָׁה יָמִים זֶה אַחַר זֶה דֶּבֶר. יָצְאוּ בְּיוֹם אֶחָד אוֹ בְּאַרְבָּעָה אֵין זֶה דֶּבֶר. וְכֵן לְפִי חֶשְׁבּוֹן זֶה. וְאֵין הַנָּשִׁים וְהַקְּטַנִּים וְהַזְּקֵנִים שֶׁשָּׁבְתוּ מִמְּלָאכָה בִּכְלַל מִנְיַן אַנְשֵׁי הַמְּדִינָה לְעִנְיָן זֶה:
הלכה ו׳
הָיָה דֶּבֶר בְּאֶרֶץ יִשְׂרָאֵל מִתְעַנִּין שְׁאָר גָּלֻיּוֹת יִשְׂרָאֵל עֲלֵיהֶן. הָיָה דֶּבֶר בַּמְּדִינָה וְשַׁיָּרוֹת הוֹלְכוֹת וּבָאוֹת מִמֶּנָּה לִמְדִינָה אַחֶרֶת שְׁתֵּיהֶן מִתְעַנּוֹת אַף עַל פִּי שֶׁהֵן רְחוֹקוֹת זוֹ מִזּוֹ:
⚔️ איומים אנושיים ומחלות מדבקות
הלכות ג׳–ד׳ עוסקות בצרות הנגרמות על ידי אויב: מלחמה, מיסים לא צודקים, גזל קרקעות, ואפילו גזרה דתית קלה מצריכות צום, וכן עצם מעבר של כל כוח מזוין — אפילו כוח שלווה העובר ביישובים יהודיים, משום שהתורה מתארת ארץ נטולת צבא כברכה. הלכות ה׳–ו׳ מגדירות דֶּבֶר בדיוק מתמטי (3 מתים במשך 3 ימים רצופים על כל 500 תושבים זכרים, ביחס פרופורציונלי) ומחייבות כי יהודי הגולה יצומו על דֶּבֶר בארץ ישראל, ובנוסף — ערים המחוברות בדרכי שיירות לאזור נגוע צמות יחד, ללא תלות במרחק.
5/9

שיטוט, זוחלים, ארבה ושדות מוכות

הלכות ז׳–י״ב
הלכה ז׳
אֵין מִתְעַנִּין עַל חַיָּה רָעָה אֶלָּא בִּזְמַן שֶׁהִיא מְשֻׁלַּחַת. כֵּיצַד. נִרְאֲתָה בָּעִיר בַּיּוֹם הֲרֵי זוֹ מְשֻׁלַּחַת. נִרְאֲתָה בַּשָּׂדֶה בַּיּוֹם אִם רָאֲתָה שְׁנֵי בְּנֵי אָדָם וְלֹא בָּרְחָה מִפְּנֵיהֶם הֲרֵי זוֹ מְשֻׁלַּחַת. וְאִם הָיָה שָׂדֶה הַסְּמוּכָה לָאֲגַם וְרָאֲתָה שְׁנֵי בְּנֵי אָדָם וְרָדְפָה אַחֲרֵיהֶם הֲרֵי זוֹ מְשֻׁלַּחַת. לֹא רָדְפָה אַחֲרֵיהֶם אֵינָהּ מְשֻׁלַּחַת. וְאִם הָיָה בָּאֲגַם אֲפִלּוּ רָדְפָה אַחֲרֵיהֶם אֵינָהּ מְשֻׁלַּחַת אֶלָּא אִם כֵּן טָרְפָה שְׁנֵיהֶם וְאָכְלָה אֶחָד מֵהֶן. אֲבָל אִם אָכְלָה שְׁנֵיהֶם בָּאֲגַם אֵינָהּ מְשֻׁלַּחַת מִפְּנֵי שֶׁזֶּה הוּא מְקוֹמָהּ וּמִפְּנֵי הָרְעָבוֹן טָרְפָה לֹא מִפְּנֵי שֶׁהִיא מְשֻׁלַּחַת:
הלכה ח׳
בָּתִּים הַבְּנוּיִים בְּמִדְבָּרוֹת וּבָאֲרָצוֹת הַנְשַׁמּוֹת הוֹאִיל וְהֵן מְקוֹם גְּדוּדֵי חַיָּה. אִם עָלְתָה לַגַּג וְנָטְלָה תִּינוֹק מֵעֲרִיסָה הֲרֵי זוֹ מְשֻׁלַּחַת. וְאִם לֹא הִגִּיעָה לְמִדָּה זוֹ אֵינָהּ מְשֻׁלַּחַת. שֶׁאֵלּוּ בְּנֵי אָדָם הֵם שֶׁסִּכְּנוּ בְּעַצְמָם וּבָאוּ לִמְקוֹם הַחַיּוֹת:
הלכה ט׳
שְׁאָר מִינֵי רֶמֶשׂ הָאָרֶץ וְרֶמֶשׂ הָעוֹף שֶׁשֻּׁלְּחוּ וְהִזִּיקוּ. כְּגוֹן שִׁלּוּחַ נְחָשִׁים וְעַקְרַבִּים וְאֵין צָרִיךְ לוֹמַר צְרָעִין וְיַתּוּשִׁין וְהַדּוֹמֶה לָהֶן אֵין מִתְעַנִּין עֲלֵיהֶם וְלֹא מַתְרִיעִין אֲבָל זוֹעֲקִין בְּלֹא תְּרוּעָה:
הלכה י׳
עַל הָאַרְבֶּה וְעַל הֶחָסִיל אֲפִלּוּ לֹא נִרְאֶה מֵהֶן אֶלָּא כָּנָף אֶחָד בְּכָל אֶרֶץ יִשְׂרָאֵל מִתְעַנִּין וּמַתְרִיעִין עֲלֵיהֶן. וְעַל הַגּוֹבַאי בְּכָל שֶׁהוּא. אֲבָל עַל הֶחָגָב אֵין מִתְעַנִּין עָלָיו וְלֹא מַתְרִיעִין אֶלָּא זוֹעֲקִין בִּלְבַד:
הלכה י״א׳
עַל הַשִּׁדָּפוֹן וְעַל הַיֵּרָקוֹן מִשֶּׁיַּתְחִיל בַּתְּבוּאָה. וַאֲפִלּוּ לֹא הִתְחִיל אֶלָּא מִמָּקוֹם קָטָן כִּמְלֹא פִּי תַּנּוּר גּוֹזְרִין עָלָיו תַּעֲנִית וּמַתְרִיעִין:
הלכה י״ב׳
עַל הַמַּפּלֶת כֵּיצַד. הֲרֵי שֶׁרָבְתָה בָּעִיר מַפּלֶת כְּתָלִים בְּרִיאִים שֶׁאֵינָן עוֹמְדִים בְּצַד הַנָּהָר הֲרֵי זוֹ צָרָה וּמִתְעַנִּין וּמַתְרִיעִין עָלֶיהָ. וְכֵן עַל הָרַעַשׁ וְעַל הָרוּחוֹת שֶׁהֵן מַפִּילִין אֶת הַבִּנְיָן וְהוֹרְגִין מִתְעַנִּין וּמַתְרִיעִין עֲלֵיהֶן:
🦁 חיות בר, חרקים ויבולים
שש הלכות מציבות מבחנים מדויקים לפי הקשר: חיה רעה נחשבת "משוטטת" רק אם נכנסת לעיר ביום, אינה בורחת מבני אדם בשדה, או — בביצה — הרגה שני אנשים אך אכלה רק אחד (הלכות ז׳–ח׳); זוחלים וחרקים כנחשים, עקרבים ויתושים — רק תפילה בלחש, לעולם לא חצוצרות (הלכה ט׳); ארבה (ארבה/חסיל/גובאי) מחייב צום מלא מהרגע הראשון שנראה, בעוד מין החגב הקטן אינו מחייב (הלכה י׳); שידפון שחור או צהוב הפוגע אפילו בשטח קטן כפתח תנור מחייב צום מיידי (הלכה י״א), וכן בתים קורסים, רעידות אדמה ורוחות הרסניות (הלכה י״ב).
6/9

מגפה, פצעים מוגלתיים וקריסה כלכלית

הלכות י״ג–י״ד
הלכה י״ג׳
עַל הֶחֳלָיִים כֵּיצַד. הֲרֵי שֶׁיָּרַד חלִי אֶחָד לַאֲנָשִׁים הַרְבֵּה בְּאוֹתָהּ הָעִיר כְּגוֹן אַסְכָּרָה אוֹ חַרְחוּר וְכַיּוֹצֵא בָּהֶן וְהָיוּ מֵתִים בְּאוֹתוֹ הַחלִי הֲרֵי זוֹ צָרַת צִבּוּר וְגוֹזְרִין לָהּ תַּעֲנִית וּמַתְרִיעִין. וְכֵן חִכּוּךְ לַח הֲרֵי הוּא כִּשְׁחִין פּוֹרֵחַ. וְאִם פָּשַׁט בְּרֹב הַצִּבּוּר מִתְעַנִּין וּמַתְרִיעִין עָלָיו. אֲבָל חִכּוּךְ יָבֵשׁ צוֹעֲקִים עָלָיו בִּלְבַד:
הלכה י״ד׳
עַל הַמְּזוֹנוֹת כֵּיצַד. הֲרֵי שֶׁהוּזְלוּ דְּבָרִים שֶׁל סְחוֹרָה שֶׁרֹב חַיֵּי אַנְשֵׁי אוֹתָהּ הָעִיר מֵהֶן. כְּגוֹן כְּלֵי פִּשְׁתָּן בְּבָבֶל וְיַיִן וְשֶׁמֶן בְּאֶרֶץ יִשְׂרָאֵל. וְנִתְמַעֵט הַמַּשָּׂא וְהַמַּתָּן עַד שֶׁיִּצְטָרֵךְ הַתַּגָּר לִמְכֹּר שְׁוִה עֲשָׂרָה בְּשִׁשָּׁה וְאַחַר כָּךְ יִמָּצֵא לוֹקֵחַ. הֲרֵי זוֹ צָרַת צִבּוּר וּמַתְרִיעִין עָלֶיהָ וְזוֹעֲקִין עָלֶיהָ בְּשַׁבָּת:
🏥 מחלה ופרנסה
הלכה י״ג מבחינה בין שלוש רמות של משברי עור ומחלה: דלקות גרון או שיתוק ילדים הגורמים למוות ציבורי מחייבים צום וחצוצרות; פצעים מוגלתיים (גרב לח) שמתפשטים בקרב הרוב מטופלים כשחין — צום ותקיעה; אך גרב יבש מצריך רק תפילה בלחש. הלכה י״ד מגדירה מצוקה כלכלית כקריסת מחיר הסחורות העיקריות של העיר ל-60% משוויין האמיתי — למשל פשתן בבבל או יין ושמן בארץ ישראל — מה שמחייב זעקה ציבורית אפילו בשבת, המשבר הלא-קשור-לסכנת-נפשות היחיד שזוכה לפריבילגיה הזו.
7/9

גשם: כשהשפע הורס והיעדרו הורג

הלכות ט״ו–י״ח
הלכה ט״ו׳
עַל הַמָּטָר כֵּיצַד. הֲרֵי שֶׁרַבּוּ עֲלֵיהֶן גְּשָׁמִים עַד שֶׁיָּצֵרוּ לָהֶן הֲרֵי אֵלּוּ מִתְפַּלְּלִין עֲלֵיהֶן. שֶׁאֵין לְךָ צָרָה יְתֵרָה מִזּוֹ שֶׁהַבָּתִּים נוֹפְלִין וְנִמְצְאוּ בָּתֵּיהֶן קִבְרֵיהֶן. וּבְאֶרֶץ יִשְׂרָאֵל אֵין מִתְפַּלְּלִין עַל רֹב הַגְּשָׁמִים מִפְּנֵי שֶׁהִיא אֶרֶץ הָרִים וּבָתֵּיהֶם בְּנוּיִים בַּאֲבָנִים וְרֹב הַגְּשָׁמִים טוֹבָה לָהֶן וְאֵין מִתְעַנִּין לְהַעֲבִיר הַטּוֹבָה:
הלכה ט״ז׳
תְּבוּאָה שֶׁצָּמְחָה וְנִמְנַע הַמָּטָר וְהִתְחִילוּ צְמָחִין לִיבַשׁ הֲרֵי אֵלּוּ מִתְעַנִּין וְזוֹעֲקִין עַד שֶׁיֵּרְדוּ גְּשָׁמִים אוֹ עַד שֶׁיִּיבְשׁוּ הַצְּמָחִים. וְכֵן אִם הִגִּיעַ זְמַן הַפֶּסַח אוֹ קָרוֹב לוֹ בְּאֶרֶץ יִשְׂרָאֵל שֶׁהוּא זְמַן פְּרִיחַת הָאִילָנוֹת שָׁם וְלֹא יָרְדוּ גְּשָׁמִים הֲרֵי אֵלּוּ מִתְעַנִּין וְזוֹעֲקִין עַד שֶׁיֵּרְדוּ גְּשָׁמִים הָרְאוּיִין לְאִילָנוֹת אוֹ עַד שֶׁיַּעֲבֹר זְמַנָּם:
הלכה י״ז׳

הלכה י״ח׳
פָּסְקוּ הַגְּשָׁמִים בֵּין גֶּשֶׁם לְגֶשֶׁם אַרְבָּעִים יוֹם בִּימוֹת הַגְּשָׁמִים הֲרֵי זוֹ מַכַּת בַּצֹּרֶת וּמִתְעַנִּים וְזוֹעֲקִים עַד שֶׁיֵּרְדוּ הַגְּשָׁמִים אוֹ עַד שֶׁיַּעֲבֹר זְמַנָּם:
🌧️ שני פני הגשם
הלכה ט״ו פוסקת כי גשמים עזים הגורמים לקריסת בתים מהווים עילה לתפילה — אבל לא בארץ ישראל, שבה השטח ההררי ובניית האבן הופכים גשמים מרובים לברכה ולא לסכנה. הלכות ט״ז–י״ז עוסקות בגשם שפסק בטרם עת: אם תבואה צמחה ואחר כך התייבשה, או אם הגיעה פסח (עונת פריחת עצים) ללא גשם, או אם סוכות הגיעה ללא גשם מספיק למילוי בורות ומערות — נדרשים צום וזעקה; ואם אין מים לשתייה, הצום חל אפילו בקיץ. הלכה י״ח מסיימת את הפרק: פער של 40 יום בין גשם לגשם בעונת הגשמים מהווה בצורת המחייבת צום מיידי.
