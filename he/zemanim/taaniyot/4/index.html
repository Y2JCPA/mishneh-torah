הלכות א׳–ד׳
הלכה א׳
בְכָל יוֹם וָיוֹם מִשֶּׁבַע תַּעֲנִיּוֹת הָאַחֲרוֹנוֹת שֶׁל מָטָר מִתְפַּלְּלִין עַל סֵדֶר זֶה. מוֹצִיאִין אֶת הַתֵּבָה לִרְחוֹבָהּ שֶׁל עִיר וְכָל הָעָם מִתְקַבְּצִים וּמִתְכַּסִּים בְּשַׂקִּים. וְנוֹתְנִין אֵפֶר מַקְלֶה עַל גַּבֵּי הַתֵּבָה וְעַל גַּבֵּי סֵפֶר תּוֹרָה כְּדֵי לְהַגְדִּיל אֶת הַבְּכִיָּה וּלְהַכְנִיעַ לִבָּם. וְאֶחָד מִן הָעָם נוֹטֵל מִן הָאֵפֶר וְנוֹתֵן בְּרֹאשׁ הַנָּשִׂיא וּבְרֹאשׁ אַב בֵּית דִּין בְּמָקוֹם הֲנָחַת תְּפִלִּין כְּדֵי שֶׁיִּכָּלְמוּ וְיָשׁוּבוּ. וְכָל אֶחָד וְאֶחָד נוֹטֵל וְנוֹתֵן בְּרֹאשׁוֹ:
הלכה ב׳
וְאַחַר כָּךְ מַעֲמִידִין בֵּינֵיהֶן זָקֵן חָכָם וְהֵן יוֹשְׁבִין. לֹא הָיָה שָׁם זָקֵן חָכָם מַעֲמִידִין חָכָם. לֹא הָיָה שָׁם לֹא זָקֵן וְלֹא חָכָם מַעֲמִידִין אָדָם שֶׁל צוּרָה. וְאוֹמֵר לִפְנֵיהֶם דִּבְרֵי כִּבּוּשִׁין. אַחֵינוּ לֹא שַׂק וְלֹא תַּעֲנִית גּוֹרְמִין אֶלָּא תְּשׁוּבָה וּמַעֲשִׂים טוֹבִים. שֶׁכֵּן מָצִינוּ בְּנִינְוֵה שֶׁלֹּא נֶאֱמַר בְּאַנְשֵׁי נִינְוֵה וַיַּרְא הָאֱלֹהִים אֶת שַׂקָּם וְאֶת תַּעֲנִיתָם אֶלָּא (יונה ג י) "וַיַּרְא הָאֱלֹהִים אֶת מַעֲשֵׂיהֶם". וּבְקַבָּלָה הוּא אוֹמֵר (יואל ב יג) "קִרְעוּ לְבַבְכֶם וְאַל בִּגְדֵיכֶם". וּמוֹסִיף בְּעִנְיָנוֹת אֵלּוּ כְּפִי כֹּחוֹ עַד שֶׁיַּכְנִיעַ לִבָּם וְיָשׁוּבוּ תְּשׁוּבָה גְּמוּרָה:
הלכה ג׳
וְאַחַר שֶׁגּוֹמֵר זֶה דִּבְרֵי כִּבּוּשִׁין עוֹמְדִין בִּתְפִלָּה וּמַעֲמִידִין שְׁלִיחַ צִבּוּר הָרָאוּי לְהִתְפַּלֵּל בְּתַעֲנִיּוֹת אֵלּוּ. וְאִם הָיָה אוֹתוֹ שֶׁאוֹמֵר דִּבְרֵי הַכִּבּוּשִׁין רָאוּי לְהִתְפַּלֵּל מִתְפַּלֵּל וְאִם לָאו מוֹרִידִין אַחֵר:
הלכה ד׳
וְאֵי זֶה הוּא הָרָאוּי לְהִתְפַּלֵּל בְּתַעֲנִיּוֹת אֵלּוּ. אִישׁ שֶׁהוּא רָגִיל בִּתְפִלָּה. וְרָגִיל לִקְרוֹת בַּתּוֹרָה נְבִיאִים וּכְתוּבִים. וּמְטוּפָּל וְאֵין לוֹ. וְיֵשׁ לוֹ יְגִיעָה בַּשָּׂדֶה. וְלֹא יִהְיֶה בְּבָנָיו וּבִבְנֵי בֵּיתוֹ וְכָל קְרוֹבָיו הַנִּלְוִים עָלָיו בַּעַל עֲבֵרָה. אֶלָּא יִהְיֶה בֵּיתוֹ רֵיקָן מִן הָעֲבֵרוֹת. וְלֹא יָצָא עָלָיו שֵׁם רַע בְּיַלְדוּתוֹ. שְׁפַל בֶּרֶךְ. וּמְרֻצֶּה לָעָם. וְיֵשׁ לוֹ נְעִימָה וְקוֹלוֹ עָרֵב. וְאִם הָיָה זָקֵן עִם כָּל הַמִּדּוֹת הָאֵלּוּ הֲרֵי זֶה מְפֹאָר. וְאִם אֵינוֹ זָקֵן הוֹאִיל וְיֵשׁ בּוֹ כָּל הַמִּדּוֹת הָאֵלּוּ יִתְפַּלֵּל:
🌑 טקסי תשובה ציבוריים
שבע התעניות האחרונות מתחילות כשהארון מוצא לרחוב, ואפר מונח על ספר התורה ועל הנשיא ואב בית הדין במקום שבו מניחים תפילין. זקן חכם עומד ומשמיע דברי תוכחה — מצטט את נינווה ואת יואל — ומדגיש שרק תשובה ומעשים טובים, ולא שק ותענית, מביאים לתגובה אלוהית. הקהל בוחר לאחר מכן חזן הבקי בכל כתבי הקודש, שיש לו ילדים אך הוא עני, שביתו נקי לחלוטין מעבירות, ושהוא עניו, אהוב על הציבור, ובעל קול ערב.

פתיחת שמונה עשרה: תהילים, זכרונות, וברכת אברהם

הלכות ה׳–ז׳
הלכה ה׳
וּשְׁלִיחַ צִבּוּר מַתְחִיל וּמִתְפַּלֵּל עַד בִּרְכַּת גּוֹאֵל יִשְׂרָאֵל. וְאוֹמֵר זִכְרוֹנוֹת וְשׁוֹפָרוֹת מֵעֵין הַצָּרָה. וְאוֹמֵר (תהילים קכ א) "אֶל ה' בַּצָּרָתָה לִּי קָרָאתִי וַיַּעֲנֵנִי". (תהילים קכא א) "אֶשָּׂא עֵינַי אֶל הֶהָרִים" וְגוֹ'. (תהילים קל א) "מִמַּעֲמַקִּים קְרָאתִיךָ ה'". (תהילים קב א) "תְּפִלָּה לְעָנִי כִי יַעֲטֹף" וְגוֹ':
הלכה ו׳
וְאוֹמֵר דִּבְרֵי תַּחֲנוּנִים כְּפִי כֹּחוֹ. וְאוֹמֵר רְאֵה נָא בְעָנְיֵנוּ וְרִיבָה רִיבֵנוּ וּמַהֵר לְגָאֳלֵנוּ. וּמִתְחַנֵּן וְאוֹמֵר בְּסוֹף תַּחֲנוּנָיו מִי שֶׁעָנָה אֶת אַבְרָהָם אָבִינוּ בְּהַר הַמּוֹרִיָּה הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע קוֹל צַעֲקַתְכֶם בַּיּוֹם הַזֶּה. בָּרוּךְ אַתָּה ה' גּוֹאֵל יִשְׂרָאֵל:
הלכה ז׳
וּמַתְחִיל לְהוֹסִיף שֵׁשׁ בְּרָכוֹת שֶׁהוּא מוֹסִיף זוֹ אַחַר זוֹ. וּמִתְחַנֵּן בְּכָל אַחַת מֵהֶן בְּדִבְרֵי תַּחֲנוּנִים וּפְסוּקִים מִדִּבְרֵי קַבָּלָה וּמִכִּתְבֵי הַקֹּדֶשׁ כְּפִי שֶׁהוּא רָגִיל וְחוֹתֵם בְּכָל אַחַת מֵהֶן בַּחֲתִימוֹת אֵלּוּ:
🙏 מבנה התפילה המורחב
החזן אומר את שמונה עשרה הרגילה עד גואל ישראל, ומוסיף פסוקי זכרונות ושופרות המתאימים לצרה הנוכחית, יחד עם ארבעה מזמורי תהילים של עוגמת נפש (120, 121, 130, 102). הוא מסיים ב"מי שענה את אברהם בהר המוריה" — מזכיר לאלוהים את זכות העקדה — ולאחר מכן מוסיף שש ברכות חדשות לגמרי, שכל אחת מורכבת מפסוקים שהחזן יודע בעל פה.

שישה גיבורים מהתנ"ך, שישה סיומי ברכות

הלכות ח׳–י״ד
הלכה ח׳
בָּרִאשׁוֹנָה הוּא חוֹתֵם מִי שֶׁעָנָה אֶת משֶׁה וַאֲבוֹתֵינוּ עַל יַם סוּף הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע קוֹל צַעֲקַתְכֶם בַּיּוֹם הַזֶּה. בָּרוּךְ אַתָּה ה' זוֹכֵר הַנִּשְׁכָּחוֹת:
הלכה ט׳
בַּשְּׁנִיָּה הוּא חוֹתֵם מִי שֶׁעָנָה אֶת יְהוֹשֻׁעַ בַּגִּלְגָּל הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע קוֹל צַעֲקַתְכֶם בַּיּוֹם הַזֶּה. בָּרוּךְ אַתָּה ה' שׁוֹמֵעַ תְּרוּעָה:
הלכה י׳
בַּשְּׁלִישִׁית הוּא חוֹתֵם מִי שֶׁעָנָה אֶת שְׁמוּאֵל בַּמִּצְפָּה הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע קוֹל צַעֲקַתְכֶם בַּיּוֹם הַזֶּה. בָּרוּךְ אַתָּה ה' שׁוֹמֵעַ צְעָקָה:
הלכה י״א׳
בָּרְבִיעִית הוּא חוֹתֵם מִי שֶׁעָנָה אֶת אֵלִיָּהוּ בְּהַר הַכַּרְמֶל הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע קוֹל צַעֲקַתְכֶם בַּיּוֹם הַזֶּה. בָּרוּךְ אַתָּה ה' שׁוֹמֵעַ תְּפִלָּה:
הלכה י״ב׳
בַּחֲמִישִׁית הוּא חוֹתֵם מִי שֶׁעָנָה אֶת יוֹנָה בִּמְעֵי הַדָּגָה הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע קוֹל צַעֲקַתְכֶם בַּיּוֹם הַזֶּה. בָּרוּךְ אַתָּה ה' הָעוֹנֶה בְּעֵת צָרָה:
הלכה י״ג׳
בַּשִּׁשִּׁית הוּא חוֹתֵם מִי שֶׁעָנָה אֶת דָּוִד וּשְׁלֹמֹה בְּנוֹ בִּירוּשָׁלַיִם הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע קוֹל צַעֲקַתְכֶם בַּיּוֹם הַזֶּה. בָּרוּךְ אַתָּה ה' הַמְרַחֵם עַל הָאָרֶץ. וְכָל הָעָם עוֹנִין אָמֵן אַחַר כָּל בְּרָכָה וּבְרָכָה כְּדֶרֶךְ שֶׁעוֹנִין אַחַר כָּל הַבְּרָכוֹת:
הלכה י״ד׳
בַּשְּׁבִיעִית אוֹמֵר רְפָאֵנוּ ה' וְנֵרָפֵא וְכוּ'. וְגוֹמֵר הַתְּפִלָּה עַל הַסֵּדֶר וְתוֹקְעִין בַּחֲצוֹצְרוֹת. וְכַסֵּדֶר הַזֶּה עוֹשִׂין בְּכָל מָקוֹם:
📜 שרשרת התגובות האלוהיות
שש הברכות הנוספות מסתיימות כל אחת בנוסח "מי שענה..." המזכיר תקדים מהתנ"ך: משה בים סוף ("זוכר הנשכחים"), יהושע בגלגל ("שומע קול שופרות"), שמואל במצפה ("שומע צעקה"), אליהו בהר הכרמל ("שומע תפילה"), יונה בדג ("עונה בעת צרה"), ודוד ושלמה בירושלים ("מרחם על הארץ"). ברכה שביעית — "רפאנו ה׳" — מסיימת את שמונה עשרה בסדר הרגיל, ולאחריה תקיעת חצוצרות; כל הסדר הזה נוהג בכל הקהילות בכל מקום.

טקס הר הבית: ברכות מורחבות ותקיעות חצוצרה מתחלפות

הלכות ט״ו–י״ז
הלכה ט״ו׳
כְּשֶׁהָיוּ מִתְפַּלְּלִין עַל הַסֵּדֶר הַזֶּה בִּירוּשָׁלַיִם הָיוּ מִתְכַּנְּסִין לְהַר הַבַּיִת כְּנֶגֶד שַׁעַר הַמִּזְרָח וּמִתְפַּלְּלִין כַּסֵּדֶר הַזֶּה. וּכְשֶׁמַּגִּיעַ שְׁלִיחַ צִבּוּר לוֹמַר מִי שֶׁעָנָה אֶת אַבְרָהָם אוֹמֵר בָּרוּךְ אַתָּה ה' אֱלֹהֵינוּ אֱלֹהַי יִשְׂרָאֵל מִן הָעוֹלָם וְעַד הָעוֹלָם בָּרוּךְ אַתָּה ה' גּוֹאֵל יִשְׂרָאֵל. וְהֵן עוֹנִין אַחֲרָיו בָּרוּךְ שֵׁם כְּבוֹד מַלְכוּתוֹ לְעוֹלָם וָעֶד. וְחַזָּן הַכְּנֶסֶת אוֹמֵר לַתּוֹקְעִים תִּקְעוּ בְּנֵי אַהֲרֹן תִּקְעוּ. וְחוֹזֵר הַמִּתְפַּלֵּל וְאוֹמֵר מִי שֶׁעָנָה אֶת אַבְרָהָם בְּהַר הַמּוֹרִיָּה הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע קוֹל צַעֲקַתְכֶם בַּיּוֹם הַזֶּה וְאַחַר כָּךְ תּוֹקְעִין הַכֹּהֲנִים וּמְרִיעִין וְתוֹקְעִין:
הלכה ט״ז׳
וְכֵן בַּבְּרָכָה הַשְּׁנִיָּה לָזוֹ שֶׁהִיא רִאשׁוֹנָה מִן הַשֵּׁשׁ שֶׁמּוֹסִיף חוֹתֵם בָּהּ בָּרוּךְ ה' אֱלֹהִים אֱלֹהֵי יִשְׂרָאֵל מִן הָעוֹלָם וְעַד הָעוֹלָם בָּרוּךְ אַתָּה ה' זוֹכֵר הַנִּשְׁכָּחוֹת. וְהֵן עוֹנִין בָּרוּךְ שֵׁם כְּבוֹד מַלְכוּתוֹ לְעוֹלָם וָעֶד. וְחַזַּן הַכְּנֶסֶת אוֹמֵר לָהֶן הָרִיעוּ בְּנֵי אַהֲרֹן הָרִיעוּ. וְחוֹזֵר שְׁלִיחַ צִבּוּר וְאוֹמֵר מִי שֶׁעָנָה אֶת משֶׁה וַאֲבוֹתֵינוּ עַל יַם סוּף הוּא יַעֲנֶה אֶתְכֶם וְיִשְׁמַע קוֹל צַעֲקַתְכֶם בַּיּוֹם הַזֶּה. וְאַחַר כָּךְ מְרִיעִין וְתוֹקְעִין וּמְרִיעִין:
הלכה י״ז׳
וְכֵן עַל כָּל בְּרָכָה וּבְרָכָה בְּאַחַת אוֹמֵר תִּקְעוּ וּבְאַחַת אוֹמֵר הָרִיעוּ עַד שֶׁיִּגְמֹר כָּל שֶׁבַע הַבְּרָכוֹת. וְנִמְצְאוּ הַכֹּהֲנִים פַּעַם תּוֹקְעִין וּמְרִיעִין וְתוֹקְעִין. וּפַעַם מְרִיעִין וְתוֹקְעִין וּמְרִיעִין שֶׁבַע פְּעָמִים. וְאֵין עוֹשִׂים הַסֵּדֶר זֶה אֶלָּא בְּהַר הַבַּיִת בִּלְבַד. וּכְשֶׁהֵן תּוֹקְעִין וּמְרִיעִין שָׁם תּוֹקְעִין בַּחֲצוֹצְרוֹת וּבְשׁוֹפָר כְּאֶחָד כְּמוֹ שֶׁאָמַרְנוּ:
🎺 המנהג הייחודי של ירושלים
בירושלים, הקהל התאסף בשער המזרחי של הר הבית, שם כל ברכה הסתיימה בנוסח מורחב שנחתם ב"ברוך אתה ה׳ אלוהינו ואלוהי ישראל מן העולם ועד העולם", והעם ענה: "ברוך שם כבוד מלכותו לעולם ועד." השמש הכריז לכוהנים "תקעו תקיעה" או "תקעו תרועה", לסירוגין על פני שבע הברכות — שבע סדרות בסך הכל — וייחודי לכאן: גם חצוצרות וגם שופר נתקעו יחד, מנהג שהיה קיים רק בהר הבית.
7/9

תהלוכה לבית הקברות, נעילה, ומדד הגשמים

הלכות י״ח–י״ט
הלכה י״ח׳
שֶׁבַע תַּעֲנִיּוֹת הָאֵלּוּ כָּל מָקוֹם שֶׁגּוֹזְרִין אוֹתָם שָׁם אַחַר שֶׁמִּתְפַּלְּלִין יוֹצְאִין כָּל הָעָם לְבֵית הַקְּבָרוֹת וּבוֹכִין וּמִתְחַנְּנִים שָׁם. כְּלוֹמַר הֲרֵי אַתֶּם מֵתִים כְּאֵלּוּ אִם לֹא תָּשׁוּבוּ מִדַּרְכֵיכֶם. וּבְכָל תַּעֲנִית מִתַּעֲנִיּוֹת הַצָּרוֹת שֶׁגּוֹזְרִין עַל הַצִּבּוּר מִתְפַּלְּלִין תְּפִלַּת נְעִילָה בְּכָל מָקוֹם:
הלכה י״ט׳
יָרְדוּ לָהֶם גְּשָׁמִים עַד מָתַי יִהְיוּ יוֹרְדִין וְיִפְסְקוּ הַצִּבּוּר מִן הַתַּעֲנִית. מִשֶּׁיֵּרְדוּ בְּעֹמֶק הָאָרֶץ הַחֲרֵבָה טֶפַח. וּבַבֵּינוֹנִית טְפָחַיִם. וּבַעֲבוּדָה עַד שֶׁיֵּרְדוּ בְּעֹמֶק שְׁלֹשָׁה טְפָחִים:
🌧️ הצעדים האחרונים של התשובה
לאחר התפילה, כל העם הולך לבית הקברות לבכות ולהתחנן, כדי להמחיש שמי שמסרב לחזור בתשובה הוא "כמתים האלה." בכל יום תענית ציבורית שנגזרה בגלל צרה, מתפללים תפילת נעילה בכל מקום. הצום מסתיים רק כשהגשם חודר: טפח אחד לתוך אדמה סדוקה, שניים לתוך אדמה רגילה, ושלושה טפחים לתוך אדמה חרושה — תגובה אלוהית שנמדדת בעומק האדמה, לא בתחושות.
8/9

🎓 עקרונות מרכזיים

פרק 4
🎭
טקס בלי תשובה הוא תיאטרון
תוכחת הזקן בהלכה ב׳ קובעת שלבישת שק ותענית הן חסרות משמעות אם לא מלוות בשינוי פנימי אמיתי ומעשים טובים, כפי שמוכיח סיפור נינווה.
🎺
החזן מגלם את הסטנדרט המוסרי של הקהילה
הלכה ד׳ דורשת שהחזן של יום התענית לא רק יהיה מיומן אלא גם צדיק באופן אישי — עני, עניו, בקי בתורה, עם משפחה נקייה מעבירות — כי זכויותיו האישיות מייצגות את כל הקהל לפני האלוהים.
📖
ההיסטוריה היא טיעון חי לתגובה האלוהית
שש הברכות הנוספות מקשרות יחד שישה רגעים שבהם האלוהים ענה לתפילה — מים סוף ועד ירושלים — וטוענות שמה שעשה האלוהים לאבותינו, הוא יעשה שוב.
🌱
לסליחה האלוהית יש סף פיזי ומדיד
הלכה י״ט מכמתת באופן ייחודי את סיום תענית הבצורת לפי עומק חדירת הגשם לאדמה, ומעגנת את המושג התיאולוגי של רחמי האלוהים במציאות חקלאית שניתן לאמת בפועל.
