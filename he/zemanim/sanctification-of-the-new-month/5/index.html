מקור הסמכות לקביעת הלוח

הלכות א׳–ג׳
הלכה א׳
כָּל מַה שֶּׁאָמַרְנוּ מִקְּבִיעוֹת רֹאשׁ חֹדֶשׁ עַל הָרְאִיָּה וְעִבּוּר הַשָּׁנָה מִפְּנֵי הַזְּמַן אוֹ מִפְּנֵי הַצֹּרֶךְ. אֵין עוֹשִׂין אוֹתוֹ אֶלָּא סַנְהֶדְרִין שֶׁבְּאֶרֶץ יִשְׂרָאֵל אוֹ בֵּית דִּין הַסְּמוּכִים בְּאֶרֶץ יִשְׂרָאֵל שֶׁנָּתְנוּ לָהֶן הַסַּנְהֶדְרִין רְשׁוּת. שֶׁכָּךְ נֶאֱמַר לְמשֶׁה וּלְאַהֲרֹן (שמות יב ב) "הַחֹדֶשׁ הַזֶּה לָכֶם רֹאשׁ חֳדָשִׁים" וּמִפִּי הַשְּׁמוּעָה לָמְדוּ אִישׁ מֵאִישׁ מִמּשֶׁה רַבֵּנוּ שֶׁכָּךְ הוּא פֵּרוּשׁ הַדָּבָר עֵדוּת זוֹ תִּהְיֶה מְסוּרָה לָכֶם וּלְכָל הָעוֹמֵד אַחֲרֵיכֶם בִּמְקוֹמְכֶם. אֲבָל בִּזְמַן שֶׁאֵין שָׁם סַנְהֶדְרִין בְּאֶרֶץ יִשְׂרָאֵל אֵין קוֹבְעִין חֳדָשִׁים וְאֵין מְעַבְּרִין שָׁנִים אֶלָּא בְּחֶשְׁבּוֹן זֶה שֶׁאָנוּ מְחַשְּׁבִין בּוֹ הַיּוֹם:
הלכה ב׳
וְדָבָר זֶה הֲלָכָה לְמשֶׁה מִסִּינַי הוּא. שֶׁבִּזְמַן שֶׁיֵּשׁ סַנְהֶדְרִין קוֹבְעִין עַל פִּי הָרְאִיָּה וּבִזְמַן שֶׁאֵין שָׁם סַנְהֶדְרִין קוֹבְעִין עַל פִּי הַחֶשְׁבּוֹן הַזֶּה שֶׁאָנוּ מְחַשְּׁבִין בּוֹ הַיּוֹם וְאֵין נִזְקָקִין לִרְאִיָּה. אֶלָּא פְּעָמִים שֶׁיִּהְיֶה יוֹם שֶׁקּוֹבְעִין בּוֹ בְּחֶשְׁבּוֹן זֶה הוּא יוֹם הָרְאִיָּה אוֹ קוֹדֵם לוֹ בַּיּוֹם אוֹ אַחֲרָיו בַּיּוֹם. וְזֶה שֶׁיִּהְיֶה אַחַר הָרְאִיָּה בְּיוֹם פֶּלֶא הוּא וּבַאֲרָצוֹת שֶׁהֵן לְמַעֲרַב אֶרֶץ יִשְׂרָאֵל:
הלכה ג׳
וּמֵאֵימָתַי הִתְחִילוּ כָּל יִשְׂרָאֵל לְחַשֵּׁב בְּחֶשְׁבּוֹן זֶה. מִסּוֹף חַכְמֵי הַגְּמָרָא בְּעֵת שֶׁחָרְבָה אֶרֶץ יִשְׂרָאֵל וְלֹא נִשְׁאַר שָׁם בֵּית דִּין קָבוּעַ. אֲבָל בִּימֵי חַכְמֵי מִשְׁנָה וְכֵן בִּימֵי חַכְמֵי הַגְּמָרָא עַד יְמֵי אַבַּיֵּי וְרָבָא עַל קְבִיעַת אֶרֶץ יִשְׂרָאֵל הָיוּ סוֹמְכִין:
🏛️ מקור הסמכות
רק הסנהדרין בארץ ישראל — או בית דין סמוך שהוסמך על ידה — רשאי לקדש את החודש על פי ראייה ולעבר שנים. המקור: 'החודש הזה לכם' (שמות י״ב:ב׳) — עדות זו נמסרה למשה ואהרון ולכל העומדים במקומם בארץ ישראל. זוהי הלכה למשה מסיני: כשיש סנהדרין, קובעים על פי ראייה; כשאין — קובעים על פי החישוב הקבוע ואין צורך בראייה כלל. החישוב הקבוע החל בסוף תקופת האמוראים, כשארץ ישראל חרבה ולא נשאר בית דין קבוע.
4/9

יום אחד מול שני ימים של יום טוב — רקע היסטורי

הלכות ד׳–ו׳
הלכה ד׳
כְּשֶׁהָיְתָה סַנְהֶדְרִין קַיֶּמֶת וְהֵן קוֹבְעִין עַל הָרְאִיָּה. הָיוּ בְּנֵי אֶרֶץ יִשְׂרָאֵל וְכָל הַמְּקוֹמוֹת שֶׁמַּגִּיעִין אֲלֵיהֶן שְׁלוּחֵי תִּשְׁרֵי עוֹשִׂין יָמִים טוֹבִים יוֹם אֶחָד בִּלְבַד. וּשְׁאָר הַמְּקוֹמוֹת הָרְחוֹקוֹת שֶׁאֵין שְׁלוּחֵי תִּשְׁרֵי מַגִּיעִין אֲלֵיהֶם הָיוּ עוֹשִׂים שְׁנֵי יָמִים מִסָּפֵק לְפִי שֶׁלֹּא הָיוּ יוֹדְעִין יוֹם שֶׁקָּבְעוּ בּוֹ בְּנֵי אֶרֶץ יִשְׂרָאֵל אֶת הַחֹדֶשׁ:
הלכה ה׳
בַּזְּמַן הַזֶּה שֶׁאֵין שָׁם סַנְהֶדְרִין וּבֵית דִּין שֶׁל אֶרֶץ יִשְׂרָאֵל קוֹבְעִין עַל חֶשְׁבּוֹן זֶה. הָיָה מִן הַדִּין שֶׁיִּהְיוּ בְּכָל הַמְּקוֹמוֹת עוֹשִׂין יוֹם טוֹב אֶחָד בִּלְבַד אֲפִלּוּ הַמְּקוֹמוֹת הָרְחוֹקוֹת שֶׁבְּחוּצָה לָאָרֶץ כְּמוֹ בְּנֵי אֶרֶץ יִשְׂרָאֵל. שֶׁהַכּל עַל חֶשְׁבּוֹן אֶחָד סוֹמְכִין וְקוֹבְעִין. אֲבָל תַּקָּנַת חֲכָמִים הוּא שֶׁיִּזָּהֲרוּ בְּמִנְהַג אֲבוֹתֵיהֶם שֶׁבִּידֵיהֶם:
הלכה ו׳
לְפִיכָךְ כָּל מָקוֹם שֶׁלֹּא הָיוּ שְׁלוּחֵי תִּשְׁרֵי מַגִּיעִין אֵלָיו כְּשֶׁהָיוּ הַשְּׁלוּחִין יוֹצְאִין. יַעֲשׂוּ שְׁנֵי יָמִים וַאֲפִלּוּ בַּזְּמַן הַזֶּה כְּמוֹ שֶׁהָיוּ עוֹשִׂין בִּזְמַן שֶׁבְּנֵי אֶרֶץ יִשְׂרָאֵל קוֹבְעִין עַל הָרְאִיָּה. וּבְנֵי אֶרֶץ יִשְׂרָאֵל בַּזְּמַן הַזֶּה עוֹשִׂין יוֹם אֶחָד כְּמִנְהָגָן שֶׁמֵּעוֹלָם לֹא עָשׂוּ שְׁנֵי יָמִים. נִמְצָא יוֹם טוֹב שֵׁנִי שֶׁאָנוּ עוֹשִׂין בְּגָלֻיּוֹת בַּזְּמַן הַזֶּה מִדִּבְרֵי סוֹפְרִים שֶׁתִּקְּנוּ דָּבָר זֶה:
📅 מנהג אבותינו
בימי הסנהדרין: מקומות שהגיעו אליהם שלוחי תשרי חגגו יום אחד; מקומות רחוקים — שני ימים מספק. כיום, אף שכולם סומכים על חישוב אחד ומן הדין יום אחד צריך להספיק בכל מקום, חכמים תיקנו לשמור על מנהג האבות. לכן: כל מקום שלא הגיעו אליו שלוחי תשרי בזמנם שומר שני ימים גם כיום. תושבי ארץ ישראל שומרים יום אחד כמנהגם מאז ומעולם. יום טוב שני של גלויות כיום הוא מדרבנן.

ראש השנה — שני ימים בכל מקום

הלכות ז׳–ח׳
הלכה ז׳
יוֹם טוֹב שֶׁל רֹאשׁ הַשָּׁנָה בִּזְמַן שֶׁהָיוּ קוֹבְעִין עַל הָרְאִיָּה הָיוּ רֹב בְּנֵי אֶרֶץ יִשְׂרָאֵל עוֹשִׂין אוֹתוֹ שְׁנֵי יָמִים מִסָּפֵק. לְפִי שֶׁלֹּא הָיוּ יוֹדְעִין יוֹם שֶׁקָּבְעוּ בּוֹ בֵּית דִּין אֶת הַחֹדֶשׁ שֶׁאֵין הַשְּׁלוּחִין יוֹצְאִין בְּיוֹם טוֹב:
הלכה ח׳
וְלֹא עוֹד אֶלָּא אֲפִלּוּ בִּירוּשָׁלַיִם עַצְמָהּ שֶׁהוּא מְקוֹם בֵּית דִּין פְּעָמִים רַבּוֹת הָיוּ עוֹשִׂין יוֹם טוֹב שֶׁל רֹאשׁ הַשָּׁנָה שְׁנֵי יָמִים. שֶׁאִם לֹא בָּאוּ הָעֵדִים כָּל יוֹם שְׁלֹשִׁים נוֹהֲגִין הָיוּ בְּאוֹתוֹ הַיּוֹם שֶׁמְּצַפִּין לְעֵדִים קֹדֶשׁ וּלְמָחָר קֹדֶשׁ. וְהוֹאִיל וְהָיוּ עוֹשִׂין אוֹתוֹ שְׁנֵי יָמִים וַאֲפִלּוּ בִּזְמַן הָרְאִיָּה הִתְקִינוּ שֶׁיִּהְיוּ עוֹשִׂין אֲפִלּוּ בְּנֵי אֶרֶץ יִשְׂרָאֵל אוֹתוֹ תָּמִיד שְׁנֵי יָמִים בַּזְּמַן הַזֶּה שֶׁקּוֹבְעִין עַל הַחֶשְׁבּוֹן. הִנֵּה לָמַדְתָּ שֶׁאֲפִלּוּ יוֹם טוֹב שֵׁנִי שֶׁל רֹאשׁ הַשָּׁנָה בַּזְּמַן הַזֶּה מִדִּבְרֵי סוֹפְרִים:
🎺 ספק ראש השנה
אפילו בתקופת הראייה, רוב תושבי ארץ ישראל שמרו ראש השנה שני ימים מספק, שכן שלוחים אינם יוצאים ביום טוב ואנשים לא ידעו מתי בית הדין קבע את החודש. אפילו בירושלים עצמה — מקום בית הדין — לעיתים קרובות חגגו שני ימים: אם לא באו עדים ביום ל׳, נהגו קדושה באותו יום וביום שלמחרתו. מכאן התקנה שגם תושבי ארץ ישראל שומרים תמיד שני ימים של ראש השנה בזמננו. יום שני של ראש השנה גם הוא מדרבנן.

כללי שני ימים לפי מיקום

הלכות ט׳–י״ב
הלכה ט׳
אֵין עֲשִׂיַּת יוֹם טוֹב אֶחָד תְּלוּיָה בִּקְרִיבַת הַמָּקוֹם. כֵּיצַד. אִם יִהְיֶה מָקוֹם בֵּינוֹ וּבֵין יְרוּשָׁלַיִם מַהֲלַךְ חֲמִשָּׁה יָמִים אוֹ פָּחוֹת שֶׁבְּוַדַּאי אֶפְשָׁר שֶׁיַּגִּיעוּ לָהֶן שְׁלוּחִין. אֵין אוֹמְרִין שֶׁאַנְשֵׁי מָקוֹם זֶה עוֹשִׂין יוֹם טוֹב אֶחָד. שֶׁמִּי יֹאמַר לָנוּ שֶׁהָיוּ הַשְּׁלוּחִים יוֹצְאִין לְמָקוֹם זֶה. שֶׁמָּא לֹא הָיוּ הַשְּׁלוּחִים יוֹצְאִין לְמָקוֹם זֶה מִפְּנֵי שֶׁלֹּא הָיוּ שָׁם יִשְׂרָאֵל וְאַחַר שֶׁחָזְרוּ לִקְבֹּעַ עַל הַחֶשְׁבּוֹן יָשְׁבוּ שָׁם יִשְׂרְאֵלִים שֶׁהֵן חַיָּבִין לַעֲשׂוֹת שְׁנֵי יָמִים. אוֹ מִפְּנֵי שֶׁהָיָה חֵרוּם בַּדֶּרֶךְ כְּדֶרֶךְ שֶׁהָיָה בֵּין יְהוּדָה וְגָלִיל בִּימֵי חַכְמֵי הַמִּשְׁנָה. אוֹ מִפְּנֵי שֶׁהָיוּ הַכּוּתִים מוֹנְעִין אֶת הַשְּׁלוּחִין לַעֲבֹר בֵּינֵיהֶן:
הלכה י׳
וְאִלּוּ הָיָה הַדָּבָר תָּלוּי בִּקְרִיבַת הַמָּקוֹם הָיוּ כָּל בְּנֵי מִצְרַיִם עוֹשִׂין יוֹם אֶחָד. שֶׁהֲרֵי אֶפְשָׁר שֶׁיַּגִּיעוּ לָהֶם שְׁלוּחֵי תִּשְׁרֵי. שֶׁאֵין בֵּין יְרוּשָׁלַיִם וּמִצְרַיִם עַל דֶּרֶךְ אַשְׁקְלוֹן אֶלָּא מַהֲלַךְ שְׁמוֹנָה יָמִים אוֹ פָּחוֹת וְכֵן רֹב סוּרְיָא. הָא לָמַדְתָּ שֶׁאֵין הַדָּבָר תָּלוּי בִּהְיוֹת הַמָּקוֹם קָרוֹב:
הלכה י״א׳
נִמְצָא עִקַּר דָּבָר זֶה עַל דֶּרֶךְ זוֹ כָּךְ הוּא. כָּל מָקוֹם שֶׁיֵּשׁ בֵּינוֹ וּבֵין יְרוּשָׁלַיִם מַהֲלָךְ יֶתֶר עַל עֲשָׂרָה יָמִים גְּמוּרִים עוֹשִׂין שְׁנֵי יָמִים לְעוֹלָם כְּמִנְהָגָם מִקֹּדֶם. שֶׁאֵין שְׁלוּחֵי כָּל תִּשְׁרֵי וְתִשְׁרֵי מַגִּיעִין אֶלָּא לְמָקוֹם שֶׁבֵּינוֹ וּבֵין יְרוּשָׁלַיִם מַהֲלַךְ עֲשָׂרָה יָמִים אוֹ פָּחוֹת. וְכָל מָקוֹם שֶׁבֵּינוֹ לְבֵין יְרוּשָׁלַיִם מַהֲלַךְ עֲשָׂרָה יָמִים בְּשָׁוֶה אוֹ פָּחוֹת שֶׁאֶפְשָׁר שֶׁיִּהְיוּ שְׁלוּחִין מַגִּיעִין אֵלָיו. רוֹאִים אִם אוֹתוֹ הַמָּקוֹם מֵאֶרֶץ יִשְׂרָאֵל שֶׁהָיוּ בָּהּ יִשְׂרָאֵל בִּשְׁעַת הָרְאִיָּה בְּכִבּוּשׁ שֵׁנִי כְּגוֹן אוּשָׁא וּשְׁפַרְעָם וְלוּז וְיַבְנֵה וְנֹב וּטְבֶרְיָה וְכַיּוֹצֵא בָּהֶן עוֹשִׂין יוֹם אֶחָד בִּלְבַד. וְאִם אוֹתוֹ הַמָּקוֹם מִסּוּרְיָא כְּגוֹן צוֹר וְדַמֶּשֶׂק וְאַשְׁקְלוֹן וְכַיּוֹצֵא בָּהֶן. אוֹ מִחוּצָה לָאָרֶץ כְּגוֹן מִצְרַיִם וְעַמּוֹן וּמוֹאָב וְכַיּוֹצֵא בָּהֶן. עוֹשִׂין כְּמִנְהַג אֲבוֹתֵיהֶן שֶׁבִּידֵיהֶן אִם יוֹם אֶחָד יוֹם אֶחָד וְאִם שְׁנֵי יָמִים שְׁנֵי יָמִים:
הלכה י״ב׳
מָקוֹם שֶׁבֵּינוֹ וּבֵין יְרוּשָׁלַיִם עֲשָׂרָה יָמִים אוֹ פָּחוֹת מֵעֲשָׂרָה וְהוּא סוּרְיָא אוֹ חוּצָה לָאָרֶץ וְאֵין לָהֶם מִנְהָג. אוֹ שֶׁהִיא עִיר שֶׁנִּתְחַדְּשָׁה בְּמִדְבַּר אֶרֶץ יִשְׂרָאֵל. אוֹ מָקוֹם שֶׁשָּׁכְנוּ בּוֹ יִשְׂרָאֵל עַתָּה. עוֹשִׂין שְׁנֵי יָמִים כְּמִנְהַג רֹב הָעוֹלָם. וְכָל יוֹם טוֹב שֵׁנִי מִדִּבְרֵי סוֹפְרִים וַאֲפִלּוּ יוֹם טוֹב שֵׁנִי שֶׁל רֹאשׁ הַשָּׁנָה שֶׁהַכּל עוֹשִׂין אוֹתוֹ בַּזְּמַן הַזֶּה:
🗺️ מרחק, מנהג ומעמד המקום
הכלל אינו תלוי במרחק הגאוגרפי בלבד — האם שלוחים הגיעו למקום תלוי גם בחסימת דרכים ובהתיישבות יהודית. הוכחה: מצרים (8 ימים מירושלים) עדיין שומרת שני ימים. הכלל הסופי: יותר מעשרה ימים מירושלים — תמיד שני ימים. עשרה ימים או פחות — בודקים: אם המקום בתוך גבולות הכיבוש השני של ארץ ישראל (למשל אושא, שפרעם, יבנה, טבריה) — יום אחד; אם בסוריה או מחוץ לארץ — לפי מנהג אבות. מקום ללא מנהג מבוסס, עיר מדבר חדשה בארץ ישראל, או יישוב חדש — שומר שני ימים כרוב העולם.

סמכות הלוח שייכת לארץ ישראל

הלכה י״ג
הלכה י״ג׳
זֶה שֶׁאָנוּ מְחַשְּׁבִין בַּזְּמַן הַזֶּה כָּל אֶחָד וְאֶחָד בְּעִירוֹ וְאוֹמְרִין שֶׁרֹאשׁ חֹדֶשׁ יוֹם פְּלוֹנִי וְיוֹם טוֹב בְּיוֹם פְּלוֹנִי. לֹא בַּחֶשְׁבּוֹן שֶׁלָּנוּ אָנוּ קוֹבְעִין וְלֹא עָלָיו אָנוּ סוֹמְכִין. שֶׁאֵין מְעַבְּרִין שָׁנִים וְקוֹבְעִין חֳדָשִׁים בְּחוּצָה לָאָרֶץ. וְאֵין אָנוּ סוֹמְכִין אֶלָּא עַל חֶשְׁבּוֹן בְּנֵי אֶרֶץ יִשְׂרָאֵל וּקְבִיעָתָם. וְזֶה שֶׁאָנוּ מְחַשְּׁבִין לְגַלּוֹת הַדָּבָר בִּלְבַד הוּא. כֵּיוָן שֶׁאָנוּ יוֹדְעִין שֶׁעַל חֶשְׁבּוֹן זֶה הֵן סוֹמְכִין אָנוּ מְחַשְּׁבִין לֵידַע יוֹם שֶׁקָּבְעוּ בּוֹ בְּנֵי אֶרֶץ יִשְׂרָאֵל אֵי זֶה יוֹם הוּא. וּקְבִיעַת בְּנֵי אֶרֶץ יִשְׂרָאֵל אוֹתוֹ הוּא שֶׁיִּהְיֶה רֹאשׁ חֹדֶשׁ אוֹ יוֹם טוֹב. לֹא מִפְּנֵי חֶשְׁבּוֹן שֶׁאָנוּ מְחַשְּׁבִין:
🌍 קביעת ארץ ישראל
החישוב שאנו עורכים בזמננו אינו קובע בעצמו ואיננו סומכים עליו לקביעה בפועל. אין מעברים שנים ואין קובעים חודשים מחוץ לארץ ישראל. אנו מחשבים רק כדי לגלות באיזה יום קבעו תושבי ארץ ישראל. קביעת תושבי ארץ ישראל היא שמקדשת את ראש חודש ויום טוב — לא חישוב שלנו.
