מיקום הירח הממוצע — קצבים יומיים וטבלאות חיפוש

הלכות א׳–ב׳
הלכה א׳
הַיָּרֵחַ שְׁנֵי מַהֲלָכִים אְמְצָעיּים יֵשׁ לוֹ. הַיָּרֵחַ עַצְמוֹ מְסַבֵּב בְּגַלְגַּל קָטָן שֶׁאֵינוֹ מַקִּיף אֶת הָעוֹלָם כֻּלּוֹ. וּמַהֲלָכוֹ הָאֶמְצָעִי בְּאוֹתוֹ הַגַּלְגַּל הַקָּטָן נִקְרָא אֶמְצָעִי הַמַּסְלוּל. וְהַגַּלְגַּל הַקָּטָן עַצְמוֹ מְסַבֵּב בְּגַלְגַּל גָּדוֹל הַמַּקִּיף אֶת הָעוֹלָם. וּבְמַהֲלַךְ אֶמְצָעִי זֶה שֶׁל גַּלְגַּל הַקָּטָן בְּאוֹתוֹ הַגַּלְגַּל הַגָּדוֹל הַמַּקִּיף אֶת הָעוֹלָם הוּא הַנִּקְרָא אֶמְצַע הַיָּרֵחַ. מַהֲלַךְ אֶמְצַע הַיָּרֵחַ בְּיוֹם אֶחָד י"ג מַעֲלוֹת וְי' חֲלָקִים וְל"ה שְׁנִיּוֹת. סִימָנָם י"ג יל"ה:
הלכה ב׳
נִמְצָא מַהֲלָכוֹ בַּעֲשָׂרָה יָמִים קל"א מַעֲלוֹת וּמ"ה חֲלָקִים וַחֲמִשִּׁים שְׁנִיּוֹת. סִימָנָם קל"א מה"נ. וְנִמְצָא שְׁאֵרִית מַהֲלָכוֹ בְּק' יוֹם רל"ז מַעֲלוֹת וְל"ח חֲלָקִים וְכ"ג שְׁנִיּוֹת. סִימָנָם רל"ז ל"ח כ"ג. וְנִמְצָא שְׁאֵרִית מַהֲלָכוֹ בְּאֶלֶף יוֹם רי"ו מַעֲלוֹת וְכ"ג חֲלָקִים וְנ' שְׁנִיּוֹת. סִימָנָם רי"ו כג"ן. וְנִמְצָא שְׁאֵרִית מַהֲלָכוֹ בְּי' אֲלָפִים יוֹם ג' מַעֲלוֹת וְנ"ח חֲלָקִים וְכ' שְׁנִיּוֹת. סִימָנָם ג' נ"ח כ'. וְנִמְצָא שְׁאֵרִית מַהֲלָכוֹ בְּכ"ט יוֹם כ"ב מַעֲלוֹת וְשִׁשָּׁה חֲלָקִים וְנ"ו שְׁנִיּוֹת. סִימָנָם כב"ו ונ"ו. וְנִמְצָא שְׁאֵרִית מַהֲלָכוֹ בְּשָׁנָה סְדוּרָה שמ"ד מַעֲלוֹת וְכ"ו חֲלָקִים וּמ"ג שְׁנִיּוֹת. סִימָן לָהֶם שד"ם כ"ו מ"ג. וְעַל דֶּרֶךְ זוֹ תִּכְפּל לְכָל מִנְיַן יָמִים אוֹ שָׁנִים שֶׁתִּרְצֶה:
🌙 תנועת הירח הממוצעת
לירח שתי תנועות ממוצעות בו-זמניות: האנומליה הממוצעת (מנת המסלול) — תנועת הירח בגלגל הקטן (אפיציקל) — ומיקום הירח הממוצע — תנועת הגלגל הקטן בגלגל הגדול המקיף את העולם. מיקום הירח הממוצע ביום הוא 13° 10' 35''. שאריות מחושבות מראש: 10 ימים (131° 45' 50''), 100 ימים (237° 38' 23''), 1000 ימים (216° 23' 50''), 10000 ימים (3° 58' 20''), 29 ימים (22° 6' 56''), שנה רגילה (344° 26' 43'').

האנומליה הממוצעת — קצבים יומיים והעידן

הלכות ג׳–ד׳
הלכה ג׳
וּמַהֲלַךְ אֶמְצַע הַמַּסְלוּל בְּיוֹם אֶחָד י"ג מַעֲלוֹת וּשְׁלֹשָׁה חֲלָקִים וְנ"ד שְׁנִיּוֹת. סִימָנָם י"ג גנ"ד. נִמְצָא מַהֲלָכוֹ בַּעֲשָׂרָה יָמִים ק"ל מַעֲלוֹת ל"ט חֲלָקִים בְּלֹא שְׁנִיּוֹת. סִימָנָם ק"ל ל"ט. וְנִמְצָא שְׁאֵרִית מַהֲלָכוֹ בְּמֵאָה יוֹם רכ"ו מַעֲלוֹת וְכ"ט חֲלָקִים וְנ"ג שְׁנִיּוֹת. סִימָנָם רכ"ו כ"ט נ"ג. וְנִמְצָא שְׁאֵרִית מַהֲלָכוֹ בְּאֶלֶף יוֹם ק"ד מַעֲלוֹת וְנ"ח חֲלָקִים וַחֲמִשִּׁים שְׁנִיּוֹת. סִימָנָם ק"ד נח"ן. וְנִמְצָא שְׁאֵרִית מַהֲלָכוֹ בַּעֲשֶׂרֶת אֲלָפִים יוֹם שכ"ט וּמ"ח חֲלָקִים וְעֶשְׂרִים שְׁנִיּוֹת. סִימָנָם שכ"ט מח"כ. וְנִמְצָא שְׁאֵרִית מַהֲלָכוֹ בְּכ"ט יוֹם י"ח מַעֲלוֹת וְנ"ג חֲלָקִים וְד' שְׁנִיּוֹת. סִימָנָם י"ח נג"ד:
הלכה ד׳
וְנִמְצָא שְׁאֵרִית מַהֲלָכוֹ בְּשָׁנָה סְדוּרָה ש"ה מַעֲלוֹת וְי"ג שְׁנִיּוֹת בְּלֹא חֲלָקִים. סִימָנָם ש"ה י"ג. מְקוֹם אֶמְצַע הַיָּרֵחַ הָיָה בִּתְחִלַּת לֵיל חֲמִישִׁי שֶׁהוּא הָעִקָּר לְחֶשְׁבּוֹנוֹת אֵלּוּ בְּמַזַּל שׁוֹר מַעֲלָה אַחַת וְי"ד חֲלָקִים וּמ"ג שְׁנִיּוֹת. סִימָנָם (א') [ל"א] י"ד מ"ג. וְאֶמְצַע הַמַּסְלוּל הָיָה בְּעִקָּר זֶה פ"ד מַעֲלוֹת וְכ"ח חֲלָקִים וּמ"ב שְׁנִיּוֹת. סִימָנָם פ"ד כ"ח מ"ב. מֵאַחֵר שֶׁתֵּדַע מַהֲלַךְ אֶמְצַע הַיָּרֵחַ וְהָאֶמְצַע שֶׁהוּא הָעִקָּר שֶׁעָלָיו תּוֹסִיף. תֵּדַע מְקוֹם אֶמְצַע הַיָּרֵחַ בְּכָל יוֹם שֶׁתִּרְצֶה עַל דֶּרֶךְ שֶׁעָשִׂיתָ בְּאֶמְצַע הַשֶּׁמֶשׁ. וְאַחַר שֶׁתּוֹצִיא אֶמְצַע הַיָּרֵחַ לִתְחִלַּת הַלַּיְלָה שֶׁתִּרְצֶה הִתְבּוֹנֵן בַּשֶּׁמֶשׁ וְדַע בְּאֵי זֶה מַזָּל הוּא:
📐 אנומליה וערכי עידן
האנומליה הממוצעת ביום היא 13° 3' 54''. טבלאות חיפוש: 10 ימים (130° 39'), 100 ימים (226° 29' 53''), 1000 ימים (104° 58' 50''), 10000 ימים (329° 48' 20''), 29 ימים (18° 53' 4''), שנה רגילה (305° 13'). העידן (ליל חמישי, ג׳ בניסן 4938): מיקום הירח הממוצע = 1° 14' 43'' בשור; אנומליה ממוצעת = 84° 28' 42''. מנקודה קבועה זו מחשבים את כל מיקומי הירח העתידיים על ידי הוספת הערכים היומיים/שנתיים המתאימים.

תיקון שמשי — התאמה לרגע הראייה

הלכות ה׳–ו׳
הלכה ה׳
אִם הָיְתָה הַשֶּׁמֶשׁ מֵחֲצִי מַזַּל דָּגִים עַד חֲצִי מַזַּל טָלֶה. תָּנִיחַ אֶמְצַע הַיָּרֵחַ כְּמוֹת שֶׁהוּא. וְאִם תִּהְיֶה הַשֶּׁמֶשׁ מֵחֲצִי מַזַּל טָלֶה עַד תְּחִלַּת מַזַּל תְּאוֹמִים. תּוֹסִיף עַל אֶמְצַע הַיָּרֵחַ ט"ו חֲלָקִים. וְאִם תִּהְיֶה הַשֶּׁמֶשׁ מִתְּחִלַּת מַזַּל תְּאוֹמִים עַד תְּחִלַּת מַזַּל אַרְיֵה. תּוֹסִיף עַל אֶמְצַע הַיָּרֵחַ ט"ו חֲלָקִים. וְאִם תִּהְיֶה הַשֶּׁמֶשׁ מִתְּחִלַּת מַזַּל אַרְיֵה עַד חֲצִי מַזַּל בְּתוּלָה תּוֹסִיף עַל אֶמְצַע הַיָּרֵחַ ט"ו חֲלָקִים. וְאִם תִּהְיֶה הַשֶּׁמֶשׁ מֵחֲצִי מַזַּל בְּתוּלָה עַד חֲצִי מֹאזְנַיִם. הָנַח אֶמְצַע הַיָּרֵחַ כְּמוֹת שֶׁהוּא. וְאִם תִּהְיֶה הַשֶּׁמֶשׁ מֵחֲצִי מֹאזְנַיִם עַד תְּחִלַּת מַזַּל קֶשֶׁת. תִּגְרַע מֵאֶמְצַע הַיָּרֵחַ ט"ו חֲלָקִים. וְאִם תִּהְיֶה הַשֶּׁמֶשׁ מִתְּחִלַּת מַזַּל קֶשֶׁת עַד תְּחִלַּת מַזַּל דְּלִי. תִּגְרַע מֵאֶמְצַע הַיָּרֵחַ ל' חֲלָקִים. וְאִם תִּהְיֶה הַשֶּׁמֶשׁ מִתְּחִלַּת מַזַּל דְּלִי עַד חֲצִי מַזַּל דָּגִים. תִּגְרַע מֵאֶמְצַע הַיָּרֵחַ ט"ו חֲלָקִים:
הלכה ו׳
וּמַה שֶּׁיִּהְיֶה הָאֶמְצַע אַחַר שֶׁתּוֹסִיף עָלָיו אוֹ תִּגְרַע מִמֶּנּוּ אוֹ תָּנִיחַ אוֹתוֹ כְּמוֹת שֶׁהוּא. הוּא אֶמְצַע הַיָּרֵחַ לְאַחַר שְׁקִיעַת הַחַמָּה בִּכְמוֹ שְׁלִישׁ שָׁעָה בְּאוֹתוֹ הַזְּמַן שֶׁתּוֹצִיא הָאֶמְצַע לוֹ. וְזֶה הוּא הַנִּקְרָא אֶמְצַע הַיָּרֵחַ לִשְׁעַת הָרְאִיָּה:
☀️ תיקון לפי מיקום השמש
אחרי חישוב מיקום הירח הממוצע, מיקום השמש במזלות קובע תיקון: מאמצע דגים עד אמצע טלה ומאמצע בתולה עד אמצע מאזניים — ללא שינוי. מאמצע טלה עד תחילת תאומים ומאריה עד אמצע בתולה — מוסיפים ט״ו חלקים. מתחילת תאומים עד תחילת אריה — מוסיפים ל׳ חלקים. מאמצע מאזניים עד תחילת קשת ומדלי עד אמצע דגים — מפחיתים ט״ו חלקים. מתחילת קשת עד תחילת דלי — מפחיתים ל׳ חלקים. התוצאה היא מיקום הירח הממוצע ברגע הראייה — כ-20 דקות אחרי שקיעה, כשהכוכבים מתחילים להיראות בארץ ישראל.

שתי תנועות ירח בו-זמניות
הירח נע בו-זמנית בשני מסלולים: הגלגל הקטן (אנומליה ממוצעת 13° 3' 54'' ביום) והגלגל הגדול (מיקום ממוצע 13° 10' 35'' ביום). שניהם חייבים להיות מחושבים בנפרד.
📊
טבלאות חיפוש מחושבות מראש למהירות
הרמב"ם מספק ערכי שארית ל-10, 100, 1000, ו-10000 ימים — מאפשרים פירוק כל מספר ימים שעברו וחיפוש בטבלה ללא כפל ארוך.
📅
העידן מעגן את כל החישוב
כל החישובים נמדדים מעידן קבוע: ליל חמישי, ג׳ בניסן 4938. מיקום הירח והאנומליה באותו רגע הם קבועים ידועים שמהם נגזר כל תאריך עתידי.
⚖️
השמש קובעת את רגע הראייה
מיקום השמש במזלות משפיע על זמן השקיעה בפועל, ולכן מוסיפים או מפחיתים עד ל׳ חלקים ממיקום הירח הממוצע כדי לקבל את מיקומו ברגע המדויק של ניסיון הראייה — כ-20 דקות אחרי השקיעה.
