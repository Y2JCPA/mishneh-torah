הבסיס לשנה מעוברת

הלכות א׳–ד׳
הלכה א׳
שָׁנָּה מְעֻבֶּרֶת הִיא שָׁנָה שֶׁמּוֹסִיפִין בָּהּ חֹדֶשׁ. וְאֵין מוֹסִיפִין לְעוֹלָם אֶלָּא אֲדָר וְעוֹשִׂין אוֹתָהּ שָׁנָה שְׁנֵי אֲדָרִין אֲדָר רִאשׁוֹן וַאֲדָר שֵׁנִי. וּמִפְּנֵי מָה מוֹסִיפִין חֹדֶשׁ זֶה מִפְּנֵי זְמַן הָאָבִיב כְּדֵי שֶׁיְּהֵא הַפֶּסַח בְּאוֹתוֹ זְמַן שֶׁנֶּאֱמַר (דברים טז א) "שָׁמוֹר אֶת חֹדֶשׁ הָאָבִיב" שֶׁיִּהְיֶה חֹדֶשׁ זֶה בִּזְמַן הָאָבִיב. וְלוּלֵא הוֹסָפַת הַחֹדֶשׁ הַזֶּה הַפֶּסַח בָּא פְּעָמִים בִּימוֹת הַחַמָּה וּפְעָמִים בִּימוֹת הַגְּשָׁמִים:
הלכה ב׳
עַל שְׁלֹשָׁה סִימָנִין מְעַבְּרִין אֶת הַשָּׁנָה. עַל הַתְּקוּפָה וְעַל הָאָבִיב וְעַל פֵּרוֹת הָאִילָן. כֵּיצַד. בֵּית דִּין מְחַשְּׁבִין וְיוֹדְעִין אִם תִּהְיֶה תְּקוּפַת נִיסָן בְּשִׁשָּׁה עָשָׂר בְּנִיסָן אוֹ אַחַר זְמַן זֶה מְעַבְּרִין אוֹתָהּ הַשָּׁנָה. וְיַעֲשׂוּ אוֹתוֹ נִיסָן אֲדָר שֵׁנִי כְּדֵי שֶׁיִּהְיֶה הַפֶּסַח בִּזְמַן הָאָבִיב. וְעַל סִימָן זֶה סוֹמְכִין וּמְעַבְּרִין וְאֵין חוֹשְׁשִׁין לְסִימָן אַחֵר:
הלכה ג׳
וְכֵן אִם רָאוּ בֵּית דִּין שֶׁעֲדַיִן לֹא הִגִּיעַ הָאָבִיב אֶלָּא עֲדַיִן אָפֵל הוּא. וְלֹא צָמְחוּ פֵּרוֹת הָאִילָן שֶׁדַּרְכָּן לִצְמֹחַ בִּזְמַן הַפֶּסַח. סוֹמְכִין עַל שְׁנֵי סִימָנִין אֵלּוּ וּמְעַבְּרִין אֶת הַשָּׁנָה. וְאַף עַל פִּי שֶׁהַתְּקוּפָה קֹדֶם לְשִׁשָּׁה עָשָׂר בְּנִיסָן הֲרֵי הֵן מְעַבְּרִין. כְּדֵי שֶׁיִּהְיֶה הָאָבִיב מָצוּי לְהַקְרִיב מִמֶּנּוּ עֹמֶר הַתְּנוּפָה בְּשִׁשָּׁה עָשָׂר בְּנִיסָן. וּכְדֵי שֶׁיִּהְיוּ הַפֵּרוֹת צוֹמְחִין כְּדֶרֶךְ כָּל זְמַן הָאָבִיב:
הלכה ד׳
וְעַל שָׁלֹשׁ אֲרָצוֹת הָיוּ סוֹמְכִין בָּאָבִיב. עַל אֶרֶץ יְהוּדָה וְעַל עֵבֶר הַיַּרְדֵּן וְעַל הַגָּלִיל. וְאִם הִגִּיעַ הָאָבִיב בִּשְׁתֵּי אֲרָצוֹת מֵאֵלּוּ וּבְאַחַת לֹא הִגִּיעַ אֵין מְעַבְּרִין. וְאִם הִגִּיעַ בְּאַחַת מֵהֶן וְלֹא הִגִּיעַ בִּשְׁתַּיִם מְעַבְּרִין אִם עֲדַיִן לֹא צָמְחוּ פֵּרוֹת הָאִילָן. וְאִלּוּ הֵן הַדְּבָרִים שֶׁהֵן הָעִקָּר שֶׁמְּעַבְּרִין בִּשְׁבִילָן כְּדֵי שֶׁיִּהְיוּ הַשָּׁנִים שְׁנֵי חַמָּה:
🌾 שלושת הסימנים העיקריים
שנה מעוברת מוסיפה חודש נוסף של אדר — ויוצרת שני אדרים — כדי לקיים את המצווה 'שמור את חודש האביב' (דברים ט״ז:א׳). שלושה סימנים עיקריים מחייבים עיבור: (1) תקופת ניסן חלה בט״ז בניסן או מאוחר יותר; (2) תבואת האביב לא בשלה; (3) פירות האילן לא גדלו. התקופה לבדה מספיקה; תבואה ופירות יחד מספיקים אפילו אם התקופה מוקדמת. בדיקת האביב מתבצעת על פי שלושה אזורים: יהודה, עבר הירדן, והגליל — אזור אחד כנגד שניים.
4/9

גורמים משניים וסיבות אסורות

הלכות ה׳–ח׳
הלכה ה׳
וְיֵשׁ שָׁם דְּבָרִים אֲחֵרִים שֶׁהָיוּ בֵּית דִּין מְעַבְּרִין בִּשְׁבִילָן מִפְּנֵי הַצֹּרֶךְ. וְאֵלּוּ הֵן. מִפְּנֵי הַדְּרָכִים שֶׁאֵינָן מְתֻקָּנִין וְאֵין הָעָם יְכוֹלִין לַעֲלוֹת מְעַבְּרִין אֶת הַשָּׁנָה עַד שֶׁיִּפָּסְקוּ הַגְּשָׁמִים וִיתַקְּנוּ הַדְּרָכִים. וּמִפְּנֵי הַגְּשָׁרִים שֶׁנֶּהֶרְסוּ וְנִמְצְאוּ הַנְּהָרוֹת מַפְסִיקִין וּמוֹנְעִין אֶת הָעָם וּמִסְתַּכְּנִין בְּעַצְמָן וּמֵתִים מְעַבְּרִין אֶת הַשָּׁנָה עַד שֶׁיְּתַקְּנוּ הַגְּשָׁרִים. וּמִפְּנֵי תַּנּוּרֵי פְּסָחִים שֶׁאָבְדוּ בַּגְּשָׁמִים וְאֵין לָהֶם מָקוֹם לִצְלוֹת אֶת פִּסְחֵיהֶם מְעַבְּרִין אֶת הַשָּׁנָה עַד שֶׁיִּבָּנוּ הַתַּנּוּרִים וְיִיבְשׁוּ. וּמִפְּנֵי גָּלֻיּוֹת יִשְׂרָאֵל שֶׁנֶּעֶקְרוּ מִמְּקוֹמָן וַעֲדַיִן לֹא הִגִּיעוּ לִירוּשָׁלַיִם מְעַבְּרִין אֶת הַשָּׁנָה כְּדֵי שֶׁיִּהְיֶה לָהֶם פְּנַאי לְהַגִּיעַ:
הלכה ו׳
אֲבָל אֵין מְעַבְּרִין אֶת הַשָּׁנָה לֹא מִפְּנֵי הַשֶּׁלֶג וְלֹא מִפְּנֵי הַצִּנָּה וְלֹא מִפְּנֵי גָּלֻיּוֹת יִשְׂרָאֵל שֶׁעֲדַיִן לֹא נֶעֶקְרוּ מִמְּקוֹמָם. וְלֹא מִפְּנֵי הַטֻּמְאָה. כְּגוֹן שֶׁהָיוּ רֹב הַקָּהָל אוֹ רֹב הַכֹּהֲנִים טְמֵאִים אֵין מְעַבְּרִין אֶת הַשָּׁנָה כְּדֵי שֶׁיִּהְיֶה לָהֶם פְּנַאי לִטַּהֵר וְיַעֲשׂוּ בְּטָהֳרָה אֶלָּא יַעֲשׂוּ בְּטֻמְאָה. וְאִם עִבְּרוּ אֶת הַשָּׁנָה מִפְּנֵי הַטֻּמְאָה הֲרֵי זוֹ מְעֻבֶּרֶת:
הלכה ז׳
יֵשׁ דְּבָרִים שֶׁאֵין מְעַבְּרִין בִּשְׁבִילָן כְּלָל אֲבָל עוֹשִׂין אוֹתָן סַעַד לְשָׁנָה שֶׁצְּרִיכָה עִבּוּר מִפְּנֵי הַתְּקוּפָה אוֹ מִפְּנֵי הָאָבִיב וּפֵרוֹת הָאִילָן. וְאֵלּוּ הֵן. מִפְּנֵי הַגְּדָיִים וְהַטְּלָאִים שֶׁעֲדַיִן לֹא נוֹלְדוּ אוֹ שֶׁהֵן מְעַט. ומִפְּנֵי הַגּוֹזָלוֹת שֶׁלֹּא פָּרְחוּ. אֵין מְעַבְּרִין בִּשְׁבִיל אֵלּוּ כְּדֵי שֶׁיִּהְיוּ הַגְּדָיִים וְהַטְּלָאִים מְצוּיִין לִפְסָחִים וְהַגּוֹזָלוֹת מְצוּיִין לִרְאִיָּה אוֹ לְמִי שֶׁנִּתְחַיֵּב בְּקָרְבַּן הָעוֹף. אֲבָל עוֹשִׂין אוֹתָן סַעַד לְשָׁנָה:
הלכה ח׳
כֵּיצַד עוֹשִׂין אוֹתָן סַעַד לְשָׁנָה. אוֹמְרִין שָׁנָה זוֹ צְרִיכָה עִבּוּר מִפְּנֵי הַתְּקוּפָה שֶׁמָּשְׁכָה אוֹ מִפְּנֵי הָאָבִיב וּפֵרוֹת הָאִילָן שֶׁלֹּא הִגִּיעוּ וְעוֹד שֶׁהַגְּדָיִים קְטַנִּים וְהַגּוֹזָלוֹת רַכִּים:
🚧 מה מצדיק ומה לא
בית הדין רשאי לעבר לצרכים מעשיים: דרכים שלא תוקנו, גשרים שנהרסו, תנורי פסח שאבדו, גולים שעדיין בדרך. אך עיבור אסור בגלל: שלג, קור, גולים שטרם יצאו, או טומאת הרוב — הם עושים את הפסח בטומאה (אך אם עיברו בגלל טומאה, הדבר תקף בדיעבד). גדיים, כבשים ויונים אינם סיבה עצמאית לעיבור, אלא משמשים כטעמים תומכים — הצדקה נוספת לשנה שממילא זקוקה לעיבור מסיבה עיקרית.

תהליך ההחלטה

הלכות ט׳–י׳
הלכה ט׳
אֵין מְעַבְּרִין אֶת הַשָּׁנָה אֶלָּא בִּמְזֻמָּנִין לָהּ. כֵּיצַד. יֹאמַר רֹאשׁ בֵּית דִּין הַגָּדוֹל לִפְלוֹנִי וּפְלוֹנִי מִן הַסַּנְהֶדְרִין הֱיוּ מְזֻמָּנִין לְמָקוֹם פְּלוֹנִי שֶׁנְּחַשֵּׁב וְנִרְאֶה וְנֵדַע אִם שָׁנָה זוֹ צְרִיכָה עִבּוּר אוֹ אֵינָהּ צְרִיכָה. וְאוֹתָן שֶׁהֻזְמְנוּ בִּלְבַד הֵן שֶׁמְּעַבְּרִין אוֹתָהּ. וּבְכַמָּה מְעַבְּרִין אוֹתָהּ. מַתְחִילִין בִּשְׁלֹשָׁה דַּיָּנִין מִכְּלַל סַנְהֶדְרֵי גְּדוֹלָה מִמִּי שֶּׁסָּמְכוּ אוֹתָן. אָמְרוּ שְׁנַיִם לֹא נֵשֵׁב וְלֹא נִרְאֶה אִם צְרִיכָה עִבּוּר אִם לָאו וְאֶחָד אָמַר נֵשֵׁב וְנִבְדֹּק. בָּטֵל יָחִיד בְּמִעוּטוֹ. אָמְרוּ שְׁנַיִם נֵשֵׁב וְנִרְאֶה וְאֶחָד אוֹמֵר לֹא נֵשֵׁב. מוֹסִיפִין עוֹד שְׁנַיִם מִן הַמְזֻמָּנִים וְנוֹשְׂאִים וְנוֹתְנִין בַּדָּבָר:
הלכה י׳
שְׁנַיִם אוֹמְרִים צְרִיכָה עִבּוּר וּשְׁלֹשָׁה אוֹמְרִין אֵינָהּ צְרִיכָה. בָּטְלוּ שְׁנַיִם בְּמִעוּטָן. שְׁלֹשָׁה אוֹמְרִין צְרִיכָה עִבּוּר וּשְׁנַיִם אוֹמְרִין אֵינָהּ צְרִיכָה עִבּוּר. מוֹסִיפִין שְׁנַיִם מִן הַמְזֻמָּנִין לָהּ וְנוֹשְׂאִין וְנוֹתְנִין וְגוֹמְרִין בְּשִׁבְעָה. אִם גָּמְרוּ כֻּלָּם לְעַבֵּר אוֹ שֶׁלֹּא לְעַבֵּר עוֹשִׂין כְּמוֹ שֶׁגָּמְרוּ. וְאִם נֶחְלְקוּ הוֹלְכִים אַחַר הָרֹב בֵּין לְעַבֵּר בֵּין שֶׁלֹּא לְעַבֵּר. וְצָרִיךְ שֶׁיְּהֵא רֹאשׁ בֵּית דִּין הַגָּדוֹל שֶׁהוּא רֹאשׁ יְשִׁיבָה שֶׁל אֶחָד וְשִׁבְעִים מִכְּלַל הַשִּׁבְעָה. וְאִם גָּמְרוּ בִּשְׁלֹשָׁה לְעַבֵּר הֲרֵי זוֹ מְעֻבֶּרֶת וְהוּא שֶׁיְּהֵא הַנָּשִׂיא עִמָּהֶן אוֹ שֶׁיִּרְצֶה. וּבְעִבּוּר הַשָּׁנָה מַתְחִילִין מִן הַצַּד. וּלְקִדּוּשׁ הַחֹדֶשׁ מַתְחִילִין מִן הַגָּדוֹל:
⚖️ נוהל הדיון
עיבור מתבצע רק על ידי מי שהוזמן במפורש על ידי ראש בית הדין הגדול. התהליך מתחיל עם שלושה דיינים סמוכים; אם שניים רוצים לדון ואחד לא — מוסיפים עוד שניים. אם שלושה בעד עיבור ושניים נגד — מוסיפים עוד שניים כדי להגיע לשבעה. השבעה מחליטים ברוב קולות. הנשיא (ראש בית הדין הגדול) חייב להיות בין השבעה, ועיבור על ידי שלושה תקף רק אם הנשיא ביניהם או מסכים. לעיבור, הדיון מתחיל מהקטן; לקידוש החודש — מהגדול.

מגבלות על הסמכות ותנאי תקפות

הלכות י״א–י״ד
הלכה י״א׳
אֵין מוֹשִׁיבִין לְעִבּוּר הַשָּׁנָה לֹא מֶלֶךְ וְלֹא כֹּהֵן גָּדוֹל. מֶלֶךְ מִפְּנֵי חֵילוֹתָיו וּמִלְחֲמוֹתָיו שֶׁמָּא דַּעְתּוֹ נוֹטָה בִּשְׁבִילָן לְעַבֵּר אוֹ שֶׁלֹּא לְעַבֵּר. וְכֹהֵן גָּדוֹל מִפְּנֵי הַצִּנָּה שֶׁמָּא לֹא תִּהְיֶה דַּעְתּוֹ נוֹטָה לְעַבֵּר כְּדֵי שֶׁלֹּא יָבֹא תִּשְׁרֵי בִּימֵי הַקֹּר וְהוּא טוֹבֵל בְּיוֹם הַכִּפּוּרִים חָמֵשׁ טְבִילוֹת:
הלכה י״ב׳
הָיָה רֹאשׁ בֵּית דִּין הַגָּדוֹל וְהוּא הַנִּקְרָא נָשִׂיא בְּדֶרֶךְ רְחוֹקָה אֵין מְעַבְּרִין אוֹתָהּ אֶלָּא עַל תְּנַאי אִם יִרְצֶה הַנָּשִׂיא. בָּא וְרָצָה הֲרֵי זוֹ מְעֻבֶּרֶת. לֹא רָצָה אֵינָהּ מְעֻבֶּרֶת. וְאֵין מְעַבְּרִין אֶת הַשָּׁנָה אֶלָּא בְּאֶרֶץ יְהוּדָה שֶׁהַשְּׁכִינָה בְּתוֹכָהּ שֶׁנֶּאֱמַר (דברים יב ה) "לְשִׁכְנוֹ תִדְרְשׁוּ". וְאִם עִבְּרוּהָ בַּגָּלִיל מְעֻבֶּרֶת. וְאֵין מְעַבְּרִין אֶלָּא בַּיּוֹם. וְאִם עִבְּרוּהָ בַּלַּיְלָה אֵינָהּ מְעֻבֶּרֶת:
הלכה י״ג׳
יֵשׁ לְבֵית דִּין לְחַשֵּׁב וְלִקְבֹּעַ וְלֵידַע אֵי זוֹ שָׁנָה תִּהְיֶה מְעֻבֶּרֶת בְּכָל עֵת שֶׁיִּרְצֶה אֲפִלּוּ לְכַמָּה שָׁנִים. אֲבָל אֵין אוֹמְרִין שָׁנָה פְּלוֹנִית מְעֻבֶּרֶת אֶלָּא אַחַר רֹאשׁ הַשָּׁנָה הוּא שֶׁאוֹמֵר שָׁנָה זוֹ מְעֻבֶּרֶת. וְדָבָר זֶה מִפְּנֵי הַדְּחָק אֲבָל שֶׁלֹּא בִּשְׁעַת הַדְּחָק אֵין מוֹדִיעִין שֶׁהִיא מְעֻבֶּרֶת אֶלָּא בַּאֲדָר הוּא שֶׁאוֹמֵר שָׁנָה זוֹ מְעֻבֶּרֶת וְחֹדֶשׁ הַבָּא אֵינוֹ נִיסָן אֶלָּא אֲדָר שֵׁנִי. אָמְרוּ לִפְנֵי רֹאשׁ הַשָּׁנָה שָׁנָה זוֹ שֶׁתִּכָּנֵס מְעֻבֶּרֶת אֵינָהּ מְעֻבֶּרֶת בַּאֲמִירָה זוֹ:
הלכה י״ד׳
הִגִּיעַ יוֹם שְׁלֹשִׁים בַּאֲדָר וְלֹא עִבְּרוּ עֲדַיִן הַשָּׁנָה לֹא יְעַבְּרוּ אוֹתָהּ כְּלָל. שֶׁאוֹתוֹ הַיּוֹם רָאוּי לִהְיוֹת רֹאשׁ חֹדֶשׁ נִיסָן וּמִשֶּׁיִּכָּנֵס נִיסָן וְלֹא עִבְּרוּ אֵינָן יְכוֹלִים לְעַבֵּר. וְאִם עִבְּרוּהָ בְּיוֹם שְׁלֹשִׁים שֶׁל אֲדָר הֲרֵי זוֹ מְעֻבֶּרֶת. בָּאוּ עֵדִים אַחַר שֶׁעִבְּרוּ וְהֵעִידוּ עַל הַיָּרֵחַ הֲרֵי אֵלּוּ מְקַדְּשִׁין אֶת הַחֹדֶשׁ בְּיוֹם שְׁלֹשִׁים וְיִהְיֶה רֹאשׁ חֹדֶשׁ אֲדָר שֵׁנִי. וְאִלּוּ קִדְּשׁוּהוּ קֹדֶם שֶׁיְּעַבְּרוּ אֶת הַשָּׁנָה שׁוּב לֹא הָיוּ מְעַבְּרִין שֶׁאֵין מְעַבְּרִין בְּנִיסָן:
🏛️ מי, היכן ומתי
מלך וכהן גדול פסולים מוועדת העיבור: המלך — חשש להטיה בגלל חייליו; הכהן הגדול — חשש שיעדיף שלא לעבר כדי שתשרי לא יחול בעונה קרה המחייבת חמש טבילות קרות ביום הכיפורים. אם הנשיא נמצא בנסיעה רחוקה — העיבור מתבצע בתנאי, בהמתנה לאישורו. עיבור תקף רק בארץ יהודה (בדיעבד, תקף בגליל), ורק ביום — בלילה הוא בטל. בית הדין רשאי לחשב שנים מראש, אך ההכרזה אינה נעשית לפני ראש השנה; בדרך כלל מכריזים רק באדר.

איסורים מיוחדים ומכתבי ההכרזה

הלכות ט״ו–י״ז
הלכה ט״ו׳
אֵין מְעַבְּרִין אֶת הַשָּׁנָה בִּשְׁנַת רְעָבוֹן שֶׁהַכּל רָצִים לְבַיִת הַגְּרָנוֹת לֶאֱכל וְלִחְיוֹת וְאִי אֶפְשָׁר לְהוֹסִיף לָהֶן זְמַן לֶאֱסֹר הֶחָדָשׁ. וְאֵין מְעַבְּרִין בִּשְׁבִיעִית שֶׁיַּד הַכּל שׁוֹלֶטֶת עַל הַסְּפִיחִין וְלֹא יִמְצְאוּ לְקַּרֵב הָעֹמֶר וּשְׁתֵּי הַלֶּחֶם. וּרְגִילִין הָיוּ לְעַבֵּר בְּעֶרֶב שְׁבִיעִית:
הלכה ט״ז׳
יֵרָאֶה לִי שֶׁזֶּה שֶׁאָמְרוּ חֲכָמִים אֵין מְעַבְּרִין בִּשְׁנַת רְעָבוֹן וּבִשְׁבִיעִית. שֶׁלֹּא יְעַבְּרוּ בָּהֶם מִפְּנֵי צֹרֶךְ הַדְּרָכִים וְהַגְּשָׁרִים וְכַיּוֹצֵא בָּהֶם. אֲבָל אִם הָיְתָה הַשָּׁנָה רְאוּיָה לְהִתְעַבֵּר מִפְּנֵי הַתְּקוּפָה אוֹ מִפְּנֵי הָאָבִיב וּפֵרוֹת הָאִילָן מְעַבְּרִין לְעוֹלָם בְּכָל זְמַן:
הלכה י״ז׳
כְּשֶׁמְּעַבְּרִין בֵּית דִּין אֶת הַשָּׁנָה כּוֹתְבִין אִגְּרוֹת לְכָל הַמְּקוֹמוֹת הָרְחוֹקִים וּמוֹדִיעִים אוֹתָן שֶׁעִבְּרוּהָ וּמִפְּנֵי מָה עִבְּרוּ. וְעַל לְשׁוֹן הַנָּשִׂיא נִכְתָּבוֹת. וְאוֹמֵר לָהֶן יוֹדֵעַ לָכֶם שֶׁהִסְכַּמְתִּי אֲנִי וַחֲבֵרַי וְהוֹסַפְנוּ עַל שָׁנָה זוֹ כָּךְ וְכָךְ. רָצוּ תִּשְׁעָה וְעֶשְׂרִים יוֹם רָצוּ שְׁלֹשִׁים יוֹם. שֶׁחֹדֶשׁ הָעִבּוּר הָרְשׁוּת לְבֵית דִּין לְהוֹסִיפוֹ מָלֵא אוֹ חָסֵר לָאֲנָשִׁים הָרְחוֹקִים שֶׁמּוֹדִיעִין אוֹתָם. אֲבָל הֵם לְפִי הָרְאִיָּה הֵם עוֹשִׂים אִם מָלֵא אִם חָסֵר:
📜 רעב, שמיטה ומכתבים רשמיים
אין לעבר בשנת רעב — כולם ממהרים לאכול מהתבואה החדשה ואי אפשר להאריך את איסור החדש. אין לעבר בשנת שמיטה — הספיחים הם הפקר ואינם יכולים לספק את העומר ושתי הלחם; המנהג הוא לעבר בשנה שלפני השמיטה. הרמב״ם מבהיר: איסור זה חל רק על עיבור לצרכים מעשיים — אך אם השנה צריכה עיבור בגלל תקופה, תבואת האביב או פירות, מעברים תמיד. לאחר ההחלטה, בית הדין כותב מכתבים רשמיים לכל הקהילות הרחוקות בשם הנשיא.
