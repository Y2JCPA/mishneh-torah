ריחוק כפול וגבולותיו

הלכות א׳–ב׳
הלכה א׳
אִם תִּרְצֶה לֵידַע מְקוֹם הַיָּרֵח הָאֲמִתִּי בְּכָל יוֹם שֶׁתִּרְצֶה. תּוֹצִיא תְּחִלָּה אֶמְצַע הַיָּרֵחַ לִשְׁעַת הָרְאִיָּה לְאוֹתוֹ הַלַּיְלָה שֶׁתִּרְצֶה. וְכֵן תּוֹצִיא אֶמְצַע הַמַּסְלוּל וְאֶמְצַע הַשֶּׁמֶשׁ לְאוֹתוֹ הָעֵת. וְתִגְרַע אֶמְצַע הַשֶּׁמֶשׁ מֵאֶמְצַע הַיָּרֵחַ. וְהַנִּשְׁאָר תִּכְפּל אוֹתוֹ. וְזֶה הוּא הַנִּקְרָא מֶרְחָק הַכָּפוּל:
הלכה ב׳
וּכְבָר הוֹדַעְנוּ שֶׁלֹּא בָּאנוּ בְּכָל אֵלּוּ הַחֶשְׁבּוֹנוֹת שֶׁעָשִׂינוּ בִּפְרָקִים אֵלּוּ אֶלָּא לָדַעַת רְאִיַּת הַיָּרֵחַ. וּלְעוֹלָם אִי אֶפְשָׁר שֶׁיִּהְיֶה מֶרְחָק זֶה הַכָּפוּל בְּלֵיל הָרְאִיָּה שֶׁיִּרְאֶה בָּהּ הַיָּרֵחַ אֶלָּא מֵחָמֵשׁ מַעֲלוֹת עַד ס"ב מַעֲלוֹת. וְאִי אֶפְשָׁר שֶׁיּוֹסִיף עַל זֶה וְלֹא יִגְרַע מִמֶּנּוּ:
🌙 חישוב הריחוק הכפול
למציאת מיקום הירח האמיתי, תחילה מחשבים את מיקום הירח הממוצע, האנומליה הממוצעת, ומיקום השמש הממוצע בזמן הראייה. מפחיתים את השמש הממוצעת מהירח הממוצע, ואז מכפילים את התוצאה — זהו הריחוק הכפול. בליל הראייה הפוטנציאלית, ערך זה לעולם אינו פחות מ-5° ולעולם אינו יותר מ-62°.

תיקון האנומליה הממוצעת

הלכה ג׳
הלכה ג׳
וְהוֹאִיל וְהַדָּבָר כֵּן. הִתְבּוֹנֵן בְּמֶרְחָק זֶה הַכָּפוּל. אִם יִהְיֶה הַמֶּרְחָק הַכָּפוּל חָמֵשׁ מַעֲלוֹת אוֹ קָרוֹב לְחָמֵשׁ אֵין חוֹשְׁשִׁין לְתוֹסֶפֶת וְלֹא תּוֹסִיף כְּלוּם. וְאִם יִהְיֶה הַמֶּרְחָק הַכָּפוּל מִשֵּׁשׁ מַעֲלוֹת עַד אַחַת עֶשְׂרֵה מַעֲלוֹת תּוֹסִיף עַל אֶמְצַע הַמַּסְלוּל מַעֲלָה אַחַת. וְאִם יִהְיֶה מֶרְחָק הַכָּפוּל מִשְּׁתֵּים עֶשְׂרֵה מַעֲלוֹת עַד י"ח מַעֲלוֹת תּוֹסִיף עַל אֶמְצַע הַמַּסְלוּל שְׁתֵּי מַעֲלוֹת. וְאִם יִהְיֶה הַמֶּרְחָק הַכָּפוּל מִי"ט מַעֲלוֹת עַד כ"ד מַעֲלוֹת תּוֹסִיף עַל אֶמְצַע הַמַּסְלוּל שָׁלֹשׁ מַעֲלוֹת. וְאִם יִהְיֶה הַמֶּרְחָק הַכָּפוּל מִכ"ה מַעֲלוֹת עַד ל"א מַעֲלוֹת תּוֹסִיף עַל אֶמְצַע הַמַּסְלוּל ד' מַעֲלוֹת. וְאִם יִהְיֶה הַמֶּרְחָק הַכָּפוּל מִל"ב מַעֲלוֹת עַד ל"ח מַעֲלוֹת תּוֹסִיף עַל אֶמְצַע הַמַּסְלוּל ה' מַעֲלוֹת. וְאִם יִהְיֶה הַמֶּרְחָק הַכָּפוּל מִל"ט מַעֲלוֹת עַד מ"ה מַעֲלוֹת תּוֹסִיף עַל אֶמְצַע הַמַּסְלוּל שֵׁשׁ מַעֲלוֹת. וְאִם יִהְיֶה הַמֶּרְחָק הַכָּפוּל מִמ"ו מַעֲלוֹת עַד נ"א מַעֲלוֹת תּוֹסִיף עַל אֶמְצַע הַמַּסְלוּל שֶׁבַע מַעֲלוֹת. וְאִם יִהְיֶה הַמֶּרְחָק הַכָּפוּל מִנ"ב מַעֲלוֹת עַד נ"ט מַעֲלוֹת תּוֹסִיף עַל אֶמְצַע הַמַּסְלוּל ח' מַעֲלוֹת. וְאִם יִהְיֶה הַמֶּרְחָק הַכָּפוּל מִס' מַעֲלוֹת עַד ס"ג מַעֲלוֹת תּוֹסִיף עַל אֶמְצַע הַמַּסְלוּל ט' מַעֲלוֹת. וּמַה שֶּׁיִּהְיֶה אֶמְצַע הַמַּסְלוּל אַחַר שֶׁתּוֹסִיף עָלָיו מַעֲלוֹת אֵלּוּ הוּא הַנִּקְרָא מַסְלוּל הַנָּכוֹן:
📐 טבלת תיקון האנומליה
על פי הריחוק הכפול, מוסיפים מעלות לאנומליה הממוצעת: ב5° או סמוך — אין תוספת; 6°-11° → מוסיפים 1°; 12°-18° → 2°; 19°-24° → 3°; 25°-31° → 4°; 32°-38° → 5°; 39°-45° → 6°; 46°-51° → 7°; 52°-59° → 8°; 60°-63° → 9°. התוצאה לאחר תוספת זו נקראת האנומליה המתוקנת.

ערך תיקון האנומליה והמיקום האמיתי

הלכות ד׳–ז׳
הלכה ד׳
וְאַחַר כָּךְ תִּרְאֶה כַּמָּה מַעֲלוֹת הוּא הַמַּסְלוּל הַנָּכוֹן. אִם הָיָה פָּחוֹת מִק"פ מַעֲלוֹת תִּגְרַע מְנַת הַמַּסְלוּל הַזֶּה הַנָּכוֹן מֵאֶמְצַע הַיָּרֵחַ לִשְׁעַת הָרְאִיָּה. וְאִם הָיָה הַמַּסְלוּל הַנָּכוֹן יוֹתֵר עַל ק"פ מַעֲלוֹת עַד ש"ס תּוֹסִיף מְנַת זֶה הַמַּסְלוּל הַנָּכוֹן עַל אֶמְצַע הַיָּרֵחַ לִשְׁעַת הָרְאִיָּה. וּמַה שֶּׁיִּהְיֶה הָאֶמְצַע אַחַר שֶׁתּוֹסִיף עָלָיו אוֹ תִּגְרַע מִמֶּנּוּ הוּא מְקוֹם הַיָּרֵחַ הָאֲמִתִּי לִשְׁעַת הָרְאִיָּה:
הלכה ה׳
וְדַע שֶׁאִם יִהְיֶה הַמַּסְלוּל הַנָּכוֹן ק"פ בְּשָׁוֶה אוֹ ש"ס בְּשָׁוֶה אֵין לוֹ מָנָה. אֶלָּא יִהְיֶה מְקוֹם הַיָּרֵחַ הָאֶמְצָעִי לִשְׁעַת הָרְאִיָּה הוּא מָקוֹם הָאֲמִתִּי:
הלכה ו׳
וְכַמָּה הִיא מְנַת הַמַּסְלוּל. אִם יִהְיֶה הַמַּסְלוּל הַנָּכוֹן עֶשֶׂר מַעֲלוֹת תִּהְיֶה מְנָתוֹ נ' חֲלָקִים. וְאִם יִהְיֶה הַמַּסְלוּל הַנָּכוֹן כ' מַעֲלוֹת תִּהְיֶה מְנָתוֹ מַעֲלָה אַחַת וְל"ח חֲלָקִים. וְאִם יִהְיֶה שְׁלֹשִׁים תִּהְיֶה מְנָתוֹ שְׁתֵּי מַעֲלוֹת וְכ"ד חֲלָקִים. וְאִם יִהְיֶה מ' תִּהְיֶה מְנָתוֹ שָׁלֹשׁ מַעֲלוֹת וְשִׁשָּׁה חֲלָקִים. וְאִם יִהְיֶה נ' תִּהְיֶה מְנָתוֹ שָׁלֹשׁ מַעֲלוֹת וּמ"ד חֲלָקִים. וְאִם יִהְיֶה ס' תִּהְיֶה מְנָתוֹ אַרְבַּע מַעֲלוֹת וְט"ז חֲלָקִים. וְאִם יִהְיֶה ע' תִּהְיֶה מְנָתוֹ אַרְבַּע מַעֲלוֹת וּמ"א חֲלָקִים. וְאִם יִהְיֶה פ' תִּהְיֶה מְנָתוֹ חָמֵשׁ מַעֲלוֹת. וְאִם יִהְיֶה צ' תִּהְיֶה מְנָתוֹ חָמֵשׁ מַעֲלוֹת וְה' חֲלָקִים. וְאִם יִהְיֶה ק' תִּהְיֶה מְנָתוֹ ה' מַעֲלוֹת וְח' חֲלָקִים. וְאִם יִהְיֶה ק"י תִּהְיֶה מְנָתוֹ ד' מַעֲלוֹת וְנ"ט חֲלָקִים. וְאִם יִהְיֶה ק"כ תִּהְיֶה מְנָתוֹ ד' מַעֲלוֹת וְכ' חֲלָקִים. וְאִם יִהְיֶה ק"ל תִּהְיֶה מְנָתוֹ ד' מַעֲלוֹת וְי"א חֲלָקִים. וְאִם יִהְיֶה ק"מ תִּהְיֶה מְנָתוֹ ג' מַעֲלוֹת וְל"ג חֲלָקִים. וְאִם יִהְיֶה ק"נ תִּהְיֶה מְנָתוֹ שָׁלֹשׁ מַעֲלוֹת וּמ"ח חֲלָקִים. וְאִם יִהְיֶה ק"ס תִּהְיֶה מְנָתוֹ מַעֲלָה אַחַת וְנ"ו חֲלָקִים. וְאִם יִהְיֶה ק"ע תִּהְיֶה מְנָתוֹ מַעֲלָה אַחַת וְנ"ט חֲלָקִים. וְאִם יִהְיֶה ק"פ בְּשָׁוֶה אֵין לוֹ מָנָה כְּמוֹ שֶׁאָמַרְנוּ אֶלָּא מְקוֹם הַיָּרֵחַ הָאֶמְצָעִי הוּא הַמָּקוֹם הָאֲמִתִּי:
הלכה ז׳
וְאִם יִהְיֶה הַמַּסְלוּל הַנָּכוֹן יוֹתֵר עַל ק"פ מַעֲלוֹת. תִּגְרַע אוֹתוֹ מִש"ס וְתֵדַע מְנָתוֹ כְּדֶרֶךְ שֶׁעָשִׂיתָ בְּמַסְלוּל הַשֶּׁמֶשׁ. וְכֵן אִם יִהְיוּ בְּמִנְיַן הַמַּסְלוּל אֲחָדִים עִם הָעֲשָׂרוֹת תִּקַּח מִן הַיּוֹתֵר שֶׁבֵּין שְׁתֵּי הַמָּנוֹת הָאֲחָדִים. כְּדֶרֶךְ שֶׁבֵּאַרְנוּ בְּמַסְלוּל הַשֶּׁמֶשׁ בַּמָּנוֹת שֶׁלּוֹ כָּךְ תַּעֲשֶׂה בַּמַּסְלוּל הַנָּכוֹן בַּמָּנוֹת שֶׁלּוֹ:
➕ להפחית או להוסיף את התיקון
אם האנומליה המתוקנת פחות מ-180° — מפחיתים את התיקון מהירח הממוצע; אם יותר מ-180° — מוסיפים. אם בדיוק 180° או 360° — אין תיקון; המיקום הממוצע הוא המיקום האמיתי. טבלת תיקונים: 10°→נ׳, 20°→1°38', 30°→2°24', 40°→3°6', 50°→3°44', 60°→4°16', 70°→4°41', 80°→5°0', 90°→5°5', 100°→5°8' (מקסימום), 110°→4°59', 120°→4°20', 130°→4°11', 140°→3°33', 150°→2°48', 160°→1°56', 170°→נ״ט. לאנומליה מתוקנת מעל 180° — מפחיתים מ-360° ומשתמשים בשארית.

הלכה ח׳
כֵּיצַד. הֲרֵי שֶׁרָצִינוּ לֵידַע מְקוֹם הַיָּרֵחַ הָאֲמִתִּי בִּתְחִלַּת לֵיל עֶרֶב שַׁבָּת שֶׁיּוֹמוֹ שֵׁנִי לְחֹדֶשׁ אִיָּר מִשָּׁנָה זוֹ שֶׁהִיא שְׁנַת הָעִקָּר. וּמִנְיַן הַיָּמִים הַגְּמוּרִים מִתְּחִלַּת לֵיל הָעִקָּר עַד תְּחִלַּת לַיִל זֶה שֶׁאָנוּ רוֹצִים לֵידַע מְקוֹם הַיָּרֵחַ הָאֲמִתִּי בּוֹ כ"ט יוֹם. תּוֹצִיא אֶמְצַע הַשֶּׁמֶשׁ תְּחִלַּת לַיִל זֶה. יֵצֵא לְךָ אֶמְצָעוֹ ל"ה מַעֲלוֹת וְל"ח חֲלָקִים וְל"ג שְׁנִיּוֹת. סִימָנָם ל"ה ל"ח ל"ג. וְתוֹצִיא אֶמְצַע הַיָּרֵחַ לִשְׁעַת הָרְאִיָּה לְעֵת זוֹ. יֵצֵא לְךָ אֶמְצָעוֹ נ"ג מַעֲלוֹת וְל"ו חֲלָקִים וְל"ט שְׁנִיּוֹת. סִימָנָם נ"ג ל"ו ל"ט. וְתוֹצִיא אֶמְצַע הַמַּסְלוּל לְעֵת זוֹ יֵצֵא לְךָ אֶמְצָעוֹ ק"ג מַעֲלוֹת וְכ"א חֲלָקִים וּמ"ו שְׁנִיּוֹת. סִימָנָם ק"ג כ"א מ"ו. תִּגְרַע אֶמְצַע הַשֶּׁמֶשׁ מֵאֶמְצַע הַיָּרֵחַ יִשָּׁאֵר י"ז מַעֲלוֹת וְנ"ח חֲלָקִים וְשֵׁשׁ שְׁנִיּוֹת. וְזֶה הוּא הַמֶּרְחָק. תִּכְפּל אוֹתוֹ יֵצֵא לְךָ הַמֶּרְחָק הַכָּפוּל ל"ה מַעֲלוֹת וְנ"ו חֲלָקִים וְי"ב שְׁנִיּוֹת. סִימָנָם ל"ה נ"ו י"ב. לְפִיכָךְ תּוֹסִיף עַל אֶמְצַע הַמַּסְלוּל חָמֵשׁ מַעֲלוֹת כְּמוֹ שֶׁהוֹדַעְנוּ וְיֵצֵא לְךָ הַמַּסְלוּל הַנָּכוֹן ק"פ מַעֲלוֹת וְכ"א חֲלָקִים. וְאֵין מַקְפִּידִין עַל הַחֲלָקִים בְּמַסְלוּל כְּדֶרֶךְ שֶׁבֵּאַרְנוּ בַּשֶּׁמֶשׁ:
הלכה ט׳
וּבָאנוּ לַחְקֹר עַל מְנַת זֶה הַמַּסְלוּל הַנָּכוֹן שֶׁהוּא ק"ח נִמְצֵאת מָנָה שֶׁלּוֹ חָמֵשׁ מַעֲלוֹת וְחֵלֶק אֶחָד. וּלְפִי שֶׁהַמַּסְלוּל הַנָּכוֹן הָיָה פָּחוֹת מִק"פ תִּגְרַע הַמָּנָה שֶׁהוּא חָמֵשׁ מַעֲלוֹת וְחֵלֶק אֶחָד מִן אֶמְצַע הַיָּרֵחַ. יִשָּׁאֵר מ"ח מַעֲלוֹת וְל"ג חֲלָקִים וְל"ט שְׁנִיּוֹת. תַּעֲשֶׂה הַשְּׁנִיּוֹת חֵלֶק וְתוֹסִיף עַל הַחֲלָקִים. וְנִמְצָא מְקוֹם הַיָּרֵחַ הָאֲמִתִּי בְּשָׁעָה זוֹ בְּמַזַּל שׁוֹר בְּי"ח מַעֲלוֹת וְל"ו חֲלָקִים מִמַּעֲלַת י"ט. סִימָנָם י"ח ל"ו. וְעַל הַדֶּרֶךְ הַזֶּה תֵּדַע מְקוֹם הַיָּרֵחַ הָאֲמִתִּי בְּכָל עֵת רְאִיָּה שֶׁתִּרְצֶה מִתְּחִלַּת שָׁנָה זוֹ שֶׁהִיא הָעִקָּר עַד סוֹף הָעוֹלָם:
🔭 חישוב מלא
מציאת מיקום הירח האמיתי לליל שבת, ב׳ באייר של שנת העידן (29 יום מהעידן). שמש ממוצעת: 35°38'33''; ירח ממוצע: 53°36'39''; אנומליה ממוצעת: 103°21'46''. ריחוק: 17°58'6''; ריחוק כפול: 35°56'12''. מוסיפים 5° → אנומליה מתוקנת 108°21'. תיקון ל-108° = 5°1'. מפחיתים מהירח הממוצע → מיקום הירח האמיתי: 18°36' בשור (המעלה ה-19).

🎓 עקרונות מרכזיים

פרק 15
🌙
הריחוק הכפול הוא השער
'הריחוק הכפול' — פעמיים ההפרש בין הירח הממוצע לשמש הממוצעת — הוא הכלי המרכזי: ערכו בליל הראייה תמיד חסום בין 5° ל-62°.
📊
שתי טבלאות שיטתיות
הרמב"ם מספק טבלאות מדויקות הן לתיקון האנומליה (עד 9° תוספת) והן לערך תיקון האנומליה המתוקנת (עד 5°8'). ערכי ביניים משתמשים באינטרפולציה יחסית.
⚖️
מתחת ל-180° מפחיתים, מעל 180° מוסיפים
כשהאנומליה המתוקנת בחצי הראשון (מתחת ל-180°), הירח קרוב לאפוגיי והתיקון מפחית; בחצי השני (מעל 180°), הוא מתקרב לפריגיי ומוסיף. בדיוק 180° או 360° — אין תיקון כלל.
🔭
מחישוב לשמיים
הרמב"ם מסיים בדוגמה מחושבת מלאה שמראה כיצד כל שלבי החישוב מתחברים לתוצאה אחת: מיקום הירח האמיתי בזמן הראייה — שיטה תקפה 'מתחילת שנה זו ועד סוף העולם.'
