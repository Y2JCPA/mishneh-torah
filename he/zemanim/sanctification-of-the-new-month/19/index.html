נטיית גלגל המזלות — יסודות נטיית המזלות

הלכות א׳–ו׳
הלכה א׳
לְפִי שֶׁאָמְרוּ חֲכָמִים שֶׁבּכְלַל דִּבָרִים שֶׁהָיוּ בּוֹדְקִין בָּהֶן אֶת הָעֵדִים אוֹמְרִין לָהֶן לְהֵיכָן הָיָה הַיָּרֵחַ נוֹטֶה. כָּשֵׁר בְּעֵינַי לְהוֹדִיעַ דֶּרֶךְ חֶשְׁבּוֹן דָּבָר זֶה. וְאֵין אֲנִי מְדַקְדֵּק בּוֹ לְפִי שֶׁאֵינוֹ מוֹעִיל בִּרְאִיָּה כְּלָל. וּתְחִלַּת חֶשְׁבּוֹן זֶה לֵידַע נְטִיַּת הַמַּזָּלוֹת תְּחִלָּה:
הלכה ב׳
הָעֲגֻלָּה שֶׁהִיא עוֹבֶרֶת בְּמַחֲצִית הַמַּזָּלוֹת שֶׁבָּהּ מַהֲלָךְ הַשֶּׁמֶשׁ. אֵינָהּ עוֹבֶרֶת בְּאֶמְצַע הָעוֹלָם מֵחֲצִי הַמִּזְרָח לַחֲצִי הַמַּעֲרָב. אֶלָּא נוֹטָה הִיא מֵעַל הַקַּו הַשָּׁוֶה הַמְסַבֵּב בְּאֶמְצַע הָעוֹלָם כְּנֶגֶד צָפוֹן וְדָרוֹם. חֶצְיָהּ נוֹטֶה לַצָּפוֹן וְחֶצְיָהּ נוֹטֶה לַדָּרוֹם:
הלכה ג׳
וּשְׁתֵּי נְקֻדּוֹת יֵשׁ בָּהּ שֶׁפּוֹגַעַת בָּהֶן בַּעֲגֻלַּת הַקַּו הַשָּׁוֶה הַמְסַבֵּב בְּאֶמְצַע הָעוֹלָם. הַנְּקֻדָּה הָאַחַת רֹאשׁ מַזַּל טָלֶה. הַנְּקֻדָּה הַשְּׁנִיָּה שֶׁכְּנֶגְדָּהּ רֹאשׁ מַזַּל מֹאזְנַיִם. וְנִמְצְאוּ שִׁשָּׁה מַזָּלוֹת נוֹטוֹת לַצָּפוֹן מִתְּחִלַּת טָלֶה עַד סוֹף בְּתוּלָה. וְשִׁשָּׁה נוֹטוֹת לַדָּרוֹם מִתְּחִלַּת מַזַּל מֹאזְנַיִם עַד סוֹף מַזַּל דָּגִים:
הלכה ד׳
וּמֵרֹאשׁ מַזַּל טָלֶה יַתְחִילוּ הַמַּזָּלוֹת לִנְטוֹת מְעַט מְעַט וּלְהִתְרַחֵק מֵעַל הַקַּו הַשָּׁוֶה כְּנֶגֶד הַצָּפוֹן עַד רֹאשׁ סַרְטָן. וְיִהְיֶה רֹאשׁ סַרְטָן רָחוֹק מֵעַל הַקַּו הַשָּׁוֶה לְרוּחַ הַצָּפוֹן שָׁלֹשׁ וְעֶשְׂרִים מַעֲלוֹת וַחֲצִי מַעֲלָה בְּקֵרוּב. וְיַחְזְרוּ הַמַּזָּלוֹת לְהִתְקָרֵב לַקַּו הַשָּׁוֶה מְעַט מְעַט עַד רֹאשׁ מֹאזְנַיִם שֶׁהוּא עַל הַקַּו הַשָּׁוֶה. וּמֵרֹאשׁ מֹאזְנַיִם יַתְחִילוּ לִנְטוֹת וּלְהִתְרַחֵק כְּנֶגֶד רוּחַ דָּרוֹם עַד רֹאשׁ גְּדִי. וְיִהְיֶה רֹאשׁ גְּדִי רָחוֹק מֵעַל הַקַּו הַשָּׁוֶה לְרוּחַ דָּרוֹם שָׁלֹשׁ וְעֶשְׂרִים מַעֲלוֹת וַחֲצִי מַעֲלָה. וְיַחְזְרוּ הַמַּזָּלוֹת לְהִתְקָרֵב מְעַט מְעַט כְּנֶגֶד הַקַּו הַשָּׁוֶה עַד רֹאשׁ טָלֶה:
הלכה ה׳
נִמְצָא רֹאשׁ טָלֶה וְרֹאשׁ מֹאזְנַיִם מְסַבֵּב עַל הַקַּו הַשָּׁוֶה. וּלְפִיכָךְ כְּשֶׁתִּהְיֶה הַשֶּׁמֶשׁ בִּשְׁנֵי רָאשִׁים אֵלּוּ לֹא תִּהְיֶה נוֹטָה לֹא לַצָּפוֹן וְלֹא לַדָּרוֹם. וְתִזְרַח בַּחֲצִי מִזְרָח וְתִשְׁקַע בַּחֲצִי מַעֲרָב. וְיִהְיֶה הַיּוֹם וְהַלַּיְלָה שָׁוִין בְּכָל הַיִּשּׁוּב:
הלכה ו׳
הֲרֵי נִתְבָּרֵר לְךָ שֶׁכָּל מַעֲלָה וּמַעֲלָה מִמַּעֲלוֹת הַמַּזָּלוֹת נוֹטָה לַצָּפוֹן אוֹ לַדָּרוֹם וְיֵשׁ לִנְטִיָּתָהּ שִׁעוּר. (וְרֹב) [וְרֹחַב] הַנְּטִיָּה לֹא תִּהְיֶה יוֹתֵר עַל שָׁלֹשׁ וְעֶשְׂרִים מַעֲלוֹת וַחֲצִי בְּקֵרוּב:
🌐 נתיב השמש הנטוי
נתיב השמש אינו עובר דרך מרכז העולם אלא נוטה מעל ומתחת לקו המשווה — חצי צפונה, חצי דרומה. שתי נקודות החיתוך עם קו המשווה הן תחילת טלה ותחילת מאזניים. שישה מזלות נוטים צפונה (טלה עד בתולה), שישה דרומה (מאזניים עד דגים). הנטייה המקסימלית היא בקירוב 23.5°, ומגיעה בתחילת סרטן (צפון) ובתחילת גדי (דרום). כשהשמש בתחילת טלה או מאזניים — יום ולילה שווים בכל מקום.

טבלת הנטייה — חישוב ערכים לכל מעלה

הלכות ז׳–ט׳
הלכה ז׳
וְאֵלּוּ הֵם הַשִּׁעוּרִים שֶׁל נְטִיּוֹת לְפִי מִנְיַן הַמַּעֲלוֹת שֶׁל מַזָּלוֹת. וְהַתְחָלָה מַתְחֶלֶת מַזַּל טָלֶה. י' מַעֲלוֹת נְטִיָּתָם ד' מַעֲלוֹת. כ' מַעֲלוֹת נְטִיָּתָם ח' מַעֲלוֹת. ל' מַעֲלוֹת נְטִיָּתָם י"א מַעֲלוֹת וּמֶחֱצָה. וּמ' מַעֲלוֹת נְטִיָּתָם ט"ו מַעֲלוֹת. נ' מַעֲלוֹת נְטִיָּתָם י"ח מַעֲלוֹת. ס' מַעֲלוֹת נְטִיָּתָם כ' מַעֲלוֹת. ע' מַעֲלוֹת נְטִיָּתָם כ"ב מַעֲלוֹת. פ' מַעֲלוֹת נְטִיָּתָם כ"ג מַעֲלוֹת. צ' מַעֲלוֹת נְטִיָּתָם כ"ג מַעֲלוֹת וַחֲצִי מַעֲלָה:
הלכה ח׳
וְאִם יִהְיוּ אֲחָדִים בְּמִנְיָן תִּקַּח לָהֶם מְנָתָם מִבֵּין שְׁתֵּי הַנְּטִיּוֹת כְּמוֹ שֶׁבֵּאַרְנוּ בַּשֶּׁמֶשׁ וּבַיָּרֵחַ. כֵּיצַד. חָמֵשׁ מַעֲלוֹת נְטִיָּתָם שְׁתֵּי מַעֲלוֹת. וְאִם הָיָה מִנְיַן הַמַּעֲלוֹת כ"ג נְטִיָּתָם ט' מַעֲלוֹת. וְעַל דֶּרֶךְ זוֹ בְּכָל הָאֲחָדִים שֶׁהֵן עִם הָעֲשָׂרוֹת:
הלכה ט׳
וּמֵאַחַר שֶׁתֵּדַע הַנְּטִיָּה שֶׁל מַעֲלוֹת מֵאַחַת עַד צ'. תֵּדַע נְטִיָּתָם כֻּלָּן כְּדֶרֶךְ שֶׁהוֹדַעְנוּךָ בְּרֹחַב הַיָּרֵחַ. שֶׁאִם הָיָה הַמִּנְיָן יֶתֶר עַל צ' עַד ק"פ תִּגְרַע אוֹתוֹ מִק"פ. וְאִם הָיָה יוֹתֵר עַל ק"פ עַד ר"ע תִּגְרַע מִמֶּנּוּ ק"פ. וְאִם הָיָה יוֹתֵר עַל ר"ע עַד ש"ס תִּגְרַע אוֹתוֹ מִש"ס. וְהַנִּשְׁאָר תֵּדַע נְטִיָּתוֹ וְהוּא נְטִיַּת אוֹתוֹ הַמִּנְיָן שֶׁבְּיָדְךָ בְּלֹא גֵּרָעוֹן וְלֹא תּוֹסֶפֶת:
📐 טבלת חיפוש וכללי צמצום
הרמב"ם מספק טבלת נטייה לכל 10°: 10°→4°, 20°→8°, 30°→11.5°, 40°→15°, 50°→18°, 60°→20°, 70°→22°, 80°→23°, 90°→23.5°. לערכי ביניים, ממצעים בין ערכים סמוכים. למיקומים מעל 90°: מפחיתים מ-180°. מעל 180°: מפחיתים ממנו 180°. מעל 270°: מפחיתים מ-360°. כך מקבלים את הנטייה המקבילה.

מרחק הירח מקו המשווה

הלכות י׳–י״א
הלכה י׳
אִם תִּרְצֶה לֵידַע כַּמָּה מַעֲלוֹת הוּא הַיָּרֵחַ נוֹטֶה מֵעַל הַקַּו הַשָּׁוֶה כְּנֶגֶד צְפוֹן הָעוֹלָם אוֹ כְּנֶגֶד דְּרוֹם הָעוֹלָם. תֵּדַע תְּחִלָּה כַּמָּה נְטִיַּת הַמַּעֲלָה שֶׁהִיא מְקוֹם הַיָּרֵחַ הָאֲמִתִּי וּלְאֵי זֶה רוּחַ הִיא נוֹטָה לַצָּפוֹן אוֹ לַדָּרוֹם. וְתַחֲזֹר וְתַחְשֹׁב וְתוֹצִיא רֹחַב הַיָּרֵחַ הָרִאשׁוֹן וְתִרְאֶה אִם הוּא צְפוֹנִי אוֹ דְּרוֹמִי. אִם נִמְצְאוּ רֹחַב הַיָּרֵחַ וּנְטִיַּת מַעֲלָתוֹ בְּרוּחַ אַחַת כְּגוֹן שֶׁהָיוּ שְׁנֵיהֶם צְפוֹנִים אוֹ דְּרוֹמִיִּים תְּקַבֵּץ שְׁנֵיהֶם. וְאִם נִמְצְאוּ בִּשְׁתֵּי רוּחוֹת כְּגוֹן שֶׁהָיָה הָאֶחָד דְּרוֹמִי וְהָאֶחָד צְפוֹנִי. תִּגְרַע הַמְּעַט מִשְּׁנֵיהֶם מִן הָרַב וְהַנִּשְׁאָר הוּא מֶרְחַק הַיָּרֵחַ מֵעַל הַקַּו הַשָּׁוֶה בְּאוֹתָהּ הָרוּחַ שֶׁהָיָה בָּהּ הָרַב בִּשְׁנֵיהֶם:
הלכה י״א׳
כֵּיצַד בָּאנוּ לֵידַע כַּמָּה הַיָּרֵחַ נוֹטֶה מֵעַל הַקַּו הַשָּׁוֶה בְּלֵיל הָרְאִיָּה שֶׁהוּא שֵׁנִי לְחֹדֶשׁ אִיָּר מִשָּׁנָה זוֹ. וּכְבָר יָדַעְתָּ שֶׁמַּעֲלַת הַיָּרֵחַ הָיְתָה י"ט מִמַּזַּל שׁוֹר. נְטִיָּתָהּ בַּצָּפוֹן כְּמוֹ י"ח מַעֲלוֹת. וְרֹחַב הַיָּרֵחַ הָיָה בַּדָּרוֹם כְּמוֹ ד' מַעֲלוֹת. תִּגְרַע הַמְּעַט מִן הָרַב יִשָּׁאֵר י"ד מַעֲלוֹת. וְנִמְצָא הַיָּרֵחַ רָחוֹק מֵעַל הַקַּו הַשָּׁוֶה י"ד מַעֲלוֹת לְרוּחַ צָפוֹן. שֶׁהֲרֵי הַמִּנְיָן הָרַב שֶׁהוּא שְׁמוֹנֶה עֶשְׂרֵה מַעֲלוֹת הָיָה צְפוֹנִי. וְכָל חֶשְׁבּוֹן זֶה בְּקֵרוּב בְּלֹא דִּקְדּוּק לְפִי שֶׁאֵינוֹ מוֹעִיל בִּרְאִיָּה:
🌙 שילוב נטייה ורוחב
למציאת מרחק הירח מקו המשווה: (1) מחשבים את נטיית מיקום הירח האמיתי במזלות (צפון או דרום); (2) מוצאים את הרוחב הראשון של הירח (צפון או דרום). אם שניהם באותו כיוון — מחברים; אם בכיוונים מנוגדים — מפחיתים את הקטן מהגדול. השארית היא מרחק הירח מקו המשווה, בכיוון הערך הגדול. דוגמה מחושבת: ב׳ באייר, ירח ב-19° שור — נטייה 18° צפון, רוחב 4° דרום; תוצאה: 14° צפון.

כיוון הירח הנראה לעדים

הלכות י״ב–י״ד
הלכה י״ב׳
אִם תִּרְצֶה לֵידַע לְאֵי זוֹ רוּחַ מֵרוּחוֹת הָעוֹלָם יֵרָאֶה הַיָּרֵחַ נוֹטֶה. תַּחְשֹׁב וְתֵדַע מֶרְחַקּוֹ מֵעַל הַקַּו הַשָּׁוֶה. אִם יִהְיֶה עַל הַקַּו הַשָּׁוֶה אוֹ קָרוֹב מִמֶּנּוּ בִּשְׁתַּיִם אוֹ שָׁלֹשׁ מַעֲלוֹת בַּצָּפוֹן אוֹ בַּדָּרוֹם. יֵרָאֶה מְכֻוָּן כְּנֶגֶד אֶמְצַע מַעֲרָב וְתֵרָאֶה פְּגִימָתוֹ מְכֻוֶּנֶת כְּנֶגֶד מִזְרַח הָעוֹלָם בְּשָׁוֶה:
הלכה י״ג׳
וְאִם יִהְיֶה רָחוֹק מֵעַל הַקַּו הַשָּׁוֶה לִצְפוֹן הָעוֹלָם. יֵרָאֶה בֵּין מַעֲרַב הָעוֹלָם וּבֵין צְפוֹנוֹ. וְתֵרָאֶה פְּגִימָתוֹ נוֹטָה מִכְּנֶגֶד מִזְרַח הָעוֹלָם כְּנֶגֶד דְּרוֹם הָעוֹלָם:
הלכה י״ד׳
וְאִם הָיָה רָחוֹק מֵעַל הַקַּו הַשָּׁוֶה לִדְרוֹם הָעוֹלָם. יֵרָאֶה בֵּין מַעֲרַב הָעוֹלָם וּבֵין דְּרוֹמוֹ. וְתֵרָאֶה פְּגִימָתוֹ נוֹטָה מִכְּנֶגֶד מִזְרַח הָעוֹלָם כְּנֶגֶד צְפוֹן הָעוֹלָם. וּלְפִי רֹב הַמֶּרְחָק וּלְפִי רֹב הַנְּטִיָּה:
👁️ מה העדים רואים
מרחק הירח מקו המשווה קובע את כיוונו הנראה: על קו המשווה או בטווח 2-3° ממנו — הוא נראה ישירות במערב, כשקעירותו פונה מזרחה. אם צפונית לקו המשווה — הוא נראה בין מערב לצפון, קעירותו נוטה דרומית-מזרחית. אם דרומית — בין מערב לדרום, קעירותו נוטה צפונית-מזרחית. ככל שהמרחק גדול יותר, כך הנטייה גדולה יותר.
7/9

גובה הירח וחתימת הלכות אלו

הלכות ט״ו–ט״ז
הלכה ט״ו׳
וּמֵחֲקִירַת הָעֵדִים שֶׁאוֹמְרִין לָהֶם כַּמָּה הָיָה גָּבוֹהַּ. וְדָבָר זֶה יִוָּדַע מִקֶּשֶׁת הָרְאִיָּה. שֶׁבִּזְמַן שֶׁתִּהְיֶה קֶשֶׁת הָרְאִיָּה קְצָרָה יֵרָאֶה הַיָּרֵחַ כְּאִלּוּ הוּא קָרוֹב מִן הָאָרֶץ. וּבִזְמַן שֶׁתִּהְיֶה אֲרֻכָּה יֵרָאֶה גָּבוֹהַּ מֵעַל הָאָרֶץ. וּלְפִי אֹרֶךְ קֶשֶׁת הָרְאִיָּה לְפִי גָּבְהוֹ מֵעַל הָאָרֶץ בִּרְאִיַּת הָעֵינַיִם:
הלכה ט״ז׳
הֲרֵי בֵּאַרְנוּ חֶשְׁבּוֹנוֹת כָּל הַדְּרָכִים שֶׁצְּרִיכִין לָהֶם בִּידִיעַת הָרְאִיָּה וּבַחֲקִירַת הָעֵדִים. כְּדֵי שֶׁיִּהְיֶה הַכּל יָדוּעַ לַמְּבִינִים וְלֹא יֶחְסְרוּ דֶּרֶךְ מִדַּרְכֵי הַתּוֹרָה וְלֹא יְשׁוֹטְטוּ לְבַקֵּשׁ אַחֲרֶיהָ בִּסְפָרִים אֲחֵרִים. (ישעיה לד טז) "דִּרְשׁוּ מֵעַל סֵפֶר ה'" וְקִרְאוּ, אַחַת מֵהֵנָּה לֹא נֶעְדָּרָה: סָלִיק הִלְכוֹת קִדּוּשׁ הַחֹדֶשׁ
📜 חותמים את הספר
תשובת העדים ל'כמה גבוה היה?' נקבעת על ידי קשת הראייה: קשת קצרה פירושה הירח נראה נמוך ליד האופק; קשת ארוכה פירושה הוא נראה גבוה. הרמב"ם מסיים: 'ביארנו את כל החשבונות בכל הדרכים שצריך לידיעת הראייה' — כדי שלא יצטרכו חכמים לחפש בספרים אחרים. הוא מצטט ישעיהו לד:טז: 'דרשו מעל ספר ה׳.' סוף הלכות קידוש החודש.
8/9

🎓 עקרונות מרכזיים

פרק 19
🌐
נתיב השמש נוטה 23.5° מקו המשווה
גלגל המזלות אינו עובר ישירות ממזרח למערב אלא נוטה עד 23.5° צפונה ודרומה — עיקרון חיוני להבנת כיוון הירח הנראה.
📐
נטייה דרך טבלת חיפוש עם צמצום
הרמב"ם מספק ערכים מדויקים לכל 10° וכללים פשוטים לצמצום כל מיקום מעל 90° חזרה לטווח 0-90°.
🌙
משלבים נטייה ורוחב לפי כיוון
אם נטיית המזל של הירח והרוחב שלו מצביעים לאותו כיוון — מחברים; אם לכיוונים מנוגדים — מפחיתים. התוצאה קובעת היכן בשמיים העדים רואים את הירח.
📜
תורה עצמאית: אין צורך במקורות חיצוניים
הרמב"ם כלל חישובים אסטרונומיים אלו במשנה תורה כדי שחכמים ימצאו הכל בתוך התורה עצמה — ולעולם לא יצטרכו לעיין בספרי אסטרונומיה של גויים.
