הלכות א׳–ז׳
הלכה א׳
חֲצָיֵי הַשְּׁקָלִים הַכּל צְרִיכִין לָהֶן כְּדֵי שֶׁיִּתֵּן כָּל אֶחָד וְאֶחָד חֲצִי שֶׁקֶל שֶׁהוּא חַיָּב. לְפִיכָךְ כְּשֶׁהָיָה אָדָם הוֹלֵךְ אֵצֶל הַשֻּׁלְחָנִי וּמְצָרֵף שֶׁקֶל בִּשְׁנֵי חֲצָאִין יִתֵּן לוֹ תּוֹסֶפֶת עַל הַשֶּׁקֶל. וְאוֹתָהּ הַתּוֹסֶפֶת נִקְרֵאת קַלְבּוֹן. לְפִיכָךְ שְׁנַיִם שֶׁנָּתְנוּ שֶׁקֶל עַל שְׁנֵיהֶם חַיָּבִין בְּקַלְבּוֹן:
הלכה ב׳
כָּל מִי שֶׁאֵינוֹ חַיָּב בִּשְׁקָלִים כְּגוֹן שְׁתֵּי נָשִׁים אוֹ שְׁנֵי עֲבָדִים שֶׁנָּתְנוּ שֶׁקֶל פְּטוּרִים מִן הַקַּלְבּוֹן. וְכֵן אִם הָיָה אֶחָד חַיָּב וְאֶחָד פָּטוּר וְנָתַן הַחַיָּב עַל יַד הַפְּטוּר כְּגוֹן אִישׁ שֶׁנָּתַן שֶׁקֶל עָלָיו וְעַל אִשָּׁה אוֹ עֶבֶד פָּטוּר מִן הַקַּלְבּוֹן. וְכֵן הַכֹּהֲנִים פְּטוּרִין מִן הַקַּלְבּוֹן וְהַשּׁוֹקֵל עַל יַד הַכֹּהֵן פָּטוּר מִן הַקַּלְבּוֹן:
הלכה ג׳
הַנּוֹתֵן שֶׁקֶל עָלָיו וְעַל הֶעָנִי אוֹ עַל שְׁכֵנוֹ אוֹ עַל בֶּן עִירוֹ. אִם נְתָנוֹ לָהֶם מַתָּנָה פָּטוּר מִן הַקַּלְבּוֹן. שֶׁהֲרֵי נָתַן חֲצִי שֶׁקֶל מַתָּנָה כְּדֵי לְהַרְבּוֹת בִּשְׁקָלִים. וְאִם נָתַן לָהֶם הַחֲצִי שֶׁשָּׁקַל עַל יָדָם דֶּרֶךְ הַלְוָאָה עַד שֶׁיַּחֲזִירוּ לוֹ כְּשֶׁתִּמְצָא יָדָם חַיָּב הַקַּלְבּוֹן:
הלכה ד׳
הָאַחִין שֶׁעֲדַיִן לֹא חָלְקוּ מַה שֶּׁהִנִּיחַ לָהֶם אֲבִיהֶם. וְכֵן הַשֻּׁתָּפִים שֶׁנָּתְנוּ שֶׁקֶל עַל יְדֵי שְׁנֵיהֶם פְּטוּרִין מִן הַקַּלְבּוֹן. בַּמֶּה דְּבָרִים אֲמוּרִים בְּשֻׁתָּפִין שֶׁנָּשְׂאוּ וְנָתְנוּ בִּמְעוֹת הַשֻּׁתָּפוּת וְנִשְׁתַּנָּה עֵין הַמָּעוֹת. אֲבָל אִם הֵבִיא זֶה מְעוֹתָיו וְזֶה מְעוֹתָיו וְעֵרְבוּם וַעֲדַיִן לֹא נִשְׁתַּנּוּ הַמָּעוֹת וְלֹא הוֹצִיאוּם הֲרֵי אֵלּוּ חַיָּבִין בְּקַלְבּוֹן. נָשְׂאוּ וְנָתְנוּ וְאַחַר זְמַן חָלְקוּ וְחָזְרוּ וְנִשְׁתַּתְּפוּ חַיָּבִין בְּקַלְבּוֹן עַד שֶׁיִּשְּׂאוּ וְיִתְּנוּ בְּשֻׁתָּפוּת זוֹ הָאַחֲרוֹנָה וְיִשְׁתַּנּוּ הַמָּעוֹת:
הלכה ה׳
הָאַחִין וְהַשֻּׁתָּפִין שֶׁהָיָה לָהֶן בְּהֵמָה וּכְסָפִים וְחָלְקוּ בַּכְּסָפִים חַיָּבִים בְּקַלְבּוֹן. אַף עַל פִּי שֶׁעֲדַיִן לֹא חָלְקוּ הַבְּהֵמָה. וְאִם חָלְקוּ הַבְּהֵמָה וְלֹא חָלְקוּ הַכְּסָפִים פְּטוּרִין מִן הַקַּלְבּוֹן עַד שֶׁיַּחְלְקוּ הַכְּסָפִים וְאֵין אוֹמְרִין הֲרֵי הֵם עוֹמְדִין לַחֲלוּקָה:
הלכה ו׳
הַנּוֹתֵן שֶׁקֶל לְהֶקְדֵּשׁ כְּדֵי שֶׁיְּחַשְּׁבוּ לוֹ מַחֲצִית הַשֶּׁקֶל שֶׁהוּא חַיָּב בָּהּ וְיִטּל חֲצִי שֶׁקֶל מִמַּה שֶּׁנִּגְבָּה מִן הָאֲחֵרִים חַיָּב שְׁנֵי קַלְבּוֹנוֹת. שֶׁאִלּוּ הָיָה הַשֶּׁקֶל כֻּלּוֹ לִשְׁקָלִים הָיָה חַיָּב קַלְבּוֹן אֶחָד:
הלכה ז׳
כַּמָּה הוּא שִׁעוּר הַקַּלְבּוֹן. בִּזְמַן שֶׁהָיוּ נוֹתְנִין בְּמַחֲצִית הַשֶּׁקֶל שְׁנֵי דִּינָרִין הָיָה הַקַּלְבּוֹן חֲצִי מָעָה שֶׁהוּא אֶחָד מִשְּׁנֵים עָשָׂר בְּדִינָר. וּמֵעוֹלָם לֹא נִתַּן הַקַּלְבּוֹן פָּחוֹת מִזֶּה. וְהַקַּלְבּוֹנוֹת אֵינָן כִּשְׁקָלִים אֶלָּא מַנִּיחִין אוֹתָן הַשֻּׁלְחָנִים בִּפְנֵי עַצְמָן עַד שֶׁיִּסְתַּפֵּק מֵהֶן הַהֶקְדֵּשׁ:
🪙 תוספת הקלבון
מי שצריך חצי שקל ומשלם שקל שלם אצל השולחני — חייב להוסיף קלבון — דמי החלפה. פטורים מקלבון: מי שאינו חייב בשקלים (נשים, עבדים), אחים ושותפים שעדיין לא חילקו את נכסיהם, ומי שנותן שקל לחברו במתנה. חייבים: שניים ששילמו שקל אחד עבור שניהם, אחים ושותפים שחילקו את כספם (גם אם לא חילקו את בהמותיהם), ומי שמוסר שקל לבית המקדש כדי לקבל חצי שקל בחזרה — חייב שני קלבונות. שיעור הקלבון: חצי מעה (שנים עשר חלק בדינר), וכסף הקלבון אינו מתקדש בקדושת שקלים.

אחריות על השקלים — שליח, שומר וניכוס

הלכות ח׳–י׳
הלכה ח׳
מִי שֶׁאָבַד שִׁקְלוֹ חַיָּב בְּאַחְרָיוּתוֹ עַד שֶׁיִּמְסְרֶנּוּ לַגִּזְבָּר. בְּנֵי הָעִיר שֶׁשָּׁלְחוּ אֶת שִׁקְלֵיהֶן בְּיַד שָׁלִיחַ וְנִגְנְבוּ אוֹ אָבְדוּ. אִם שׁוֹמֵר חִנָּם הוּא הֲרֵי זֶה נִשְׁבָּע לָהֶם וְנִפְטָר כְּדִין כָּל שׁוֹמְרֵי חִנָּם וְהֵן חוֹזְרִין וְנוֹתְנִין שִׁקְלֵיהֶן פַּעַם שְׁנִיָּה. וְאִם אָמְרוּ אַנְשֵׁי הָעִיר הוֹאִיל וְאָנוּ מְשַׁלְּמִין שְׁקָלֵינוּ אֵין רְצוֹנֵנוּ שֶׁיִּשָּׁבַע הַשָּׁלִיחַ שֶׁהוּא נֶאֱמָן לָנוּ אֵין שׁוֹמְעִין לָהֶן. לְפִי שֶׁתַּקָּנַת חֲכָמִים הִיא שֶׁאֵין הֶקְדֵּשׁ יוֹצֵא בְּלֹא שְׁבוּעָה. נִמְצְאוּ הַשְּׁקָלִים הָרִאשׁוֹנִים אַחַר שֶׁנִּשְׁבַּע הַשָּׁלִיחַ אֵלּוּ וְאֵלּוּ שְׁקָלִים הֵם וְאֵין עוֹלִין לָהֶן לְשָׁנָה אַחֶרֶת. וְהָרִאשׁוֹנִים יִפְּלוּ לְשִׁקְלֵי הַשָּׁנָה וְהָאַחֲרוֹנִים יִפְּלוּ לְשִׁקְלֵי שָׁנָה שֶׁעָבְרָה:
הלכה ט׳
שָׁלְחוּ אֶת שִׁקְלֵיהֶם בְּיַד שׁוֹמֵר שָׂכָר שֶׁהֲרֵי הוּא חַיָּב בִּגְנֵבָה וַאֲבֵדָה וְאָבְדוּ מִמֶּנּוּ בְּאֹנֶס. כְּגוֹן שֶׁלְּקָחוּם לִסְטִים מְזֻיָּנִים שֶׁהוּא פָּטוּר. רוֹאִין אִם אַחַר שֶׁנִּתְרְמָה הַתְּרוּמָה נֶאֱנַס נִשְׁבָּע הַשָּׁלִיחַ לַגִּזְבָּרִים וּבְנֵי הָעִיר פְּטוּרִין. שֶׁהַתּוֹרֵם תּוֹרֵם עַל הַגָּבוּי וְעַל הֶעָתִיד לִגָּבוֹת וּבִרְשׁוּת הֶקְדֵּשׁ הֵן וּבְנֵי הָעִיר מֶה הָיָה לָהֶן לַעֲשׂוֹת הֲרֵי לֹא מָסְרוּ אֶלָּא לְשׁוֹמֵר שָׂכָר שֶׁהוּא חַיָּב בִּגְנֵבָה וַאֲבֵדָה. אֲבָל הָאֹנֶס אֵינוֹ מָצוּי. וְאִם אָבְדוּ קֹדֶם שֶׁנִּתְרְמָה הַתְּרוּמָה עֲדַיִן בִּרְשׁוּת בְּנֵי הָעִיר הֵם וְהַשָּׁלִיחַ נִשְׁבָּע לִפְנֵי אַנְשֵׁי הָעִיר וְהֵן מְשַׁלְּמִין. נִשְׁבַּע וְגָבוּ שְׁקָלִים שֵׁנִית וְאַחַר כָּךְ הֶחֱזִירוּם הַלִּסְטִים אֵלּוּ וְאֵלּוּ שְׁקָלִים וְאֵין עוֹלִין לָהֶן לְשָׁנָה אַחֶרֶת וְהַשְּׁנִיִּים יִפְּלוּ לְשִׁקְלֵי שָׁנָה שֶׁעָבְרָה. יֵשׁ מִי שֶׁאוֹמֵר שֶׁהַשְּׁקָלִים הָרִאשׁוֹנִים שֶׁיִּפְּלוּ לְשִׁקְלֵי הַשָּׁנָה הֵם הַשְּׁקָלִים שֶׁנִּגְּבּוּ בַּתְּחִלָּה וְאָבְדוּ אוֹ נֶאֶנְסוּ וְחָזְרוּ. וְיֵשׁ מִי שֶׁאוֹמֵר שֶׁהַשְּׁקָלִים הָרִאשׁוֹנִים הֵן שֶׁהִגִּיעוּ לְיַד הַגִּזְבָּר תְּחִלָּה:
הלכה י׳
הַנּוֹתֵן חֲצִי שֶׁקֶל לַחֲבֵרוֹ לְהוֹלִיכוֹ לְשֻׁלְחָנִי לִשְׁקל אוֹתוֹ עַל יָדוֹ. הָלַךְ וּשְׁקָלוֹ עַל יְדֵי עַצְמוֹ כְּדֵי שֶׁלֹּא יְמַשְׁכְּנוּ אוֹתוֹ. אִם נִתְרְמָה הַתְּרוּמָה מֵעַל הַשּׁוֹקֵל שֶׁזֶּה הַשֶּׁקֶל בִּרְשׁוּת הֶקְדֵּשׁ הוּא שֶׁכְּבָר תְּרָמוֹ עַל הֶעָתִיד לִגְבּוֹת וְנִמְצָא זֶה הִצִּיל עַצְמוֹ בְּמָמוֹן הַהֶקְדֵּשׁ וְנֶהֱנָה בַּשֶּׁקֶל הַזֶּה. וְאִם לֹא נִתְרְמָה הַתְּרוּמָה לֹא מָעַל וְהוּא חַיָּב לִתֵּן לַחֲבֵרוֹ חֲצִי שֶׁקֶל שֶׁנָּתַן לוֹ. וְכֵן הַגּוֹזֵל אוֹ הַגּוֹנֵב חֲצִי שֶׁקֶל וּשְׁקָלוֹ יָצָא וְהוּא חַיָּב לְשַׁלֵּם לַבְּעָלִים שְׁנַיִם אוֹ לְהוֹסִיף חֹמֶשׁ:
📦 אחריות ושליחות
מי ששקלו אבד — חייב עד שיגיע לגזבר. עיר ששלחה שקלים ביד שליח: אם נגנבו — השליח משלם; אם נגנבו לאחר שהורמה התרומה — הגזבר משלם, וחובת העיר נפטרה. שלחו ביד שומר שכר ואבדו באונס — השומר פטור, והעיר צריכה לשלם שוב. אם מסר אדם חצי שקל לשליח והשליח השתמש בו לעצמו: אם כבר הורמה התרומה — שניהם עלולים להיות מחויבים לשלם.

הלכה י״א׳
הַנּוֹתֵן מַחֲצִית הַשֶּׁקֶל מִן הַהֶקְדֵּשׁ וְנִתְרְמָה הַתְּרוּמָה מִמֶּנּוּ. כְּשֶׁיִּסְתַּפְּקוּ מִמֶּנָּה יִתְחַיֵּב בִּמְעִילָה וְיָצָא יְדֵי מַחֲצִית הַשֶּׁקֶל. נְתָנוֹ מִמְּעוֹת מַעֲשֵׂר שֵׁנִי יֹאכַל כְּנֶגְדּוֹ בִּירוּשָׁלַיִם. מִדְּמֵי שְׁבִיעִית יֹאכַל כְּנֶגְדּוֹ בִּקְדֻשַּׁת שְׁבִיעִית. הָיָה שֶׁל עִיר הַנִּדַּחַת לֹא עָשָׂה כְּלוּם:
הלכה י״ב׳
הַמַּפְרִישׁ שִׁקְלוֹ וְסָבוּר שֶׁהוּא חַיָּב בּוֹ וְנִמְצָא שֶׁאֵינוֹ חַיָּב לֹא קָדַּשׁ. הַמַּפְרִישׁ שְׁנַיִם וְנִמְצָא שֶׁאֵינוֹ חַיָּב אֶלָּא אֶחָד. אִם בְּזֶה אַחַר זֶה הָאַחֲרוֹן לֹא קָדַּשׁ. וְאִם בְּבַת אַחַת הָאֶחָד שְׁקָלִים וְהַשֵּׁנִי מוֹתַר שְׁקָלִים. הִפְרִישׁ שִׁקְלוֹ וָמֵת יִפּל לִנְדָבָה:
הלכה י״ג׳
הַלּוֹקֵחַ מָעוֹת בְּיָדוֹ וְאָמַר אֵלּוּ לְשִׁקְלִי. אוֹ שֶׁהָיָה מְלַקֵּט מָעָה מָעָה אוֹ פְּרוּטָה פְּרוּטָה וּכְשֶׁהִתְחִיל לְלַקֵּט אָמַר הֲרֵינִי מְלַקֵּט מָעוֹת לְשִׁקְלִי. אֲפִלּוּ לִקֵּט מְלֹא כִּיס נוֹתֵן מֵהֶן חֲצִי שֶׁקֶל שֶׁהוּא חַיָּב בּוֹ וְהַשְּׁאָר חֻלִּין שֶׁמּוֹתַר הַשְּׁקָלִים חֻלִּין:
✨ קדושה וגבולותיה
מי שנותן חצי שקל מרכוש הקדש — יוצא ידי חובתו, אך כשאותו כסף מנוצל מתוך התרומה — הוא חייב מעילה. מי שהפריש שקל בטעות (חשב שהוא חייב אך לא היה חייב) — אינו מתקדש. הפריש שניים כשחייב רק אחד — אחד קדוש, השני אינו קדוש. מי שאוחז מטבעות בידו ואומר 'אלו לחצי שקלי': אם אחז יותר מחצי שקל — השאר חולין; אך אם אמר 'אני אוסף לשקלי' — הכול מתקדש ברגע שהשלים את הסכום הדרוש, וכל עודף הוא נדבה.
6/8

מטבעות שנמצאו בין קופות הבית

הלכות י״ד–ט״ו
הלכה י״ד׳
מָעוֹת שֶׁנִּמְצְאוּ בֵּין תֵּבָה שֶׁל שְׁקָלִים לְתֵבָה שֶׁל נְדָבָה. קָרוֹב לַשְּׁקָלִים יִפְּלוּ לִשְׁקָלִים. קָרוֹב לַנְּדָבָה יִפְּלוּ לִנְדָבָה. מֶחֱצָה עַל מֶחֱצָה יִפְּלוּ לִנְדָבָה מִפְּנֵי שֶׁהַנְּדָבָה כֻּלָּהּ עוֹלָה לַאִשִּׁים וְהַשְּׁקָלִים מִסְתַּפְּקִין מֵהֶן לְעוֹלוֹת וְלִדְבָרִים אֲחֵרִים:
הלכה ט״ו׳
וְכֵן כָּל הַמָּעוֹת הַנִּמְצָאִים בֵּין כָּל תֵּבָה וְתֵבָה יִפְּלוּ לַקָּרוֹב. נִמְצְאוּ מֶחֱצָה לְמֶחֱצָה. אִם בֵּין עֵצִים לַלְּבוֹנָה נִמְצְאוּ יִפְּלוּ לַלְּבוֹנָה. בֵּין קִנִּים לְגוֹזָלֵי עוֹלָה יִפְּלוּ לְגוֹזָלֵי עוֹלָה. זֶה הַכְּלָל הוֹלְכִין אַחַר הַקָּרוֹב בְּכֻלָּן. מֶחֱצָה לְמֶחֱצָה לְהַחֲמִיר. וְכָל הַמָּעוֹת הַנִּמְצָאוֹת בְּהַר הַבַּיִת חֻלִּין שֶׁאֵין הַגִּזְבָּר מוֹצִיא מָעוֹת מִתְּרוּמַת הַלִּשְׁכָּה עַד שֶׁהוּא מְחַלְּלָן עַל הַבְּהֵמוֹת שֶׁלּוֹקֵחַ לְקָרְבָּנוֹת:
🏛️ מטבעות שנמצאו
מטבעות שנמצאו בין קופת השקלים לקופת הנדבה — הולכים לפי קרבה: קרוב לשקלים — ילכו לשקלים; קרוב לנדבה — לנדבה; בדיוק באמצע — לנדבה. אותו כלל חל בין כל שתי קופות. נמצאו בדיוק באמצע בין עצים לקטורת — ילכו לקטורת; בין קינים לעולת העוף — לעולת העוף; בין עולות לשלמים — לעולות. העיקרון: בספק בין קל לחמור — פוסקים לחומרא.
7/8

🎓 עקרונות מרכזיים

🪙
הקלבון מוטל על מי שצריך החלפה
רק מי שחייב בשקל חייב בקלבון. מתנה פוטרת מקלבון. הקלבון מבטיח שמי שנהנה מההחלפה הוא שנושא בעלותה.
📬
האחריות נמשכת עד מסירה לגזבר
המצווה אינה נשלמת עם המסירה לשליח. כל עוד לא הגיע השקל לגזבר — הבעלים נשאר אחראי.
⚖️
הפרשה בטעות אינה מקדשת
קדושה תלויה בכוונה ובמציאות. מי שהפריש בחשבו שהוא חייב אך לא היה — הקדושה אינה חלה.
📐
קרבה מכריעה בספק חלוקת מטבעות
כשיש ספק לאיזו קופה שייכים המטבעות — הולכים אחר הקרבה. כשנמצאו בדיוק באמצע — נוטים לקטגוריה החמורה יותר.
