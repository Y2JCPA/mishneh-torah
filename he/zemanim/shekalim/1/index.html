חובת מחצית השקל

הלכות א׳–ב׳
הלכה א׳
מִצְוַת עֲשֵׂה מִן הַתּוֹרָה לִתֵּן כָּל אִישׁ מִיִּשְׂרָאֵל מַחֲצִית הַשֶּׁקֶל בְּכָל שָׁנָה וְשָׁנָה. אֲפִלּוּ עָנִי הַמִּתְפַּרְנֵס מִן הַצְּדָקָה חַיָּב. וְשׁוֹאֵל מֵאֲחֵרִים אוֹ מוֹכֵר כְּסוּת שֶׁעַל כְּתֵפוֹ וְנוֹתֵן מַחֲצִית הַשֶּׁקֶל כֶּסֶף שֶׁנֶּאֱמַר (שמות ל טו) "הֶעָשִׁיר לֹא יַרְבֶּה וְהַדַּל לֹא יַמְעִיט" וְגוֹ'. וְאֵינוֹ נוֹתְנוֹ בִּפְעָמִים רַבּוֹת הַיּוֹם מְעַט וּלְמָחָר מְעַט אֶלָּא נוֹתְנוֹ כֻּלּוֹ כְּאַחַת בְּפַעַם אַחַת:
הלכה ב׳
מִנְיַן כֶּסֶף הָאָמוּר בַּתּוֹרָה בְּאוֹנֵס וּבִמְפַתֶּה וּבְמוֹצִיא שֵׁם רַע וְהוֹרֵג עֶבֶד הוּא כֶּסֶף הוּא שֶׁקֶל הַנֶּאֱמָר בְּכָל מָקוֹם בַּתּוֹרָה. וּמִשְׁקָלוֹ שְׁלֹשׁ מֵאוֹת וְעֶשְׂרִים שְׂעוֹרָה. וּכְבָר הוֹסִיפוּ חֲכָמִים עָלָיו וְעָשׂוּ מִשְׁקָלוֹ כְּמִשְׁקַל הַמַּטְבֵּעַ הַנִּקְרָא סֶלַע בִּזְמַן בַּיִת שֵׁנִי. וְכַמָּה הוּא מִשְׁקַל הַסֶּלַע שְׁלֹשׁ מֵאוֹת וְאַרְבַּע וּשְׁמוֹנִים שְׂעוֹרָה בֵּינוֹנִית:
💰 מהות המצווה
מצוות עשה מן התורה לכל יהודי לתת מחצית השקל בכל שנה — אפילו עני המתפרנס מן הצדקה חייב, עד כדי שילווה מאחרים או ימכור את בגדו. נותנים בבת אחת, לא בתשלומים. שיעור השקל: 320 שעורות מן התורה; חכמים הגדילו ל-384 שעורות (שווה לסלע של בית שני).

מערכת המטבעות

הלכות ג׳–ד׳
הלכה ג׳
הַסֶּלַע אַרְבָּעָה דִּינָרִין וְהַדִּינָר שֵׁשׁ מָעִין. וּמָעָה הִיא הַנִּקְרֵאת בִּימֵי משֶׁה רַבֵּנוּ גֵּרָה. וּמָעָה הִיא שְׁנֵי פּוּנְדְיוֹנִין וּפוּנְדְיוֹן שְׁנֵי אִיסָרִין. וּפְרוּטָה אַחַת מִשְּׁמוֹנָה בְּאִיסָר. נִמְצָא מִשְׁקַל הַמָּעָה וְהִיא הַגֵּרָה שֵׁשׁ עֶשְׂרֵה שְׂעוֹרוֹת. וּמִשְׁקַל הָאִיסָר אַרְבַּע שְׂעוֹרוֹת. וּמִשְׁקַל הַפְּרוּטָה חֲצִי שְׂעוֹרָה:
הלכה ד׳
וְעוֹד מַטְבֵּעַ אַחֵר הָיָה שָׁם שֶׁהָיָה מִשְׁקָלוֹ שְׁתֵּי סְלָעִים וְהוּא הָיָה נִקְרָא דַּרְכּוֹן. וְאֵלּוּ הַמַּטְבְּעוֹת כֻּלָּן שֶׁאָמַרְנוּ וּבֵאַרְנוּ מִשְׁקַל כָּל אֶחָד מֵהֶן הֵן שֶׁמְּשַׁעֲרִין בָּהֶן בְּכָל מָקוֹם. וּכְבָר בֵּאַרְנוּם כְּדֵי שֶׁלֹּא אֶהְיֶה צָרִיךְ לְפָרֵשׁ מִשְׁקָלָם בְּכָל מָקוֹם:
🪙 שיעורי המטבע
סלע = 4 דינרים; דינר = 6 מעות (= גרה בלשון התורה); מעה = 2 פונדיון = 4 איסר; פרוטה = 1/8 איסר. משקל מעה = 16 שעורות; איסר = 4 שעורות; פרוטה = 1/2 שעורה. דרקון = 2 סלעים — מטבע כפול המשמש כמידה כללית.

עקרון המינימום והתאמה למטבע הנהוג

הלכות ה׳–ו׳
הלכה ה׳
מַחֲצִית הַשֶּׁקֶל זוֹ מִצְוָתָהּ שֶׁיִּתֵּן מַחֲצִית מַטְבֵּעַ שֶׁל אוֹתוֹ הַזְּמַן אֲפִלּוּ הָיָה אוֹתוֹ מַטְבֵּעַ גָּדוֹל מִשֶּׁקֶל הַקֹּדֶשׁ. וּלְעוֹלָם אֵינוֹ שׁוֹקֵל פָּחוֹת מֵחֲצִי הַשֶּׁקֶל שֶׁהָיָה בִּימֵי משֶׁה רַבֵּנוּ שֶׁהוּא מִשְׁקָלוֹ מֵאָה וְשִׁשִּׁים שְׂעוֹרָה:
הלכה ו׳
בִּזְמַן שֶׁהָיָה מַטְבֵּעַ שֶׁל אוֹתוֹ זְמַן דַּרְכּוֹנוֹת הָיָה כָּל אֶחָד וְאֶחָד נוֹתֵן בְּמַחֲצִית הַשֶּׁקֶל שֶׁלּוֹ סֶלַע. וּבִזְמַן שֶׁהָיָה הַמַּטְבֵּעַ סְלָעִים הָיָה נוֹתֵן כָּל אֶחָד בְּמַחֲצִית הַשֶּׁקֶל שֶׁלּוֹ חֲצִי סֶלַע שֶׁהוּא שְׁנֵי דִּינָרִין. וּבִזְמַן שֶׁהָיָה הַמַּטְבֵּעַ חֲצִי סֶלַע הָיָה כָּל אֶחָד וְאֶחָד נוֹתֵן בְּמַחֲצִית הַשֶּׁקֶל אוֹתוֹ חֲצִי הַסֶּלַע. וּמֵעוֹלָם לֹא שָׁקְלוּ יִשְׂרָאֵל בְּמַחֲצִית הַשֶּׁקֶל פָּחוֹת מֵחֲצִי שֶׁקֶל שֶׁל תּוֹרָה:
⚖️ מינימום ומקסימום
נותנים מחצית המטבע הנהוג, אפילו אם גדול מהשקל המקורי. אך לעולם לא פחות ממחצית השקל של התורה (160 שעורות). לכן: כשהדרקון היה נהוג — נתנו סלע שלם; כשסלעים היו נהוגים — חצי סלע; כשחצי סלעים היו קיימים — אותו חצי סלע.

מי חייב ומי פטור

הלכות ז׳–ח׳
הלכה ז׳
הַכּל חַיָּבִין לִתֵּן מַחֲצִית הַשֶּׁקֶל כֹּהֲנִים לְוִיִּים וְיִשְׂרְאֵלִים וְגֵרִים וַעֲבָדִים מְשֻׁחְרָרִים. אֲבָל לֹא נָשִׁים וְלֹא עֲבָדִים וְלֹא קְטַנִּים. וְאִם נָתְנוּ מְקַבְּלִין מֵהֶם. אֲבָל הַכּוּתִים שֶׁנָּתְנוּ מַחֲצִית הַשֶּׁקֶל אֵין מְקַבְּלִין מֵהֶם. קָטָן שֶׁהִתְחִיל אָבִיו לִתֵּן עָלָיו מַחֲצִית הַשֶּׁקֶל שׁוּב אֵינוֹ פּוֹסֵק אֶלָּא נוֹתֵן עָלָיו בְּכָל שָׁנָה וְשָׁנָה עַד שֶׁיַּגְדִּיל וְיִתֵּן עַל עַצְמוֹ:
הלכה ח׳
הַשְּׁקָלִים אֵינָן נוֹהֲגִין אֶלָּא בִּפְנֵי הַבַּיִת. וּבִזְמַן שֶׁבֵּית הַמִּקְדָּשׁ קַיָּם נוֹתְנִין אֶת הַשְּׁקָלִים בֵּין בְּאֶרֶץ יִשְׂרָאֵל בֵּין בְּחוּצָה לָאָרֶץ. וּבִזְמַן שֶׁהוּא חָרֵב אֲפִלּוּ בְּאֶרֶץ יִשְׂרָאֵל אֵין נוֹהֲגִין:
👥 חייבים ופטורים
חייבים: כוהנים, לויים, ישראלים, גרים ועבדים משוחררים. פטורים: נשים, עבדים, קטנים — אך אם נתנו, מקבלים מהם. מכותים — לא מקבלים לעולם. אב שהתחיל לתת עבור בנו אינו רשאי לפסוק עד שיגדל. המצווה חלה רק בזמן שבית המקדש קיים — בארץ ישראל ובחוץ לארץ.
7/9

לוח הגבייה ואכיפתה

הלכות ט׳–י׳
הלכה ט׳
בְּאֶחָד בַּאֲדָר מַשְׁמִיעִין עַל הַשְּׁקָלִים כְּדֵי שֶׁיָּכִין כָּל אֶחָד וְאֶחָד מַחֲצִית הַשֶּׁקֶל שֶׁלּוֹ. וְיִהְיֶה עָתִיד לִתֵּן. בַּחֲמִשָּׁה עָשָׂר בּוֹ יָשְׁבוּ הַשֻּׁלְחָנִים בְּכָל מְדִינָה וּמְדִינָה וְתוֹבְעִין בְּנַחַת כָּל מִי שֶׁיִּתֵּן לָהֶם יְקַבְּלוּ מִמֶּנּוּ וּמִי שֶׁלֹּא נָתַן אֵין כּוֹפִין אוֹתוֹ לִתֵּן. בַּחֲמִשָּׁה וְעֶשְׂרִים בּוֹ יָשְׁבוּ בַּמִּקְדָּשׁ לִגְבּוֹת. וּמִכָּאן וְאֵילָךְ כּוֹפִין אֶת מִי שֶׁלֹּא נָתַן עַד שֶׁיִּתֵּן. וְכָל מִי שֶׁלֹּא יִתֵּן מְמַשְׁכְּנִין אוֹתוֹ וְלוֹקְחִין עֲבוֹטוֹ בַּעַל כָּרְחוֹ וַאֲפִלּוּ כְּסוּתוֹ:
הלכה י׳
כָּל מִי שֶׁאֵינוֹ חַיָּב בִּשְׁקָלִים אַף עַל פִּי שֶׁדַּרְכּוֹ לִתֵּן אוֹ הוּא עָתִיד לִתֵּן אֵין מְמַשְׁכְּנִין אוֹתוֹ. וְאֵין מְמַשְׁכְּנִין אֶת הַכֹּהֲנִים לְעוֹלָם מִפְּנֵי דַּרְכֵי שָׁלוֹם אֶלָּא כְּשֶׁיִּתְּנוּ מְקַבְּלִין מֵהֶן וְתוֹבְעִין אוֹתָן עַד שֶׁיִּתְּנוּ:
📅 ציר הזמן של הגבייה
א׳ אדר: הכרזה. ט״ו אדר: שולחנות מוקמים בכל עיר — הגבאים שואלים בעדינות, ללא כפייה. כ״ה אדר: שולחנות במקדש — מתחילים לכפות, ממשכנים אפילו את בגדו. פטורים (נשים, קטנים) — אין ממשכנים. כוהנים אינם נמשכנים מפני שלום, אך שואלים אותם עד שייתנו.

שוויון לפני המצווה
עשיר ועני נותנים בדיוק אותו הסכום — מחצית השקל. שוויון זה מסמל שכל נשמות ישראל שוות לפני האל.
⏳
המצווה תלויה בבית המקדש
חובת השקל חלה רק בזמן שבית המקדש קיים. כיום הגבייה מופסקת; המנהג לתת לפני פורים הוא זכר בלבד.
📐
דיוק במידות
הרמב"ם מפרט כל שיעור מטבע כדי לקבוע פעם אחת ולתמיד את הבסיס לכל דיני ממונות בתורה.
🤝
אכיפה מדורגת
התורה דרשה גבייה אך קבעה שלבים: תחילה הכרזה, אחר כך שכנוע עדין, ולבסוף כפייה ממשית — מידתיות המשקפת כבוד האדם לצד חובת המצווה.
