הוצאות מתרומת הלשכה

הלכות א׳–ב׳
הלכה א׳
תְּרוּמַת הַלִּשְׁכָּה מַה יֵּעָשֶׂה בָּהּ. לוֹקְחִין מִמֶּנָּה תְּמִידִין שֶׁל כָּל יוֹם וְהַמּוּסָפִין וְכָל קָרְבְּנוֹת הַצִּבּוּר וְנִסְכֵּיהֶם וְהַמֶּלַח שֶׁמּוֹלְחִין בּוֹ כָּל הַקָּרְבָּנוֹת. וְכֵן הָעֵצִים אִם לֹא הֵבִיאוּ עֵצִים וְלֹא נִמְצְאוּ אֶלָּא בְּדָמִים. וְהַקְּטֹרֶת וּשְׂכַר עֲשִׂיָּתָהּ. וְלֶחֶם הַפָּנִים וּשְׂכַר עוֹשֵׂי לֶחֶם הַפָּנִים. וְהָעֹמֶר וּשְׁתֵּי הַלֶּחֶם וּפָרָה אֲדֻמָּה וְשָׂעִיר הַמִּשְׁתַּלֵּחַ וְלָשׁוֹן שֶׁל זְהוֹרִית שֶׁקּוֹשְׁרִין בֵּין קַרְנָיו. כָּל אֵלּוּ בָּאִין מִתְּרוּמַת הַלִּשְׁכָּה:
הלכה ב׳
אֲבָל פַּר הֶעְלֵם דָּבָר שֶׁל צִבּוּר וּשְׂעִירֵי עֲבוֹדַת כּוֹכָבִים בַּתְּחִלָּה גּוֹבִין לָהֶן וְאֵינָן בָּאִין מִתְּרוּמַת הַלִּשְׁכָּה. פָּרוֹכוֹת שֶׁל הֵיכָל תַּחַת בִּנְיָן עֲשׂוּיוֹת וְאֵינָן בָּאִין מִתְּרוּמַת הַלִּשְׁכָּה אֶלָּא מִקָּדְשֵׁי בֶּדֶק הַבַּיִת. אֲבָל פָּרוֹכוֹת שֶׁל שְׁעָרִים בָּאִין מִתְּרוּמַת הַלִּשְׁכָּה. הַמְּנוֹרָה וּכְלֵי שָׁרֵת מִצְוָתָן שֶׁיָּבוֹאוּ מִמּוֹתַר הַנְּסָכִים. וּבְהִלְכוֹת כְּלֵי הַמִּקְדָּשׁ וְהָעוֹבְדִים בּוֹ יִתְבָּאֵר מַה הוּא מוֹתַר הַנְּסָכִים. וְאִם לֹא הָיָה לָהֶן מוֹתַר נְסָכִים יָבִיאוּ מִתְּרוּמַת הַלִּשְׁכָּה. בִּגְדֵי כְּהֻנָּה בֵּין בִּגְדֵי כֹּהֵן גָּדוֹל בֵּין שְׁאָר בִּגְדֵי הַכֹּהֲנִים שֶׁעוֹבְדִין בָּהֶן בַּמִּקְדָּשׁ הַכּל מִתְּרוּמַת הַלִּשְׁכָּה:
🏛️ מה ממומן מהתרומה
תרומת הלשכה מממנת את כל קרבנות הציבור של תמיד ומוסף, נסכיהם, מלח, עצים, קטורת, לחם הפנים, עומר, שתי הלחם, פרה אדומה ושעיר המשתלח. לעומת זאת, פר העלם דבר של ציבור ושעירי עבודה זרה אינם באים מהתרומה אלא מגביה מיוחדת. פרוכות ההיכל באות מכספי בדק הבית (כיוון שהן מבניות), ואילו פרוכות השערים באות מהתרומה. המנורה וכלי השרת באים מעודפי הנסכים, ובגדי כהונה — מהתרומה.

מקרים מיוחדים ושכר הממונים

הלכות ג׳–ו׳
הלכה ג׳
כָּל הַבְּהֵמוֹת הַנִּמְצָאוֹת בִּירוּשָׁלַיִם אוֹ בְּחוּצָה לָהּ בְּקָרוֹב מִמֶּנָּה בָּאוֹת עוֹלוֹת כְּמוֹ שֶׁיִּתְבָּאֵר בִּפְסוּלֵי הַמֻּקְדָּשִׁין. נִסְכֵּיהֶן בָּאִין מִתְּרוּמַת הַלִּשְׁכָּה. וְכֵן עוֹבֵד כּוֹכָבִים שֶׁשָּׁלַח עוֹלָתוֹ מִמְּדִינָה אַחֶרֶת וְלֹא שָׁלַח עִמָּהּ דְּמֵי נְסָכִים יָבִיאוּ נִסְכֵּיהֶם מִתְּרוּמַת הַלִּשְׁכָּה:
הלכה ד׳
גֵּר שֶׁמֵּת וְהִנִּיחַ זְבָחִים אִם יֵשׁ לוֹ נְסָכִים קְרֵבִין מִשֶּׁלּוֹ וְאִם לָאו בָּאִין מִתְּרוּמַת הַלִּשְׁכָּה. כֹּהֵן גָּדוֹל שֶׁמֵּת וְלֹא מִנּוּ אַחֵר תַּחְתָּיו מַקְרִיבִין אֶת הַחֲבִיתִין מִתְּרוּמַת הַלִּשְׁכָּה. מְבַקְּרֵי מוּמִים שֶׁבִּירוּשָׁלַיִם וְתַלְמִידֵי חֲכָמִים הַמְלַמְּדִים הִלְכוֹת שְׁחִיטָה לַכֹּהֲנִים וְהִלְכוֹת קְמִיצָה וְנָשִׁים הַמְגַדְּלוֹת בְּנֵיהֶן לְפָרָה אֲדֻמָּה כֻּלָּן נוֹטְלִין שְׂכָרָן מִתְּרוּמַת הַלִּשְׁכָּה. וְכַמָּה הוּא שְׂכָרָן. כְּמוֹ שֶׁיִּפְסְקוּ לָהֶן בֵּית דִּין:
הלכה ה׳
בִּשְׁנַת הַשְּׁמִטָּה שֶׁהִיא הֶפְקֵר שׂוֹכְרִין בֵּית דִּין שׁוֹמְרִין שֶׁיִּשְׁמְרוּ מִקְצָת סְפִיחִים שֶׁצָּמְחוּ כְּדֵי שֶׁיָּבִיאוּ מֵהֶן הָעֹמֶר וּשְׁתֵּי הַלֶּחֶם שֶׁאֵין בָּאִים אֶלָּא מִן הֶחָדָשׁ. וְאֵלּוּ הַשּׁוֹמְרִין נוֹטְלִין שְׂכָרָן מִתְּרוּמַת הַלִּשְׁכָּה:
הלכה ו׳
מִי שֶׁהִתְנַדֵּב לִשְׁמֹר בְּחִנָּם אֵין שׁוֹמְעִין לוֹ מִשּׁוּם בַּעֲלֵי זְרוֹעַ שֶׁמָּא יָבוֹאוּ וְיִטְּלוּם מֵהֶן. לְפִיכָךְ תִּקְּנוּ לָהֶם חֲכָמִים שֶׁיִּטְּלוּ שָׂכָר מִן הַלִּשְׁכָּה כְּדֵי שֶׁיִּפְרְשׁוּ הַכּל מֵאוֹתוֹ מָקוֹם שֶׁאֵלּוּ שׁוֹמְרִים שָׁם:
👥 שכר ומקרי קצה
בהמות שנמצאו בירושלים מוקרבות כעולות עם נסכים מהתרומה; כך גם עולת גוי שלא שלח כספי נסכים. גר שמת והניח קרבנות ללא נסכים — הנסכים באים מהתרומה. אם מת הכהן הגדול ולא מינו תחתיו — מנחת החביתין היומית באה מהתרומה. בודקי מומים, חכמים המלמדים שחיטה וקמיצה, נשים המגדלות פרות לפרה אדומה — כולם מקבלים שכרם מהתרומה כפי שקבע בית הדין. בשנת שמיטה שוכר בית הדין שומרים לשמור על הספיחים לצורך העומר ושתי הלחם. מתנדב השומר חינם — מסורב — שמא יבואו בעלי כוח ויתחזקו בו; שמירה בשכר מרתיעה את כולם.

עודפי הלשכה וקרבנות המזבח

הלכות ז׳–ט׳
הלכה ז׳
מַגִּיהֵי סְפָרִים שֶׁבִּירוּשָׁלַיִם וְדַיָּנִין שֶׁדָּנִין אֶת הַגַּזְלָנִין בִּירוּשָׁלַיִם נוֹטְלִין שְׂכָרָן מִתְּרוּמַת הַלִּשְׁכָּה. וְכַמָּה הָיוּ נוֹטְלִים תִּשְׁעִים מָנֶה בְּכָל שָׁנָה. וְאִם לֹא הִסְפִּיקוּ לָהֶן אַף עַל פִּי שֶׁלֹּא רָצוּ מוֹסִיפִין לָהֶן כְּדֵי צָרְכָּן הֵם וּנְשֵׁיהֶם וּבְנֵיהֶם וּבְנֵי בֵּיתָן:
הלכה ח׳
כֶּבֶשׁ הָיוּ בּוֹנִין מֵהַר הַבַּיִת לְהַר הַמִּשְׁחָה שֶׁעָלָיו מוֹצִיאִין פָּרָה אֲדֻמָּה. וְכֶבֶשׁ שֶׁמּוֹצִיאִין עָלָיו שָׂעִיר הַמִּשְׁתַּלֵּחַ. וּשְׁנֵיהֶם נַעֲשִׂין מִשְּׁיָרֵי הַלִּשְׁכָּה. וְכֵן מִזְבַּח הָעוֹלָה וְהַהֵיכָל וְהָעֲזָרוֹת נַעֲשִׂין מִשְּׁיָרֵי הַלִּשְׁכָּה. אַמַּת הַמַּיִם שֶׁבִּירוּשָׁלַיִם וְחוֹמַת יְרוּשָׁלַיִם וְכָל מִגְדְּלוֹתֶיהָ וְכָל צָרְכֵי הָעִיר בָּאִין מִשְּׁיָרֵי הַלִּשְׁכָּה. וְעוֹבֵד כּוֹכָבִים שֶׁהִתְנַדֵּב מָעוֹת לַדְּבָרִים הָאֵלּוּ אוֹ לַעֲשׂוֹת עִמָּהֶם בְּחִנָּם אֵין מְקַבְּלִין מִמֶּנּוּ וַאֲפִלּוּ גֵּר תּוֹשָׁב. שֶׁנֶּאֱמַר (עזרא ד ג) "לֹא לָכֶם וָלָנוּ לִבְנוֹת" וְגוֹ' וְנֶאֱמַר (נחמיה ב כ) "וְלָכֶם אֵין חֵלֶק" וְגוֹ':
הלכה ט׳
מוֹתַר תְּרוּמַת הַלִּשְׁכָּה וּשְׁיָרֵי הַלִּשְׁכָּה לוֹקְחִין בּוֹ זְכָרִים וִיקָרְבוּ כֻּלָּן עוֹלוֹת שֶׁתְּנַאי בֵּית דִּין הוּא עַל כָּל הַמּוֹתָרוֹת שֶׁיִּקְרְבוּ עוֹלַת בְּהֵמָה. אֲבָל לֹא עוֹלַת הָעוֹף שֶׁאֵין בְּקָרְבְּנוֹת הַצִּבּוּר עוֹף. וְאֵלּוּ הָעוֹלוֹת הַבָּאִין מִמּוֹתַר הַשְּׁקָלִים הֵם הַנִּקְרָאִים קַיִץ לַמִּזְבֵּחַ:
🏗️ עודפים ו'קיץ המזבח'
מגיהי ספרי התורה ודיינים הדנים בגזלות בירושלים מקבלים תשעים מנה לשנה מהתרומה; ואם אין בכך די, מוסיפים להם לצורכיהם, נשיהם וילדיהם. עודפי הלשכה (מה שנשאר לאחר שלוש ההרמות) מממנים: כבש פרה אדומה ושעיר המשתלח, מזבח העולה, ההיכל, העזרות, האמה, חומות ירושלים ומגדליה, וכל צורכי העיר. תרומות גויים לצורכים אלו אינן מתקבלות — ואפילו מגר תושב — שנאמר 'לא לכם ולנו לבנות.' עודפי התרומה וכספי הלשכה משמשים לקניית בהמות זכרים לעולות הנקראות 'קיץ המזבח'; עופות אינם נקנים לעולם, שכן קרבנות ציבור אינם כוללים עופות.

חסרונות ומעבר ניסן

הלכות י׳–י״ב
הלכה י׳
שְׁקָלִים שֶׁלֹּא הִסְפִּיקוּ לָהֶן לְכָל קָרְבְּנוֹת הַצִּבּוּר מוֹצִיאִין אֶת הָרָאוּי לָהֶם מִקָּדְשֵׁי בֶּדֶק הַבַּיִת. אֲבָל אֵין בֶּדֶק הַבַּיִת מוֹצִיא אֶת הָרָאוּי לוֹ מִקָּדְשֵׁי הַמִּזְבֵּחַ:
הלכה י״א׳
מִשֶּׁיַגִּיעַ רֹאשׁ חֹדֶשׁ נִיסָן אֵין מַקְרִיבִין קָרְבְּנוֹת צִבּוּר אֶלָּא מִתְּרוּמָה חֲדָשָׁה. וְאִם לֹא בָּאָה הַחֲדָשָׁה לוֹקְחִין מִן הַיְשָׁנָה. לְפִיכָךְ אִם הִגִּיעַ רֹאשׁ חֹדֶשׁ נִיסָן וְיֵשׁ עִמָּהֶן בְּהֵמוֹת לִתְמִידִים מִתְּרוּמָה יְשָׁנָה פּוֹדִין אוֹתָן וְיוֹצְאִין לְחֻלִּין אַף עַל פִּי שֶׁהֵן תְּמִימִין וְיִפְּלוּ דְּמֵיהֶן לִתְרוּמָה יְשָׁנָה שֶׁמְּקִיצִין בָּהּ אֶת הַמִּזְבֵּחַ. שֶׁתְּנַאי בֵּית דִּין הוּא עַל כָּל הַבְּהֵמוֹת שֶׁלּוֹקְחִין לִתְמִידִין שֶׁאִם לֹא יִהְיוּ צְרִיכִין לָהֶן יֵצְאוּ לְחֻלִּין:
הלכה י״ב׳

🔄 המעבר בראש חודש ניסן
אם השקלים לא הספיקו לקרבנות — לוקחים מכספי בדק הבית; אך בדק הבית לעולם אינו לוקח מכספי המזבח. מראש חודש ניסן, הקרבנות באים רק מהתרומה החדשה; אם טרם הגיעה, מותר להשתמש בישנה. בהמות תמיד שנותרו מהתרומה הישנה נפדות אף שאינן בעלות מום, ותמורתן הולכת לתרומה הישנה לצורך 'קיץ המזבח' — שכן תנאי בית הדין קובע שבהמות שאינן נצרכות חוזרות למעמד חולין. עודפי קטורת: קדושתה מועברת על שכר האומנים ← הכסף הולך לקיץ המזבח ← האומנים לוקחים את הקטורת כשכרם ← בית הדין קונה אותה בחזרה מהם בכספי התרומה החדשה. אם אין תרומה חדשה — הקטורת הישנה נשרפת.
