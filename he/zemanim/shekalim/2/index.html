קופות גביית השקלים בערים ובמקדש

הלכות א׳–ג׳
הלכה א׳
כֵּיצַד כּוֹנְסִין הַשֻּׁלְחָנִים אֶת הַשְּׁקָלִים בְּכָל מְדִינָה וּמְדִינָה. מַנִּיחִין לִפְנֵיהֶם שְׁתֵּי תֵּבוֹת שׁוּלֵי הַתֵּבָה רְחָבִין מִלְּמַטָּה וּפִיהָ צַר מִלְּמַעְלָה כְּמִין שׁוֹפָר כְּדֵי שֶׁיַּשְׁלִיכוּ לְתוֹכָן וְלֹא יִהְיֶה אֶפְשָׁר לִקַּח מֵהֶן בְּנַחַת. וְלָמָּה עוֹשִׂין שְׁתֵּי תֵּבוֹת. אַחַת שֶׁמַּשְׁלִיכִין בָּהּ שְׁקָלִים שֶׁל שָׁנָה זוֹ. וְהַשְּׁנִיָּה מַנִּיחִין בָּהּ שִׁקְלֵי שָׁנָה שֶׁעָבְרָה. שֶׁגּוֹבִין מִמִּי שֶׁלֹּא שָׁקַל בְּשָׁנָה שֶׁעָבְרָה:
הלכה ב׳
וּבַמִּקְדָּשׁ הָיָה לִפְנֵיהֶם תָּמִיד שְׁלֹשׁ עֶשְׂרֵה תֵּבוֹת. כָּל תֵּבָה כְּמִין שׁוֹפָר. רִאשׁוֹנָה לְשִׁקְלֵי שָׁנָה זוֹ. שְׁנִיָּה לְשִׁקְלֵי שָׁנָה שֶׁעָבְרָה. שְׁלִישִׁית לְכָל מִי שֶׁיֵּשׁ עָלָיו קָרְבַּן שְׁתֵּי תּוֹרִים אוֹ שְׁנֵי בְּנֵי יוֹנָה אֶחָד עוֹלָה וְאֶחָד חַטָּאת מַשְׁלִיךְ דְּמֵיהֶן לְתֵבָה זוֹ. רְבִיעִית לְכָל מִי שֶׁיֵּשׁ עָלָיו עוֹלַת הָעוֹף בִּלְבַד מַשְׁלִיךְ דָּמֶיהָ לְתֵבָה זוֹ. חֲמִישִׁית לְמִי שֶׁהִתְנַדֵּב מָעוֹת לִקְנוֹת עֵצִים לַמַּעֲרָכָה בָּהֶן. שִׁשִּׁית לְמִי שֶׁהִתְנַדֵּב מָעוֹת לִלְבוֹנָה. שְׁבִיעִית לְמִי שֶׁהִתְנַדֵּב זָהָב לַכַּפֹּרֶת. שְׁמִינִית לְמוֹתַר חַטָּאת כְּגוֹן שֶׁהִפְרִישׁ מָעוֹת לְחַטָּאתוֹ וְלָקַח חַטָּאת וְהוֹתִיר מִן הַמָּעוֹת יַשְׁלִיךְ הַשְּׁאָר לְתוֹכָהּ. תְּשִׁיעִית לְמוֹתַר אָשָׁם. עֲשִׂירִית לְמוֹתַר קִנֵּי זָבִים וְזָבוֹת וְיוֹלְדוֹת. אַחַת עֶשְׂרֵה לְמוֹתַר קָרְבְּנוֹת נָזִיר. שְׁתֵּים עֶשְׂרֵה לְמוֹתַר אֲשַׁם מְצֹרָע. שְׁלֹשׁ עֶשְׂרֵה לְמִי שֶׁהִתְנַדֵּב מָעוֹת לְעוֹלַת בְּהֵמָה:
הלכה ג׳
וְשֵׁם כָּל דָּבָר שֶׁיִּהְיוּ מָעוֹתָיו בְּתוֹךְ הַתֵּבָה כָּתוּב עַל הַתֵּבָה מִבַּחוּץ. וּתְנַאי בֵּית דִּין הוּא עַל כָּל הַמּוֹתָרוֹת שֶׁיִּקָּרְבוּ עוֹלַת בְּהֵמָה. נִמְצֵאתָ לָמֵד שֶׁכָּל הַמָּעוֹת הַנִּמְצָאוֹת בְּשֵׁשׁ הַתֵּבוֹת הָאַחֲרוֹנוֹת לְעוֹלַת בְּהֵמָה וְעוֹרוֹתֶיהָ לַכֹּהֲנִים כִּשְׁאָר כָּל הָעוֹרוֹת. וְכָל הַמָּעוֹת שֶׁבְּתֵבָה שְׁלִישִׁית לִקַּח בָּהֶן עוֹפוֹת חֶצְיָן עוֹלוֹת וְחֶצְיָן חַטָּאוֹת. וְשֶׁבָּרְבִיעִית כֻּלָּן עוֹלַת הָעוֹף:
🏛️ מבנה האוצר
בכל עיר הוניחו שתי קופות בצורת שופר — אחת לשנה הנוכחית ואחת לשנה שעברה. במקדש היו שלוש עשרה קופות: שתיים לשקלים, ואחת עשרה לנדבות שונות — עולות, חטאות, מותר אשם ונזיר, נדבת עצים ולבונה, וזהב לכפורת. תנאי בית דין קבע שכל מותר כספי הקרבנות ישמש לעולות ציבוריות, וגינותיהן ילכו לכוהנים.

הובלת השקלים ואחסונם בלשכה

הלכה ד׳
הלכה ד׳
בְּכָל מְדִינָה וּמְדִינָה כְּשֶׁגּוֹבִין הַשְּׁקָלִים מְשַׁלְּחִין אוֹתָן בְּיַד שְׁלוּחִים לַמִּקְדָּשׁ. וְיֵשׁ לָהֶם לְצָרֵף אוֹתָן בְּדִינְרֵי זָהָב מִפְּנֵי מַשּׂאוֹי הַדֶּרֶךְ. וְהַכּל מִתְקַבְּצִין לַמִּקְדָּשׁ וּמַנִּיחִין אוֹתָן בְּלִשְׁכָּה אַחַת מִן הַלְּשָׁכוֹת בַּמִּקְדָּשׁ וְסוֹגְרִין כָּל דַּלְתוֹתֶיהָ בְּמַפְתְּחוֹת וְחוֹתְמִין עָלֶיהָ חוֹתָמוֹת וּמְמַלְּאִין מִכָּל הַשְּׁקָלִים שֶׁיִּתְקַבְּצוּ שָׁם שָׁלֹשׁ קֻפּוֹת גְּדוֹלוֹת שִׁעוּר כָּל קֻפָּה כְּדֵי שֶׁתָּכִיל תֵּשַׁע סְאִין וְהַשְּׁאָר מַנִּיחִין אוֹתוֹ בַּלִּשְׁכָּה. וְזֶה שֶׁבְּתוֹךְ הַקֻּפּוֹת הוּא הַנִּקְרָא תְּרוּמַת הַלִּשְׁכָּה וְזֶה שֶׁיִּשָּׁאֵר שָׁם יֶתֶר עַל מַה שֶּׁיֵּשׁ בַּקֻּפּוֹת הוּא הַנִּקְרָא שְׁיָרֵי הַלִּשְׁכָּה:
🚚 מסע הכסף
שקלים שנגבו בכל מחוז נשלחו למקדש באמצעות שלוחים ממונים, וניתן היה להמירם בדינרי זהב להקלת הדרך. הכסף נאסף בלשכה תחת מנעול וחותם. הוא מולא לתוך שלושה כלים גדולים של תשעה סאים כל אחד — זוהי תרומת הלשכה. הנותר נקרא מותר הלשכה.

שלוש עונות התרומה וסדרן

הלכות ה׳–ז׳
הלכה ה׳
בִּשְׁלֹשָׁה פְּרָקִים בַּשָּׁנָה תּוֹרְמִין אֶת הַלִּשְׁכָּה. בְּרֹאשׁ חֹדֶשׁ נִיסָן. וּבְרֹאשׁ חֹדֶשׁ תִּשְׁרֵי קֹדֶם יוֹם טוֹב אוֹ אַחֲרָיו. וְקֹדֶם עֲצֶרֶת בַּחֲמִשָּׁה עָשָׂר יוֹם. וְכֵיצַד תּוֹרְמִין אוֹתָהּ. אֶחָד נִכְנַס לִפְנִים מִן הַלִּשְׁכָּה וְהַשּׁוֹמְרִין עוֹמְדִים בַּחוּץ וְהוּא אוֹמֵר לָהֶם אֶתְרֹם וְהֵן אוֹמְרִים לוֹ תְּרֹם תְּרֹם תְּרֹם שְׁלֹשָׁה פְּעָמִים. וְאַחַר כָּךְ מְמַלֵּא שָׁלֹשׁ קֻפּוֹת קְטַנּוֹת כָּל קֻפָּה מֵהֶן מְכִילָה שָׁלֹשׁ סְאִין מֵאוֹתָן שָׁלֹשׁ קֻפּוֹת הַגְּדוֹלוֹת וּמוֹצִיאָן לַחוּץ כְּדֵי לְהִסְתַּפֵּק מֵהֶן עַד שֶׁיִּכְלוּ. וְחוֹזֵר וּמְמַלֵּא אוֹתָן הַשָּׁלֹשׁ קֻפּוֹת הַקְּטַנּוֹת מִן שָׁלֹשׁ גְּדוֹלוֹת פַּעַם שְׁנִיָּה קֹדֶם עֲצֶרֶת וּמִסְתַּפְּקִין מֵהֶן עַד שֶׁיִּכְלוּ:
הלכה ו׳
וְחוֹזֵר וּמְמַלֵּא אוֹתָן פַּעַם שְׁלִישִׁית מִן הַשָּׁלֹשׁ קֻפּוֹת הַגְּדוֹלוֹת בְּתִשְׁרֵי וּמִסְתַּפְּקִין מֵהֶן עַד שֶׁיִּכְלוּ עַד רֹאשׁ חֹדֶשׁ נִיסָן. וּבְרֹאשׁ חֹדֶשׁ נִיסָן תּוֹרְמִין מִתְּרוּמָה חֲדָשָׁה. לֹא הִסְפִּיקוּ לָהֶן הַשְּׁקָלִים שֶׁבְּשָׁלֹשׁ קֻפּוֹת הַגְּדוֹלוֹת וְכָלוּ עַד שֶׁלֹּא הִגִּיעַ נִיסָן חוֹזְרִין וְתוֹרְמִין מִשְּׁיָרֵי הַלִּשְׁכָּה:
הלכה ז׳
שָׁלֹשׁ קֻפּוֹת הַקְּטַנּוֹת שֶׁהוּא תּוֹרֵם בָּהֶם וּמוֹצִיאָן לַחוּץ כָּתוּב עֲלֵיהֶן אָלֶ"ף בֵּי"ת גִּימֶ"ל כְּדֵי שֶׁיִּסְתַּפְּקוּ מִן הָרִאשׁוֹנָה עַד שֶׁתִּכְלֶה. וְאַחַר כָּךְ מִסְתַּפְּקִין מִן הַשְּׁנִיָּה. וְאַחַר כָּךְ מִסְתַּפְּקִין מִן הַשְּׁלִישִׁית. וְתוֹרֵם רִאשׁוֹנָה מִן הַקֻּפָּה הָאַחַת הַגְּדוֹלָה וּמְחַפֶּה אוֹתָהּ הַגְּדוֹלָה בְּמִטְפַּחַת. וְתוֹרֵם הַשְּׁנִיָּה מֵהַקֻּפָּה הַגְּדוֹלָה הַשְּׁנִיָּה וּמְחַפֶּה אוֹתָהּ הַגְּדוֹלָה בְּמִטְפַּחַת. וְתוֹרֵם הַשְּׁלִישִׁית מִן הַקֻּפָּה הַגְּדוֹלָה הַשְּׁלִישִׁית וְאֵינוֹ מְחַפֶּה אוֹתָהּ בְּמִטְפַּחַת כְּדֵי שֶׁתִּהְיֶה נִכֶּרֶת שֶׁבָּהּ סִיֵּם. וְיַתְחִיל מִמֶּנָּה בַּתְּחִלָּה בְּפַעַם שְׁנִיָּה כְּשֶׁיִּכָּנֵס קֹדֶם עֲצֶרֶת. וְתוֹרֵם הָרִאשׁוֹנָה מִן הַגְּדוֹלָה שֶׁהָיְתָה מְגֻלָּה וּמְחַפֶּה אוֹתָהּ. וְתוֹרֵם הַשְּׁנִיָּה מִן הַגְּדוֹלָה שֶׁתָּרַם מִמֶּנָּה בָּרִאשׁוֹנָה תְּחִלָּה וּמְחַפֶּה אוֹתָהּ. וְתוֹרֵם הַשְּׁלִישִׁית מִן הַגְּדוֹלָה הַסְּמוּכָה לָהּ וְאֵינוֹ מְחַפֶּה אוֹתָהּ כְּדֵי שֶׁיַּתְחִיל מִמֶּנָּה בַּתְּחִלָּה בְּפַעַם שְׁלִישִׁית כְּשֶׁיִּכָּנֵס בְּתִשְׁרֵי. עַד שֶׁנִּמְצָא תּוֹרֵם רִאשׁוֹנָה וּשְׁנִיָּה וּשְׁלִישִׁית הַקְּטַנּוֹת מִכָּל אַחַת וְאַחַת מִן הַגְּדוֹלוֹת:
📅 מחזור שנתי
שלוש פעמים בשנה נלקחה התרומה מהלשכה: ראש חודש ניסן, לפני/אחרי ראש השנה, וחמישה עשר יום לפני שבועות. בכל פעם מולאו שלושה כלים קטנים (שלושה סאים כל אחד) מתוך הגדולים ונעשה בהם שימוש עד שנגמרו. הכלים הקטנים סומנו אלף, בית, גימל, והכלי הגדול האחרון שממנו נלקח הכסף הושאר בלא כיסוי כסימן להתחיל ממנו בפעם הבאה.

הלכה ח׳
כְּשֶׁהוּא תּוֹרֵם שָׁלֹשׁ קֻפּוֹת אֵלּוּ תּוֹרֵם אֶת הָרִאשׁוֹנָה לְשֵׁם אֶרֶץ יִשְׂרָאֵל. וְהַשְּׁנִיָּה לְשֵׁם הַכְּרַכִּין הַמֻּקָּפִין לָהּ וּלְשֵׁם כָּל (אֶרֶץ) יִשְׂרָאֵל. וְהַשְּׁלִישִׁית לְשֵׁם בָּבֶל וּלְשֵׁם מָדַי וּלְשֵׁם הַמְּדִינוֹת הָרְחוֹקוֹת וּלְשֵׁם שְׁאָר כָּל יִשְׂרָאֵל:
הלכה ט׳
כְּשֶׁהוּא תּוֹרֵם מִתְכַּוֵּן לִתְרֹם עַל הַגָּבוּי שֶׁיֵּשׁ בַּלִּשְׁכָּה. וְעַל הַגָּבוּי שֶׁעֲדַיִן לֹא הִגִּיעַ לַלִּשְׁכָּה. וְעַל הֶעָתִיד לִגָּבוֹת כְּדֵי שֶׁיִּהְיוּ אֵלּוּ הַשְּׁקָלִים שֶׁהוֹצִיאָן לְהִסְתַּפֵּק מֵהֶן כַּפָּרָה עַל כָּל יִשְׂרָאֵל וּכְאִלּוּ הִגִּיעוּ כָּל שִׁקְלֵיהֶן לַלִּשְׁכָּה וְנִתְרְמָה מֵהֶן תְּרוּמָה זוֹ:
🌍 שותפות כל ישראל
הכלי הראשון נלקח בשם ארץ ישראל, השני עבור הסביבה וכל ישראל, והשלישי עבור בבל, מדי והמחוזות הרחוקים. כוונת התרומה כוללת את מה שכבר נמצא בלשכה, את מה שנגבה אך טרם הגיע, ואת מה שייגבה בעתיד — כדי שהשקלים יכפרו על כל ישראל.
7/9

כללים לממרים את התרומה — מניעת חשד

הלכה י׳
הלכה י׳
כְּשֶׁיִּכָּנֵס הַתּוֹרֵם לִתְרֹם לֹא יִכָּנֵס בְּבֶגֶד שֶׁאֶפְשָׁר לְהַחְבּוֹת בּוֹ כֶּסֶף וְלֹא בְּמִנְעָל וְלֹא בְּסַנְדָּל וְלֹא בִּתְפִלָּה וְלֹא בְּקָמֵעַ שֶׁמָּא יַחְשְׁדוּ אוֹתוֹ הָעָם וְיֹאמְרוּ הֶחְבִּיא מִמְּעוֹת הַלִּשְׁכָּה תַּחְתָּיו כְּשֶׁתְּרָמָהּ. וּמְדַבְּרִים הָיוּ עִמּוֹ מִשָּׁעָה שֶׁיִּכָּנֵס עַד שָׁעָה שֶׁיֵּצֵא כְּדֵי שֶׁלֹּא יִתֵּן לְתוֹךְ פִּיו. וְאַף עַל פִּי שֶׁנִּזְהָרִים כָּל כָּךְ עָנִי אוֹ מִי שֶׁהוּא נִבְהַל לַהוֹן לֹא יִתְרֹם מִפְּנֵי הַחֲשָׁד שֶׁנֶּאֱמַר (במדבר לב כב) "וִהְיִיתֶם נְקִיִּים מֵה' וּמִיִּשְׂרָאֵל":
🔒 מניעת חשד
מי שנכנס לתרום את התרומה אינו רשאי ללבוש בגדים שניתן להסתיר בהם מטבעות, ואינו יכול לנעול נעליים, סנדלים, להניח תפילין או לשאת קמיע — כדי למנוע חשד. אחרים מדברים איתו ללא הפסקה מכניסתו עד יציאתו כדי שלא יוכל לשים מטבע בפיו. גם עם כל האמצעים — עני או מי שנפשו חשקה בממון אינו רשאי לתרום את התרומה, כמו שנאמר: 'והייתם נקיים מה׳ ומישראל.'
8/9

הכוונה כוללת את כל ישראל
מי שתורם מהלשכה מתכוון לכלול אפילו שקלים עתידיים שעוד לא נגבו, כדי שכל ישראל ישתתף בכפרת הקרבנות הציבוריים.

נקיים מה׳ ומישראל
