כמה נרות? רמות הידור במצווה

הלכות א׳–ד׳
הלכה א׳
כַּמָּה נֵרוֹת הוּא מַדְלִיק בַּחֲנֻכָּה. מִצְוָתָהּ שֶׁיִּהְיֶה כָּל בַּיִת וּבַיִת מַדְלִיק נֵר אֶחָד בֵּין שֶׁהָיוּ אַנְשֵׁי הַבַּיִת מְרֻבִּין בֵּין שֶׁלֹּא הָיָה בּוֹ אֶלָּא אָדָם אֶחָד. וְהַמְהַדֵּר אֶת הַמִּצְוָה מַדְלִיק נֵרוֹת כְּמִנְיַן אַנְשֵׁי הַבַּיִת נֵר לְכָל אֶחָד וְאֶחָד בֵּין אֲנָשִׁים בֵּין נָשִׁים. וְהַמְהַדֵּר יוֹתֵר עַל זֶה וְעוֹשֶׂה מִצְוָה מִן הַמֻּבְחָר מַדְלִיק נֵר לְכָל אֶחָד בַּלַּיְלָה הָרִאשׁוֹן וּמוֹסִיף וְהוֹלֵךְ בְּכָל לַיְלָה וְלַיְלָה נֵר אֶחָד:
הלכה ב׳
כֵּיצַד. הֲרֵי שֶׁהָיוּ אַנְשֵׁי הַבַּיִת עֲשָׂרָה. בַּלַּיְלָה הָרִאשׁוֹן מַדְלִיק עֲשָׂרָה נֵרוֹת וּבְלֵיל שֵׁנִי עֶשְׂרִים וּבְלֵיל שְׁלִישִׁי שְׁלֹשִׁים עַד שֶׁנִּמְצָא מַדְלִיק בְּלֵיל שְׁמִינִי שְׁמוֹנִים נֵרוֹת:
הלכה ג׳
מִנְהָג פָּשׁוּט בְּכָל עָרֵינוּ בִּסְפָרַד שֶׁיִּהְיוּ כָּל אַנְשֵׁי הַבַּיִת מַדְלִיקִין נֵר אֶחָד בַּלַּיְלָה הָרִאשׁוֹן וּמוֹסִיפִין וְהוֹלְכִין נֵר בְּכָל לַיְלָה עַד שֶׁנִּמְצָא מַדְלִיק בְּלֵיל שְׁמִינִי שְׁמוֹנָה נֵרוֹת בֵּין שֶׁהָיוּ אַנְשֵׁי הַבַּיִת מְרֻבִּים בֵּין שֶׁהָיָה אָדָם אֶחָד:
הלכה ד׳
נֵר שֶׁיֵּשׁ לוֹ שְׁתֵּי פִּיּוֹת עוֹלֶה לִשְׁנֵי בְּנֵי אָדָם. מִלֵּא קְעָרָה שֶׁמֶן וְהִקִּיפָהּ פְּתִילוֹת. אִם כָּפָה עָלֶיהָ כְּלִי כָּל פְּתִילָה וּפְתִילָה נֶחְשֶׁבֶת כְּנֵר אֶחָד. לֹא כָּפָה עָלֶיהָ כְּלִי נַעֲשֵׂית כִּמְדוּרָה וַאֲפִלּוּ כְּנֵר אֶחָד אֵינָהּ נֶחְשֶׁבֶת:
🕎 שלוש רמות קיום המצווה
החיוב המינימלי הוא נר אחד לכל בית בכל לילה; המהדרין מדליקים נר אחד לכל אדם; והמהדרין מן המהדרין מוסיפים נר אחד בכל לילה נוסף (הלכה א׳). המנהג הספרדי שנהגו קהילות הרמב״ם הוא לפי הרמה השלישית עבור הבית כיחידה—נר אחד בלילה הראשון, עד שמונה בלילה השמיני—ללא קשר לגודל הבית (הלכה ג׳). מנורה עם שני פיות ושני פתילות עולה לחשבון שני אנשים (הלכה ד׳).
4/9

מתי מדליקים ובמה מדליקים

הלכות ה׳–ו׳
הלכה ה׳
אֵין מַדְלִיקִין נֵרוֹת חֲנֻכָּה קֹדֶם שֶׁתִּשְׁקַע הַחַמָּה אֶלָּא עִם שְׁקִיעָתָהּ לֹא מְאַחֲרִין וְלֹא מַקְדִּימִין. שָׁכַח אוֹ הֵזִיד וְלֹא הִדְלִיק עִם שְׁקִיעַת הַחַמָּה מַדְלִיק וְהוֹלֵךְ עַד שֶׁתִּכְלֶה רֶגֶל מִן הַשּׁוּק. וְכַמָּה הוּא זְמַן זֶה כְּמוֹ חֲצִי שָׁעָה אוֹ יֶתֶר. עָבַר זְמַן זֶה אֵינוֹ מַדְלִיק. וְצָרִיךְ שֶׁיִּתֵּן שֶׁמֶן בַּנֵּר כְּדֵי שֶׁתִּהְיֶה דּוֹלֶקֶת וְהוֹלֶכֶת עַד שֶׁתִּכְלֶה רֶגֶל מִן הַשּׁוּק. הִדְלִיקָהּ וְכָבְתָה אֵינוֹ זָקוּק לְהַדְלִיקָהּ פַּעַם אַחֶרֶת. נִשְׁאֲרָה דּוֹלֶקֶת אַחַר שֶׁכָּלְתָה רֶגֶל מִן הַשּׁוּק אִם רָצָה לְכַבּוֹתָהּ אוֹ לְסַלְּקָהּ עוֹשֶׂה:
הלכה ו׳
כָּל הַשְּׁמָנִים וְכָל הַפְּתִילוֹת כְּשֵׁרוֹת לְנֵר חֲנֻכָּה וְאַף עַל פִּי שֶׁאֵין הַשְּׁמָנִים נִמְשָׁכִים אַחַר הַפְּתִילָה וְאֵין הָאוּר נִתְלֵית יָפֶה בְּאוֹתָן הַפְּתִילוֹת. וַאֲפִלּוּ בְּלֵילֵי שַׁבָּת שֶׁבְּתוֹךְ יְמֵי חֲנֻכָּה מֻתָּר לְהַדְלִיק הַשְּׁמָנִים וְהַפְּתִילוֹת שֶׁאָסוּר לְהַדְלִיק בָּהֶן נֵר שַׁבָּת. לְפִי שֶׁאָסוּר לְהִשְׁתַּמֵּשׁ לְנֵר חֲנֻכָּה בֵּין בְּשַׁבָּת בֵּין בְּחל וַאֲפִלּוּ לִבְדֹּק מָעוֹת אוֹ לִמְנוֹתָן לְאוֹרָהּ אָסוּר:
⏰ זמן הדלקה וחומרים
אסור להדליק נרות חנוכה לפני השקיעה והם חייבים לדלוק לפחות שלושים דקות תוך הלילה כדי לקיים את מצוות פרסום הנס (הלכה ה׳). בניגוד לנרות שבת, כל השמנים וכל הפתילות כשרים—אפילו אלה שאינם שואבים יפה—שכן מטרת המצווה היא פרסומי ניסא, ולא איכות האור (הלכה ו׳).

היכן מניחים ומי חייב להדליק

הלכות ז׳–י׳
הלכה ז׳
נֵר חֲנֻכָּה מִצְוָה לְהַנִּיחוֹ עַל פֶּתַח בֵּיתוֹ מִבַּחוּץ בַּטֶּפַח הַסָּמוּךְ לַפֶּתַח עַל שְׂמֹאל הַנִּכְנָס לַבַּיִת כְּדֵי שֶׁתִּהְיֶה מְזוּזָה מִיָּמִין וְנֵר חֲנֻכָּה מִשְּׂמֹאל. וְאִם הָיָה דָּר בַּעֲלִיָּה מַנִּיחוֹ בַּחַלּוֹן הַסְּמוּכָה לִרְשׁוּת הָרַבִּים. וְנֵר חֲנֻכָּה שֶׁהִנִּיחוֹ לְמַעְלָה מֵעֶשְׂרִים אַמָּה לֹא עָשָׂה כְּלוּם לְפִי שֶׁאֵינוֹ נִכָּר:
הלכה ח׳
בִּימֵי הַסַּכָּנָה מַנִּיחַ אָדָם נֵר חֲנֻכָּה בְּתוֹךְ בֵּיתוֹ מִבִּפְנִים וַאֲפִלּוּ הִנִּיחוֹ עַל שֻׁלְחָנוֹ דַּיּוֹ. וְצָרִיךְ לִהְיוֹת בְּתוֹךְ הַבַּיִת נֵר אַחֵר לְהִשְׁתַּמֵּשׁ לְאוֹרוֹ. וְאִם הָיְתָה שָׁם מְדוּרָה אֵינוֹ צָרִיךְ נֵר אַחֵר. וְאִם אָדָם חָשׁוּב הוּא שֶׁאֵין דַּרְכּוֹ לְהִשְׁתַּמֵּשׁ לִמְדוּרָה צָרִיךְ נֵר אַחֵר:
הלכה ט׳
נֵר חֲנֻכָּה שֶׁהִדְלִיקוֹ חֵרֵשׁ שׁוֹטֶה וְקָטָן אוֹ עַכּוּ"ם לֹא עָשָׂה כְּלוּם עַד שֶׁיַּדְלִיקֶנּוּ מִי שֶׁהוּא חַיָּב בַּהַדְלָקָה. הִדְלִיקוֹ מִבִּפְנִים וְהוֹצִיאוֹ דָּלוּק וְהִנִּיחוֹ עַל פֶּתַח בֵּיתוֹ לֹא עָשָׂה כְּלוּם עַד שֶׁיַּדְלִיקֶנּוּ בִּמְקוֹמוֹ. אָחַז הַנֵּר בְּיָדוֹ וְעָמַד לֹא עָשָׂה כְּלוּם שֶׁהָרוֹאֶה אוֹמֵר לְצָרְכּוֹ הוּא עוֹמֵד. עֲשָׁשִׁית שֶׁהָיְתָה דּוֹלֶקֶת כָּל הַיּוֹם כֻּלּוֹ לְמוֹצָאֵי שַׁבָּת מְכַבֶּה וּמְבָרֵךְ וּמַדְלִיקָהּ שֶׁהַהַדְלָקָה הִיא הַמִּצְוָה וְלֹא הַהַנָּחָה. וּמֻתָּר לְהַדְלִיק נֵר חֲנֻכָּה מִנֵּר חֲנֻכָּה:
הלכה י׳
חָצֵר שֶׁיֵּשׁ לָהּ שְׁנֵי פְּתָחִים בִּשְׁתֵּי רוּחוֹת צְרִיכָה שְׁתֵּי נֵרוֹת. שֶׁמָּא יֹאמְרוּ הָעוֹבְרִים בְּרוּחַ זוֹ לֹא הִנִּיחַ נֵר חֲנֻכָּה. אֲבָל אִם הָיוּ בְּרוּחַ אַחַת מַדְלִיק בְּאֶחָד מֵהֶן:
🏠 מיקום והדלקה כשרה
יש להניח את המנורה מחוץ לפתח בצד שמאל, בתוך עשרה טפחים מהקרקע ולא גבוה מעשרים אמה, כדי שעוברי אורח יראו אותה (הלכה ז׳); בעת סכנה מותר להניחה בפנים (הלכה ח׳). הדלקה על ידי חרש, שוטה, קטן או גוי פסולה—חייב להדליקה מבוגר שהוא בר חיוב במקומה הסופי, ואסור להזיזה לאחר ההדלקה (הלכה ט׳). חצר שיש לה שני פתחים בכיוונים שונים טעונה מנורה בכל פתח (הלכה י׳).

אורחים וחביבות המצווה

הלכות י״א–י״ב
הלכה י״א׳
אוֹרֵחַ שֶׁמַּדְלִיקִין עָלָיו בְּתוֹךְ בֵּיתוֹ אֵינוֹ צָרִיךְ לְהַדְלִיק עָלָיו בְּמָקוֹם שֶׁנִּתְאָרֵחַ בּוֹ. אֵין לוֹ בַּיִת לְהַדְלִיק עָלָיו בּוֹ צָרִיךְ לְהַדְלִיק בַּמָּקוֹם שֶׁנִּתְאָרֵחַ בּוֹ. וּמִשְׁתַּתֵּף עִמָּהֶן בַּשֶּׁמֶן. וְאִם הָיָה לוֹ בַּיִת בִּפְנֵי עַצְמוֹ אַף עַל פִּי שֶׁמַּדְלִיקִין עָלָיו בְּתוֹךְ בֵּיתוֹ צָרִיךְ לְהַדְלִיק בַּבַּיִת שֶׁהוּא בּוֹ מִפְּנֵי הָעוֹבְרִין:
הלכה י״ב׳
מִצְוַת נֵר חֲנֻכָּה מִצְוָה חֲבִיבָה הִיא עַד מְאֹד וְצָרִיךְ אָדָם לְהִזָּהֵר בָּהּ כְּדֵי לְהוֹדִיעַ הַנֵּס וּלְהוֹסִיף בְּשֶׁבַח הָאֵל וְהוֹדָיָה לוֹ עַל הַנִּסִּים שֶׁעָשָׂה לָנוּ. אֲפִלּוּ אֵין לוֹ מַה יֹּאכַל אֶלָּא מִן הַצְּדָקָה שׁוֹאֵל אוֹ מוֹכֵר כְּסוּתוֹ וְלוֹקֵחַ שֶׁמֶן וְנֵרוֹת וּמַדְלִיק:
🌟 מי חייב ולמה זה חשוב
אורח שמשפחתו מדליקה עבורו בביתו אינו צריך להדליק במקום הלינה שלו; אם אין מי שמדליק עבורו, עליו להדליק בעצמו או לתרום כסף למארחו (הלכה י״א). המצווה מתוארת כחביבה ביותר: אפילו מי שחי מהצדקה חייב למשכן או למכור בגדים כדי לקנות שמן ומנורות, מפני שחובת פרסום נסי ה׳ עולה על הנוחות האישית (הלכה י״ב).

חיובים מתחרים: סדרי עדיפויות בהוצאות

הלכות י״ג–י״ד
הלכה י״ג׳
הֲרֵי שֶׁאֵין לוֹ אֶלָּא פְּרוּטָה אַחַת וּלְפָנָיו קִדּוּשׁ הַיּוֹם וְהַדְלָקַת נֵר חֲנֻכָּה מַקְדִּים לִקְנוֹת שֶׁמֶן לְהַדְלִיק נֵר חֲנֻכָּה עַל הַיַּיִן לְקִדּוּשׁ הַיּוֹם. הוֹאִיל וּשְׁנֵיהֶם מִדִּבְרֵי סוֹפְרִים מוּטָב לְהַקְדִּים נֵר חֲנֻכָּה שֶׁיֵּשׁ בּוֹ זִכְרוֹן הַנֵּס:
הלכה י״ד׳
הָיָה לְפָנָיו נֵר בֵּיתוֹ וְנֵר חֲנֻכָּה אוֹ נֵר בֵּיתוֹ וְקִדּוּשׁ הַיּוֹם נֵר בֵּיתוֹ קוֹדֵם מִשּׁוּם שְׁלוֹם בֵּיתוֹ שֶׁהֲרֵי הַשֵּׁם נִמְחָק לַעֲשׂוֹת שָׁלוֹם בֵּין אִישׁ לְאִשְׁתּוֹ. גָּדוֹל הַשָּׁלוֹם שֶׁכָּל הַתּוֹרָה נִתְּנָה לַעֲשׂוֹת שָׁלוֹם בָּעוֹלָם שֶׁנֶּאֱמַר (משלי ג יז) "דְּרָכֶיהָ דַרְכֵי נֹעַם וְכָל נְתִיבֹתֶיהָ שָׁלוֹם":
⚖️ חנוכה מול שבת: מה קודם?
כשיש לאדם רק פרוטה אחת ועליו לבחור בין שמן לחנוכה ליין לקידוש, שמן חנוכה קודם כי ניתן לקדש על הלחם—ואילו לפרסומי ניסא אין תחליף (הלכה י״ג). אך אם הוא נאלץ לבחור בין נרות שבת ונר חנוכה, נרות שבת גוברים מפני שהם מביאים שלום בית—שלום בבית—וזה חשוב מכל (הלכה י״ד).
8/9

🕎
פרסומי ניסא הוא הבסיס
כל דין בהדלקת חנוכה—הזמן, המיקום ומשך ההדלקה המינימלי—נובע מציווי אחד: לפרסם את הנס ברבים.
⚖️
מצוות מתחרות מסודרות בהיררכיה ברורה
כשחיובים מתנגשים, שמן חנוכה גובר על יין קידוש, אך נרות שבת גוברים על נר חנוכה—כל אחד מדורג לפי תרומתו הייחודית לקדושה או לשלום הבית.
🏠
המקום והאדם שניהם חלק מהמצווה
את המנורה חייב להדליק מבוגר בר חיוב ולהניחה במקומה הסופי לפני ההדלקה—הזזת נר דולק או הדלקה בלי הנחה במקום—שניהם פוסלים את המצווה.
✨
הידור מצווה מומלץ אך לא כפוי
מערכת שלוש הרמות—מינימום, מהדרין ומהדרין מן המהדרין—מזמינה לקיום הולך ומשתפר, תוך כדי הבטחה שאף אחד, אפילו העני, אינו פטור מהמצווה הבסיסית.
