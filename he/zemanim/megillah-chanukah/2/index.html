הלכות א׳–ו׳
הלכה א׳
הַקוֹרֵא אֶת הַמְּגִלָּה לְמַפְרֵעַ לֹא יָצָא. קָרָא וְשָׁכַח פָּסוּק אֶחָד וְקָרָא פָּסוּק שֵׁנִי לוֹ וְחָזַר וְקָרָא פָּסוּק שֶׁשָּׁכַח וְחָזַר וְקָרָא פָּסוּק שְׁלִישִׁי לֹא יָצָא מִפְּנֵי שֶׁקָּרָא פָּסוּק אֶחָד לְמַפְרֵעַ. אֶלָּא כֵּיצַד עוֹשֶׂה. מַתְחִיל מִפָּסוּק שֵׁנִי שֶׁשָּׁכַח וְקוֹרֵא עַל הַסֵּדֶר:
הלכה ב׳
מָצָא צִבּוּר שֶׁקָּרְאוּ חֶצְיָהּ לֹא יֹאמַר אֶקְרָא חֶצְיָהּ הָאַחֲרוֹן עִם הַצִּבּוּר וְאֶחֱזֹר וְאֶקְרָא חֶצְיָהּ רִאשׁוֹן, שֶׁזֶּה קוֹרֵא לְמַפְרֵעַ. אֶלָּא קוֹרֵא מִתְּחִלָּה וְעַד סוֹף עַל הַסֵּדֶר. קָרָא וְשָׁהָה מְעַט וְחָזַר וְקָרָא אַף עַל פִּי שֶׁשָּׁהָה כְּדֵי לִגְמֹר אֶת כֻּלָּהּ הוֹאִיל וְקָרָא עַל הַסֵּדֶר יָצָא:
הלכה ג׳
הַקּוֹרֵא אֶת הַמְּגִלָּה עַל פֶּה לֹא יָצָא יְדֵי חוֹבָתוֹ. הַלּוֹעֵז שֶׁשָּׁמַע אֶת הַמְּגִלָּה הַכְּתוּבָה בִּלְשׁוֹן הַקֹּדֶשׁ וּבִכְתַב הַקֹּדֶשׁ אַף עַל פִּי שֶׁאֵינוֹ יוֹדֵעַ מַה הֵן אוֹמְרִין יָצָא יְדֵי חוֹבָתוֹ. וְכֵן אִם הָיְתָה כְּתוּבָה יְוָנִית וּשְׁמָעָהּ יָצָא אַף עַל פִּי שֶׁאֵינוֹ מַכִּיר, וַאֲפִלּוּ הָיָה הַשּׁוֹמֵעַ עִבְרִי:
הלכה ד׳
הָיְתָה כְּתוּבָה תַּרְגּוּם אוֹ בְּלָשׁוֹן אַחֶרֶת מִלְּשׁוֹנוֹת הַגּוֹיִם לֹא יָצָא יְדֵי חוֹבָתוֹ בִּקְרִיאָתָהּ אֶלָּא הַמַּכִּיר אוֹתָהּ הַלָּשׁוֹן בִּלְבַד. וְהוּא שֶׁתִּהְיֶה כְּתוּבָה בִּכְתַב אוֹתוֹ הַלָּשׁוֹן. אֲבָל אִם הָיְתָה כְּתוּבָה בִּכְתָב עִבְרִי וּקְרָאָהּ אֲרָמִית לַאֲרָמִי לֹא יָצָא שֶׁנִּמְצָא זֶה קוֹרֵא עַל פֶּה. וְכֵיוָן שֶׁלֹּא יָצָא יְדֵי חוֹבָתוֹ הַקּוֹרֵא לֹא יָצָא הַשּׁוֹמֵעַ מִמֶּנּוּ:
הלכה ה׳
הַקּוֹרֵא אֶת הַמְּגִלָּה בְּלֹא כַּוָּנָה לֹא יָצָא. כֵּיצַד. הָיָה כּוֹתְבָהּ אוֹ דּוֹרְשָׁהּ אוֹ מַגִּיהָהּ אִם כִּוֵּן אֶת לִבּוֹ לָצֵאת בִּקְרִיאָה זוֹ יָצָא וְאִם לֹא כִּוִּן לִבּוֹ לֹא יָצָא. קָרָא וְהוּא מִתְנַמְנֵם הוֹאִיל וְלֹא נִרְדַּם בַּשֵּׁנָה יָצָא:
הלכה ו׳
בַּמֶּה דְּבָרִים אֲמוּרִים שֶׁהַמְכַוֵּן לִבּוֹ בַּכְּתִיבָה יָצָא. בְּשֶׁנִּתְכַּוֵּן לָצֵאת בַּקְּרִיאָה שֶׁקּוֹרֵא בְּסֵפֶר שֶׁמַּעְתִּיק מִמֶּנּוּ בְּשָׁעָה שֶׁהוּא כּוֹתֵב. אֲבָל אִם לֹא נִתְכַּוֵּן לָצֵאת בִּקְרִיאָה זוֹ שֶׁכָּתַב לֹא יָצָא. שֶׁאֵינוֹ יוֹצֵא יְדֵי חוֹבָתוֹ אֶלָּא בִּקְרִיאָתָהּ מִסֵּפֶר שֶׁכֻּלָּהּ כְּתוּבָה בּוֹ בִּשְׁעַת קְרִיאָה:
📜 כלל הסדר
יש לקרוא את המגילה בסדר מהתחלה עד הסוף — מי שדילג על פסוק וחזר אליו אחר כך לא יצא ידי חובתו, וצריך לחזור ולקרוא מהפסוק שדילג (הלכה א'). כמו כן, אם הצטרף לציבור באמצע הקריאה ותכנן להשלים בהמשך — זה לא תקף; צריך לקרוא את כל המגילה ברצף (הלכה ב'). קריאה על פה לא מועילה בשום מצב, ואפילו סופר שמעתיק את המגילה יוצא ידי חובה רק אם כיוון לצאת בקריאת המגילה המקור, לא זו שהוא כותב (הלכות ג', ו').

שפה, כתב וכוונה

הלכות ג׳–ה׳
הלכה ג׳
הַקּוֹרֵא אֶת הַמְּגִלָּה עַל פֶּה לֹא יָצָא יְדֵי חוֹבָתוֹ. הַלּוֹעֵז שֶׁשָּׁמַע אֶת הַמְּגִלָּה הַכְּתוּבָה בִּלְשׁוֹן הַקֹּדֶשׁ וּבִכְתַב הַקֹּדֶשׁ אַף עַל פִּי שֶׁאֵינוֹ יוֹדֵעַ מַה הֵן אוֹמְרִין יָצָא יְדֵי חוֹבָתוֹ. וְכֵן אִם הָיְתָה כְּתוּבָה יְוָנִית וּשְׁמָעָהּ יָצָא אַף עַל פִּי שֶׁאֵינוֹ מַכִּיר, וַאֲפִלּוּ הָיָה הַשּׁוֹמֵעַ עִבְרִי:
הלכה ד׳
הָיְתָה כְּתוּבָה תַּרְגּוּם אוֹ בְּלָשׁוֹן אַחֶרֶת מִלְּשׁוֹנוֹת הַגּוֹיִם לֹא יָצָא יְדֵי חוֹבָתוֹ בִּקְרִיאָתָהּ אֶלָּא הַמַּכִּיר אוֹתָהּ הַלָּשׁוֹן בִּלְבַד. וְהוּא שֶׁתִּהְיֶה כְּתוּבָה בִּכְתַב אוֹתוֹ הַלָּשׁוֹן. אֲבָל אִם הָיְתָה כְּתוּבָה בִּכְתָב עִבְרִי וּקְרָאָהּ אֲרָמִית לַאֲרָמִי לֹא יָצָא שֶׁנִּמְצָא זֶה קוֹרֵא עַל פֶּה. וְכֵיוָן שֶׁלֹּא יָצָא יְדֵי חוֹבָתוֹ הַקּוֹרֵא לֹא יָצָא הַשּׁוֹמֵעַ מִמֶּנּוּ:
הלכה ה׳
הַקּוֹרֵא אֶת הַמְּגִלָּה בְּלֹא כַּוָּנָה לֹא יָצָא. כֵּיצַד. הָיָה כּוֹתְבָהּ אוֹ דּוֹרְשָׁהּ אוֹ מַגִּיהָהּ אִם כִּוֵּן אֶת לִבּוֹ לָצֵאת בִּקְרִיאָה זוֹ יָצָא וְאִם לֹא כִּוִּן לִבּוֹ לֹא יָצָא. קָרָא וְהוּא מִתְנַמְנֵם הוֹאִיל וְלֹא נִרְדַּם בַּשֵּׁנָה יָצָא:
🗣️ שפה וכוונה
מי שאינו דובר עברית ושמע את המגילה הנקראת בכתב עברי יוצא ידי חובתו אפילו בלי להבין, וכן הדין ביוונית (הלכה ג'). מגילה שנכתבה בארמית או בשפת גויים אחרת כשרה רק למי שמבין את אותה שפה, והמגילה חייבת להיות כתובה בכתב של אותה שפה (הלכה ד'). קריאה ללא כוונה — למשל בזמן כתיבה, הסבר או הגהה — לא מועילה, אלא אם כיוון את ליבו לצאת ידי חובה (הלכה ה').

דרישות פיזיות של מגילת הקלף

הלכות ט׳–י״א
הלכה ט׳
אֵין כּוֹתְבִין הַמְּגִלָּה אֶלָּא בִּדְיוֹ עַל הַגְּוִיל אוֹ עַל הַקְּלָף כְּסֵפֶר תּוֹרָה. וְאִם כְּתָבָהּ בְּמֵי עַפְּצָא וְקַנְקַנְתּוֹם כְּשֵׁרָה. כְּתָבָהּ בִּשְׁאָר מִינֵי צִבְעוֹנִין פְּסוּלָה. וּצְרִיכָה שִׂרְטוּט כַּתּוֹרָה עַצְמָהּ. וְאֵין הָעוֹר שֶׁלָּהּ צָרִיךְ לְעַבֵּד לִשְׁמָהּ. הָיְתָה כְּתוּבָה עַל הַנְּיָר אוֹ עַל עוֹר שֶׁאֵינוֹ מְעֻבָּד אוֹ שֶׁכְּתָבָהּ עַכּוּ"ם אוֹ אֶפִּיקוֹרוֹס פְּסוּלָה:
הלכה י׳
הָיוּ בָּהּ אוֹתִיּוֹת מְטֻשְׁטָשׁוֹת אוֹ מְקֹרָעוֹת. אִם רִשּׁוּמָן נִכָּר אֲפִלּוּ הָיוּ רֻבָּהּ כְּשֵׁרָה. וְאִם אֵין רִשּׁוּמָן נִכָּר אִם הָיָה רֻבָּהּ שָׁלֵם כְּשֵׁרָה. וְאִם לָאו פְּסוּלָה וְהַקּוֹרֵא בָּהּ לֹא יָצָא. הִשְׁמִיט בָּהּ הַסּוֹפֵר אוֹתִיּוֹת אוֹ פְּסוּקִים וּקְרָאָן הַקּוֹרֵא אֶת הַמְּגִלָּה עַל פֶּה יָצָא:
הלכה י״א׳
הַמְּגִלָּה צְרִיכָה שֶׁתְּהֵא תְּפוּרָה כֻּלָּהּ עַד שֶׁיִּהְיוּ כָּל עוֹרוֹתֶיהָ מְגִלָּה אַחַת. וְאֵינָהּ נִתְפֶּרֶת אֶלָּא בְּגִידִין כְּסֵפֶר תּוֹרָה. וְאִם תְּפָרָהּ שֶׁלֹּא בְּגִידִין פְּסוּלָה. וְאֵינוֹ צָרִיךְ לִתְפֹּר אֶת כָּל הַיְרִיעָה בְּגִידִין כְּסֵפֶר תּוֹרָה אֶלָּא אֲפִלּוּ תָּפַר בְּגִידִין שָׁלֹשׁ תְּפִירוֹת בִּקְצֵה הַיְרִיעָה וְשָׁלֹשׁ בְּאֶמְצָעָהּ וְשָׁלֹשׁ בַּקָּצֶה הַשֵּׁנִי כְּשֵׁרָה מִפְּנֵי שֶׁנִּקְרֵאת אִגֶּרֶת:
📋 מפרט המגילה
המגילה חייבת להיות כתובה בדיו על גויל או קלף — כמו ספר תורה — עם שורות מסורגלות, ופסולה אם כתבה גוי או כופר (הלכה ט'). אותיות שנמחקו או נקרעו פוסלות רק אם רוב האותיות אבדו ואין שום עקבות; כל עוד ניתן לזהות עקבות או שרוב האותיות שלמות — המגילה כשרה (הלכה י'). דפי הקלף של המגילה חייבים להיות תפורים בגידין, אך במגילה (הנקראת 'אגרת') מספיקות שלוש תפירות בכל קצה ובאמצע, ולא כל התפר המלא הנדרש בספר תורה (הלכה י"א).

מנהגי הקריאה ואיסורי הפורים

הלכות ז׳–י״ג
הלכה ז׳
הַקּוֹרֵא אֶת הַמְּגִלָּה וְטָעָה בִּקְרִיאָתָהּ וְקָרָא קְרִיאָה מְשֻׁבֶּשֶׁת יָצָא. לְפִי שֶׁאֵין מְדַקְדְּקִין בִּקְרִיאָתָהּ. קְרָאָהּ עוֹמֵד אוֹ יוֹשֵׁב יָצָא וַאֲפִלּוּ בְּצִבּוּר. אֲבָל לֹא יִקְרָא בְּצִבּוּר יוֹשֵׁב לְכַתְּחִלָּה מִפְּנֵי כְּבוֹד הַצִּבּוּר. קְרָאוּהָ שְׁנַיִם אֲפִלּוּ עֲשָׂרָה כְּאֶחָד יָצְאוּ הַקּוֹרְאִין וְהַשּׁוֹמְעִים מִן הַקּוֹרְאִין. וְקוֹרֵא אוֹתָהּ גָּדוֹל עִם הַקָּטָן וַאֲפִלּוּ בְּצִבּוּר:
הלכה ח׳
אֵין קוֹרְאִין בְּצִבּוּר בִּמְגִלָּה הַכְּתוּבָה בֵּין הַכְּתוּבִים. וְאִם קָרָא לֹא יָצָא. אֶלָּא אִם כֵּן הָיְתָה יְתֵרָה עַל שְׁאָר הַיְרִיעוֹת אוֹ חֲסֵרָה כְּדֵי שֶׁיְּהֵא לָהּ הֶכֵּר. אֲבָל הַיָּחִיד קוֹרֵא בָּהּ וַאֲפִלּוּ אֵינָהּ חֲסֵרָה וְלֹא יְתֵרָה וְיוֹצֵא בָּהּ יְדֵי חוֹבָתוֹ:
הלכה י״ב׳
וְצָרִיךְ הַקּוֹרֵא לִקְרוֹת עֲשֶׂרֶת בְּנֵי הָמָן וַעֲשֶׂרֶת בִּנְשִׁימָה אַחַת כְּדֵי לְהוֹדִיעַ לְכָל הָעָם שֶׁכֻּלָּם נִתְלוּ וְנֶהֶרְגוּ כְּאֶחָד. וּמִנְהַג כָּל יִשְׂרָאֵל שֶׁהַקּוֹרֵא הַמְּגִלָּה קוֹרֵא וּפוֹשֵׁט כְּאִגֶּרֶת לְהַרְאוֹת הַנֵּס וּכְשֶׁיִּגְמֹר חוֹזֵר וְכוֹרְכָהּ כֻּלָּהּ וּמְבָרֵךְ:
הלכה י״ג׳
שְׁנֵי הַיָּמִים הָאֵלּוּ שֶׁהֵן י"ד וְט"ו אֲסוּרִין בְּהֶסְפֵּד וְתַעֲנִית לְכָל אָדָם בְּכָל מָקוֹם. בֵּין לִבְנֵי כְּרַכִּין שֶׁהֵן עוֹשִׂין ט"ו בִּלְבַד. בֵּין לִבְנֵי עֲיָרוֹת שֶׁהֵן עוֹשִׂין י"ד בִּלְבַד. וּשְׁנֵי הַיָּמִים אֲסוּרִין בְּהֶסְפֵּד וְתַעֲנִית בַּאֲדָר הָרִאשׁוֹן וּבָאֲדָר הַשֵּׁנִי. אַנְשֵׁי כְּפָרִים שֶׁהִקְדִּימוּ וְקָרְאוּ בְּשֵׁנִי אוֹ בַּחֲמִישִׁי הַסָּמוּךְ לְפוּרִים מֻתָּרִים בְּהֶסְפֵּד וְתַעֲנִית בְּיוֹם קְרִיאָתָהּ וַאֲסוּרִין בְּהֶסְפֵּד וְתַעֲנִית בִּשְׁנֵי הַיָּמִים הָאֵלּוּ אַף עַל פִּי שֶׁאֵין קוֹרְאִין בָּהֶן:
🎭 מנהגי הקריאה
מספר אנשים — אפילו עשרה — שקוראים בו-זמנית כולם יוצאים ידי חובתם, ומבוגר יכול לקרוא יחד עם ילד אפילו עבור ציבור; אך אסור לקרוא לציבור כשיושבים מפני הכבוד (הלכה ז'). שמות עשרת בני המן חייבים להיקרא בנשימה אחת, ומנהג ישראל הוא לפרוס את המגילה כמו אגרת כדי להראות את הנס, ולגוללה לגמרי לפני ברכת הסיום (הלכה י"ב). גם י"ד וגם ט"ו באדר אסורים בהספד ותענית לכולם בכל מקום, כולל שני ימי האדרים בשנה מעוברת (הלכה י"ג).
7/9

מצוות פורים: סעודה, מתנות וחגיגה נצחית

הלכות י״ד–י״ח
הלכה י״ד׳
מִצְוַת יוֹם י"ד לִבְנֵי כְּפָרִים וַעֲיָרוֹת וְיוֹם ט"ו לִבְנֵי כְּרַכִּים לִהְיוֹת יוֹם שִׂמְחָה וּמִשְׁתֶּה וּמִשְׁלוֹחַ מָנוֹת לְרֵעִים וּמַתָּנוֹת לָאֶבְיוֹנִים. וּמֻתָּר בַּעֲשִׂיַּת מְלָאכָה, וְאַף עַל פִּי כֵן אֵין רָאוּי לַעֲשׂוֹת בּוֹ מְלָאכָה. אָמְרוּ חֲכָמִים כָּל הָעוֹשֶׂה מְלָאכָה בְּיוֹם פּוּרִים אֵינוֹ רוֹאֶה סִימַן בְּרָכָה לְעוֹלָם. בְּנֵי כְּפָרִים שֶׁקָּדְמוּ וְקָרְאוּ בְּשֵׁנִי אוֹ בַּחֲמִישִׁי אִם חִלְּקוּ מָעוֹת לָאֶבְיוֹנִים בְּיוֹם קְרִיאָתָן יָצְאוּ. אֲבָל הַשִּׂמְחָה וְהַמִּשְׁתֶּה אֵין עוֹשִׂין אוֹתָם אֶלָּא בְּיוֹם י"ד. וְאִם הִקְדִּימוּ לֹא יָצְאוּ. וּסְעֻדַּת פּוּרִים שֶׁעֲשָׂאָהּ בַּלַּיְלָה לֹא יָצָא יְדֵי חוֹבָתוֹ:
הלכה ט״ו׳
כֵּיצַד חוֹבַת סְעֻדָּה זוֹ. שֶׁיֹּאכַל בָּשָׂר וִיתַקֵּן סְעֻדָּה נָאָה כְּפִי אֲשֶׁר תִּמְצָא יָדוֹ. וְשׁוֹתֶה יַיִן עַד שֶׁיִּשְׁתַּכֵּר וְיֵרָדֵם בְּשִׁכְרוּתוֹ. וְכֵן חַיָּב אָדָם לִשְׁלֹחַ שְׁתֵּי מְנוֹת בָּשָׂר אוֹ שְׁנֵי מִינֵי תַּבְשִׁיל אוֹ שְׁנֵי מִינֵי אֳכָלִין לַחֲבֵרוֹ שֶׁנֶּאֱמַר (אסתר ט יט) "וּמִשְׁלוֹחַ מָנוֹת אִישׁ לְרֵעֵהוּ", שְׁתֵּי מָנוֹת לְאִישׁ אֶחָד. וְכָל הַמַּרְבֶּה לִשְׁלֹחַ לְרֵעִים מְשֻׁבָּח. וְאִם אֵין לוֹ מַחֲלִיף עִם חֲבֵרוֹ זֶה שׁוֹלֵחַ לְזֶה סְעֻדָּתוֹ וְזֶה שׁוֹלֵחַ לְזֶה סְעֻדָּתוֹ כְּדֵי לְקַיֵּם וּמִשְׁלוֹחַ מָנוֹת אִישׁ לְרֵעֵהוּ:
הלכה ט״ז׳
וְחַיָּב לְחַלֵּק לָעֲנִיִּים בְּיוֹם הַפּוּרִים. אֵין פּוֹחֲתִין מִשְּׁנֵי עֲנִיִּים נוֹתֵן לְכָל אֶחָד מַתָּנָה אַחַת אוֹ מָעוֹת אוֹ מִינֵי תַּבְשִׁיל אוֹ מִינֵי אֳכָלִין שֶׁנֶּאֱמַר (אסתר ט כב) "וּמַתָּנוֹת לָאֶבְיוֹנִים", שְׁתֵּי מַתָּנוֹת לִשְׁנֵי עֲנִיִּים. וְאֵין מְדַקְדְּקִין בִּמְעוֹת פּוּרִים אֶלָּא כָּל הַפּוֹשֵׁט יָדוֹ לִטּל נוֹתְנִין לוֹ. וְאֵין מְשַׁנִּין מָעוֹת פּוּרִים לִצְדָקָה אַחֶרֶת:
הלכה י״ז׳
מוּטָב לָאָדָם לְהַרְבּוֹת בְּמַתְּנוֹת אֶבְיוֹנִים מִלְּהַרְבּוֹת בִּסְעֻדָּתוֹ וּבְשִׁלּוּחַ מָנוֹת לְרֵעָיו. שֶׁאֵין שָׁם שִׂמְחָה גְּדוֹלָה וּמְפֹאָרָה אֶלָּא לְשַׂמֵּחַ לֵב עֲנִיִּים וִיתוֹמִים וְאַלְמָנוֹת וְגֵרִים. שֶׁהַמְשַׂמֵּחַ לֵב הָאֻמְלָלִים הָאֵלּוּ דּוֹמֶה לַשְּׁכִינָה שֶׁנֶּאֱמַר (ישעיה נז טו) "לְהַחֲיוֹת רוּחַ שְׁפָלִים וּלְהַחֲיוֹת לֵב נִדְכָּאִים":
הלכה י״ח׳
כָּל סִפְרֵי הַנְּבִיאִים וְכָל הַכְּתוּבִים עֲתִידִין לִבָּטֵל לִימוֹת הַמָּשִׁיחַ חוּץ מִמְּגִלַּת אֶסְתֵּר וַהֲרֵי הִיא קַיֶּמֶת כַּחֲמִשָּׁה חֻמְּשֵׁי תּוֹרָה וְכַהֲלָכוֹת שֶׁל תּוֹרָה שֶׁבְּעַל פֶּה שֶׁאֵינָן בְּטֵלִין לְעוֹלָם. וְאַף עַל פִּי שֶׁכָּל זִכְרוֹן הַצָּרוֹת יְבֻטַּל שֶׁנֶּאֱמַר (ישעיה סה טז) "כִּי נִשְׁכְּחוּ הַצָּרוֹת הָרִאשׁוֹנוֹת וְכִי נִסְתְּרוּ מֵעֵינִי". יְמֵי הַפּוּרִים לֹא יִבָּטְלוּ שֶׁנֶּאֱמַר (אסתר ט כח) "וִימֵי הַפּוּרִים הָאֵלֶּה לֹא יַעַבְרוּ מִתּוֹךְ הַיְּהוּדִים וְזִכְרָם לֹא יָסוּף מִזַּרְעָם":
🎁 ארבע מצוות הפורים
סעודת הפורים דורשת אכילת בשר ושתיית יין עד שנרדמים מרוב שתייה, וחייבת להתקיים ביום — סעודה שנערכה בלילה לא מועילה (הלכות י"ד–ט"ו). צריך לשלוח שתי מנות לחבר אחד ולתת מתנה אחת לכל אחד משני עניים, וכספי צדקה לפורים אסור להפנות למטרות אחרות (הלכות ט"ו–ט"ז). מעל הכל, הרמב"ם מדגיש שנתינה נדיבה לעניים היא השמחה הגדולה והמהודרת ביותר, המשולה לשכינה עצמה — וחגיגת המגילה לעולם לא תתבטל, אפילו בימות המשיח (הלכות י"ז–י"ח).

קרא בסדר, ללא הפסקה
יש לקרוא את המגילה ברצף מהתחלה עד הסוף; כל פסוק שנקרא שלא בסדר פוסל את הקריאה, וצריך לחזור ולהתחיל מהמקום שדילגו.
🕯️
כוונה היא הכרחית
קריאת המגילה — בין אם כתיבה, הגהה או שמיעה — מועילה רק כשהקורא מכוון בליבו לצאת ידי חובת המצווה.
⚖️
כשרות פיזית על פי ספר תורה
כמו ספר תורה, המגילה חייבת להיות כתובה בדיו על קלף כשר, מסורגלת, תפורה בגידין וכתובה על ידי יהודי — ובכל זאת ישנן הקלות מסוימות משום שהיא נקראת 'אגרת'.
❤️
שמח את העניים מעל הכל
הרמב"ם פוסק שנתינה נדיבה לעניים, יתומים, אלמנות וגרים עולה על פני סעודה מפוארת או משלוח מנות, ומי שעושה כן דומה לשכינה.
